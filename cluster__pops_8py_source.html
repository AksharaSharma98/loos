37 def read_frame_ranges(filename):
38  filenames = []
39  ends = []
40  with open(filename) as f:
41  # skip the first 2 lines, stop when a line doesn't start with "#"
42  for line in f.readlines()[2:]:
43  if line.startswith("#"):
44  try:
45  hash, index, start, end, filename = line.split()
46  filenames.append(filename)
47  ends.append(int(end))
48  except ValueError:
49  break
50  else:
51  break
52  return (filenames, numpy.array(ends))

55 def usage():
56  string = f"""
57  Usage: {sys.argv[0]} json_file rmsds_file outfile [split]
58  json_file: json output from cluster-kgs
59  rmsds_file: file of distances input to cluster-kgs,
60  presumably generated by rmsds or multi-rmsds
61  outfile: file of cluster populations, in numpy matrix format.
62  The shape of the matrix will be (#clusters, # trajectories)
63  split: optional integer for splitting the trajectories into
64  2 groups. For example, if you have 4 "wild type"
65  trajectories and 4 "mutant" trajectories, then
66  passing a value of 4 would print out the average
67  wild type and mutant populations, as well as the std dev
68  and std error of the populations.
69 
70  """
71  return string

95  if len(filenames):
96  data = numpy.zeros((len(clusters), len(filenames)))
97  else:
98  data = numpy.zeros((len(clusters)))
99 
100  for i in range(len(clusters)):

102  arr = numpy.array(clusters[i])

104  indices -= 1

120  first_err = first_dev / sqrt(split)
