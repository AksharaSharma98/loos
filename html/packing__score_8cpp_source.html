55 string fullHelpMessage(void) {
56  string msg =
57 "\n"
58 "SYNOPSIS\n"
59 "\n"
60 "Compute the packing score, a measure of contact between two selection.\n"
61 "\n"
62 "DESCRIPTION\n"
63 "\n"
64 "This tool computes the packing score, a simple measure of the contact between\n"
65 "two selections, over the course of a trajectory. The packing score is defined\n"
66 "as the sum of the inverse of the pairwise distance raised to the sixth power,\n"
67 "computed over all pairs of atoms in 2 selections. In essence, you can think \n"
68 "of it as the attractive component of the van der Waal's interaction with sigma \n"
69 "and epsilon set to 1.\n"
70 "The packing score was originally defined in Grossfield, A., Feller, S. E., \n"
71 "Pitman, M. C., A role for direct interactions in the modulation of rhodopsin \n"
72 "by omega-3 polyunsaturated lipids, Proc. Nat. Acad. Sci. USA, 2006, 103, \n"
73 "4888-4893\n"
74 "\n"
75 "Required Flags\n"
76 " --sel1 'selection string' : the first selection of atoms\n"
77 " --sel2 'selection string' : the second selection of atoms\n"
78 "\n"
79 "Options\n"
80 "\n"
81 " --skip N : skip the first N frames from the trajectory\n"
82 " --normalize : if specified, divides the packing score by the product of the \n"
83 " number of atoms in selection 1 and in selection 2\n"
84 "\n"
85 "EXAMPLE\n"
86 "\n"
87 "An example command line would be:\n"
88 "\n"
89 "packing_score --skip 20 --normalize --sel1 'resid >= 35 && resid <= 64 && segname == \"RHOD\"' --sel2 'resid>=71 && resid<=100 && segname == \"RHOD\"' rhod_namd.psf rhod_control.dcd\n"
90 "\n"
91 "which would compute the packing score between two chunks of the segment \n"
92 "\"RHOD\", one residues 35-64 and the other 71-100, normalizing the value,\n"
93 "and skipping the first 20 frames.\n"
94 "\n"
95 "HINTS\n"
96 "-- One trick to speed things up is to add \"&& !hydrogen\" to your \n"
97 " selections. The answers with and without the hydrogens should be\n"
98 " almost perfectly proportional, but you'll greatly reduce the number\n"
99 " of distance calculations\n"
100 "\n"
101 "-- The program verifies that the two selections don't have any atoms in \n"
102 " common, and quits immediately if they do, since then the packing score\n"
103 " would be infinite.\n"
104 "\n"
105 "-- The normalize option could be useful for identifying the pieces of an\n"
106 " interface that pack tightly in a size-independent manner. Otherwise,\n"
107 " large chunks (e.g. tryptophans vs. alanines) will tend to have higher \n"
108 " scores just because they have more atoms.\n"
109 "\n";
110  return(msg);
111 
112 }

152 int main(int argc, char *argv[]) {
153 
154  string header = invocationHeader(argc, argv);
155 
156  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
157  opts::BasicTrajectory* tropts = new opts::BasicTrajectory;
158  ToolOptions* topts = new ToolOptions;
159  opts::AggregateOptions options;
160  options.add(bopts).add(tropts).add(topts);
161 
162  // Parse the command-line. If an error occurred, help will already
163  // be displayed and it will return a FALSE value.
164  if (!options.parse(argc, argv))
165  exit(-1);
166 
167  cout << "# " << header << endl;
168 
169  AtomicGroup model = tropts->model;
170  pTraj traj = tropts->trajectory;
171  AtomicGroup set1 = selectAtoms(model, sel1);
172  AtomicGroup set2 = selectAtoms(model, sel2);
173 
174  // Check that the two groups don't overlap, since you'll get
175  // a NaN packing score.
176  AtomicGroup overlap = set1.intersect(set2);
177  if (overlap.size()) {
178  cerr << "Error: the two selections have the following atoms in common:"
179  << endl;
180  for (uint i=0; i < overlap.size(); i++) {
181  cerr << *(overlap[i]) << endl;
182  }
183  exit(1);
184  }
185 
186  uint num_pairs = 0;
187  if (normalize) {
188  num_pairs = set1.size() * set2.size();
189  }
190 
191 
192  // Now iterate over all frames in the trajectory (excluding the skip
193  // region)
194  uint frame = tropts->skip;
195  while (traj->readFrame()) {
196  traj->updateGroupCoords(tropts->model);
197 
198  double score = 0.0;
199 
200  for (uint i=0; i<set1.size(); i++) {
201  for (uint j=0; j<set2.size(); j++) {
202  GCoord diff = set1[i]->coords() - set2[j]->coords();
203  double dist2 = diff.length2();
204  double dist6 = dist2*dist2*dist2;
205  score += 1.0/dist6;
206  }
207  }
208 
209  if (normalize) {
210  score /= num_pairs;
211  }
212 
213  cout << frame << "\t" << score << endl;
214  frame++;
215  }
216 
217 }
