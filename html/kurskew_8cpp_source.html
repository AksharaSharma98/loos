38 double mean(const DoubleMatrix& M, const uint i) {
39 
40  double m = 0.0;
41  for (uint j=0; j<M.rows(); ++j)
42  m += M(j, i);
43  return(m / M.rows());
44 }

47 double stddev(const DoubleMatrix& M, const uint i, const double m) {
48 
49  double s = 0.0;
50  for (uint j=0;j<M.rows(); ++j) {
51  double d = M(j, i) - m;
52  s += d*d;
53  }
54 
55  s /= (M.rows() - 1);
56  return(sqrt(s));
57 }

60 double moment(const DoubleMatrix& M, const uint i, const double m, const double s, const double p) {
61 
62  double x = 0.0;
63  for (uint j=0; j<M.rows(); ++j) {
64  double d = (M(j, i) - m) / s;
65  x += pow(d, p);
66  }
67  return(x / M.rows());
68 }

105 
106  DoubleMatrix M;
107  readAsciiMatrix(argv[1], M);
108 
109  DoubleMatrix K(M.cols(), 3);
110 
111  for (uint i=0; i<M.cols(); ++i) {
112  double avg = mean(M, i);
113  double dev = stddev(M, i, avg);
114 
115  K(i, 0) = i;
116  K(i, 1) = moment(M, i, avg, dev, 3);
117  K(i, 2) = moment(M, i, avg, dev, 4) - 3.0;
118  }
119 
120  writeAsciiMatrix(cout, K, hdr);
