38  std::vector<double> gaussian1d(const int w, const double sigma) {
39 
40  double a = 1.0 / sqrt(2.0 * M_PI * sigma);
41  double b = -1.0 / (2.0 * sigma);
42 
43  std::vector<double> kernel;
44  for (int i=0; i<=w; ++i) {
45  double x = 2.0*i/w-1.0;

47  }
