820  // Find all atoms in the current group that are within dist
821  // angstroms of any atom in the passed group. The distance
822  // calculation is determined by the passed functor so that the
823  // same code can be used for both periodic and non-periodic
824  // coordinates.
825 
826  template <typename DistanceCalc>
827  AtomicGroup within_private(const double dist, AtomicGroup& grp, const DistanceCalc& distance_functor) const {
828 
829  AtomicGroup res;
830  res.box = box;
831 
832  double dist2 = dist * dist;
833  std::vector<uint> indices;
834 
835  for (uint j=0; j<size(); j++) {
836  GCoord c = atoms[j]->coords();
837  for (uint i=0; i<grp.size(); i++) {
838  if (distance_functor(c, grp.atoms[i]->coords()) <= dist2) {
839  indices.push_back(j);
840  break;
841  }
842  }
843  }
844 
845  if (indices.size() == 0)
846  return(res);
847 
848  for (std::vector<uint>::const_iterator ci = indices.begin(); ci != indices.end(); ++ci)
849  res.addAtom(atoms[*ci]);
850 
851  return(res);
852  }

855  template<typename DistanceCalc>
856  bool contactwith_private(const double dist, const AtomicGroup& grp, const uint min_contacts, const DistanceCalc& distance_function) const {
857  double dist2 = dist * dist;
858  uint ncontacts = 0;
859 
860  for (uint j = 0; j<size(); ++j) {
861  GCoord c = atoms[j]->coords();
862  for (uint i = 0; i<grp.size(); ++i)
863  if (distance_function(c, grp.atoms[i]->coords()) <= dist2)
864  if (++ncontacts >= min_contacts)
865  return(true);
866  }
867  return(false);
868  }

875  template<typename DistanceCalc>
876  void findBondsImpl(const double dist, const DistanceCalc& distance_function) {
877  iterator ij;
878  double dist2 = dist * dist;
879  double current_dist2;
880 
881  for (ij = begin(); ij != end() - 1; ++ij) {
882  iterator ii;
883  GCoord u = (*ij)->coords();
884 
885  for (ii = ij + 1; ii != end(); ++ii) {
886  current_dist2 = distance_function(u, (*ii)->coords());
887  if (current_dist2 < dist2) {
888  (*ij)->addBond(*ii);
889  (*ii)->addBond(*ij);
890  }
891  }
892  }
893  }
