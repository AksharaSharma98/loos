50 struct DistanceCalculation {
51  virtual double operator()(const AtomicGroup&, const AtomicGroup&) = 0;
52  virtual ~DistanceCalculation() { }
53 
54  void usePeriodicity(const bool flag) { _use_periodicity = flag; }
55  double distance(const GCoord& u, const GCoord& v) {
56  if (_use_periodicity)
57  return(u.distance(v, _box));
58  return(u.distance(v));
59  }
60 
61  void setBox(const GCoord& v) { _box = v; }
62 
63  bool _use_periodicity;
64  GCoord _box;
65 };

91 struct CenterDistanceZ : public DistanceCalculation {
92  double operator()(const AtomicGroup& u, const AtomicGroup& v) {
93  GCoord cu = u.centroid();
94  GCoord cv = v.centroid();
95 
96  cu.x() = cu.y() = 0.0;
97  cv.x() = cv.y() = 0.0;
98  return(distance(cu, cv));
99  }

106 struct MinDistance : public DistanceCalculation {
107  double operator()(const AtomicGroup& u, const AtomicGroup& v) {
108  double mind = numeric_limits<double>::max();
109 
110  for (AtomicGroup::const_iterator aj = v.begin(); aj != v.end(); ++aj) {
111  GCoord y = (*aj)->coords();
112 
113  for (AtomicGroup::const_iterator ai = u.begin(); ai != u.end(); ++ai) {
114  double d = distance(y, (*ai)->coords());
115  if (d < mind)
116  mind = d;
117  }
118  }
119 
120  return(mind);
121  }
122 };
123 
124 
125 // Maximum distance between any member of group u vs any member of
126 // group v
127 
128 struct MaxDistance : public DistanceCalculation {
129  double operator()(const AtomicGroup& u, const AtomicGroup& v) {
130  double maxd = 0.0;
131 
132  for (AtomicGroup::const_iterator aj = v.begin(); aj != v.end(); ++aj) {
133  GCoord y = (*aj)->coords();
134 
135  for (AtomicGroup::const_iterator ai = u.begin(); ai != u.end(); ++ai) {
136  double d = distance(y, (*ai)->coords());
137  if (d > maxd)
138  maxd = d;
139  }
140  }
141 
142  return(maxd);
143  }
144 };

149 class ToolOptions : public opts::OptionsPackage {
150 public:
151  ToolOptions() : mode_name("center") { }

153  void addGeneric(po::options_description& o) {
154  o.add_options()

160  void addHidden(po::options_description& o) {

164  }
165 
166  void addPositional(po::positional_options_description& p) {
167  p.add("target", 1);
168  p.add("selection", -1);
169  }
170 
171  bool check(po::variables_map& map) {
172  return(selection_names.empty() || target_name.empty());
173  }
174 
175  bool postConditions(po::variables_map& map) {
176  if (mode_name == "center")
177  calc_type = new CenterDistance;
178  else if (mode_name == "min")
179  calc_type = new MinDistance;
180  else if (mode_name == "max")
181  calc_type = new MaxDistance;
182  else if (mode_name == "zonly")
183  calc_type = new CenterDistanceZ;
184  else if (mode_name == "mass")
185  calc_type = new CenterOfMassDistance;
186  else {
187  cerr << "Error- calculation mode must be either 'center', 'mass', 'min', 'max', or 'zonly'\n";
188  return(false);
189  }
190 
191  calc_type->usePeriodicity(periodic);
192 
193  if (map.count("threshold")) {
194  threshold = map["threshold"].as<double>();
195  segment_output = true;
196  }
197 
198  return(true);
199  }
200 
201  string help() const { return("target selection [selection ...]"); }
202  string print() const {
203  ostringstream oss;
204  oss << boost::format("mode='%s', target='%s', selections=(%s), periodic=%d")
205  % mode_name
206  % target_name
207  % vectorAsStringWithCommas<string>(selection_names)
208  % periodic;
209 
210  if (segment_output)
211  oss << boost::format("threshold=%f") % threshold;
212 
213  return(oss.str());
214  }

224 string fullHelpMessage(void) {
225  string msg =

227  "SYNOPSIS\n"
228  "\n"
229  "Calculate the distance between two selections over a trajectory\n"
230  "\n"
231  "DESCRIPTION\n"
232  "\n"
233  "Given a model and a trajectory this tool will parse the simulation\n"
234  "and return the distance between a user supplied target selection and\n"
235  "any number of probe selections.\n"
236  "\n"
237  "There are several modes that can be selected for this tool. Each one\n"
238  "specifies a different way of determining the location within the\n"
239  "selection string to use in the distance calculation:\n"
240  "\t center - the geometric center\n"
241  "\t mass - the center of mass\n"
242  "\t min - the minimum distance\n"
243  "\t max - the maximum distance\n"
244  "\t zonly - only the z-component\n"
245  "\n"
246  "\n"
247  "EXAMPLE\n"
248  "\n"
249  "\tinterdist model.pdb traj.dcd 'name==\"CA\" && resid==133' 'name==\"CA\" && resid==234'\n"
250  "Calculate the CA to CA distance between residues 133 and 234 over the\n"
251  "course of trajectory traj.dcd This will print a frame number and a \n"
252  "distance for each frame to stdout.\n"
253  "\n"
254  "\tinterdist --mode min model.pdb traj.dcd 'name==\"NE\" && resid==135'\\\n"
255  "\t 'name=~\"OE.\" && resid==247'\n"
256  "This example is similar to the first, but --mode min returns the\n"
257  "minimum distance specifically. Note the change in the second\n"
258  "selection string. Here a regular expression was supplied which\n"
259  "will select either the OE1 or OE2 atom (charmm27). The --mode min \n"
260  "option will only return the distance to the closer atom.\n"
261  "\n"
262  "\tinterdist --mode zonly -r 50:250 model.pdb traj.dcd 'segid==\"PROT\"' \\\n"
263  "\t 'name==\"P\" && segid==\"TPE\"'\n"
264  "Here --mode z-only indicates thatwe are only taking the z-component\n"
265  "of the distance in this measurement. the supplied range -r 50:250 \n"
266  "is used to specify frames 50 to 250 for output.\n"

288  pTraj traj = tropts->trajectory;
289 
290  if (topts->periodic && !traj->hasPeriodicBox()) {
291  cerr << "Error- periodicity requested but trajectory is not periodic\n";
292  exit(-10);
293  }
294 
295  vector<uint> indices = tropts->frameList();
296 
297  AtomicGroup src = selectAtoms(model, topts->target_name);
298 
299  cout << "# " << header << endl;
300  cout << "# frame ";
301 
302  vector<AtomicGroup> targets;
303  for (uint i=0; i<topts->selection_names.size(); ++i) {
304  AtomicGroup trg = selectAtoms(model, topts->selection_names[i]);
305  targets.push_back(trg);
306  cout << "d_0_" << i << " ";
307  }
308  cout << endl;
309 
310  for (uint j=0; j<indices.size(); ++j) {
311  traj->readFrame(indices[j]);
312  traj->updateGroupCoords(model);
313 
314  cout << j << " ";
315 
316  topts->calc_type->setBox(model.periodicBox());
317 
318  for (vector<AtomicGroup>::iterator i = targets.begin(); i != targets.end(); ++i) {
319  double d = (*(topts->calc_type))(src, *i);
320  if (segment_output)
321  cout << (d <= threshold);
322  else
323  cout << d;
324 
325  if (i < targets.end()-1)
326  cout << '\t';

329  cout << endl;
