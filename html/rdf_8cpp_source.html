164 split_mode parseSplit(const string &split_by)
165  {
166  split_mode split;
167  // figure out the splitting mode
168  if (!split_by.compare("by-residue"))
169  {
170  split = BY_RESIDUE;
171  }
172  else if (!split_by.compare("by-segment"))
173  {
174  split = BY_SEGMENT;
175  }
176  else if (!split_by.compare("by-molecule"))
177  {
178  split = BY_MOLECULE;
179  }
180  else if (!split_by.compare("none"))
181  {
182  split = NONE;
183  }
184  else
185  {
186  cerr << "--split-mode(2) must be: by-residue|by-segment|by-molecule|none"
187  << endl;
188  exit(-1);
189  }
190  return (split);
191  }

193 uint doSplit(const AtomicGroup &system, const string selection,
194  const split_mode split, vector<AtomicGroup> &grouping)
195  {
196 
197  // make sure the "result" vector<AG> is empty to start
198  grouping.clear();
199 
200  vector<AtomicGroup> tmp;
201  if (split == BY_MOLECULE)
202  {
203  tmp = system.splitByMolecule();
204  }
205  if (split == BY_RESIDUE)
206  {
207  tmp = system.splitByResidue();
208  }
209  else if (split == BY_SEGMENT)
210  {
211  tmp = system.splitByUniqueSegid();
212  }
213  else if (split == NONE)
214  {
215  tmp.push_back(system);
216  }
217 
218  Parser parser(selection);
219  KernelSelector parsed_sel(parser.kernel());
220 
221  vector<AtomicGroup>::iterator t;
222  for (t=tmp.begin(); t!=tmp.end(); ++t)
223  {
224  AtomicGroup newgroup = t->select(parsed_sel);
225  if (newgroup.size() > 0)
226  {
227  grouping.push_back(newgroup);
228  }
229  }
230  return grouping.size();
231  }

234 int main (int argc, char *argv[])
235 {
236 
237 // parse the command line options
238 string hdr = invocationHeader(argc, argv);
239 opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
240 opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
241 opts::WeightsOptions* wopts = new opts::WeightsOptions;
242 ToolOptions* topts = new ToolOptions;
243 
244 opts::AggregateOptions options;
245 options.add(bopts).add(tropts).add(wopts).add(topts);
246 if (!options.parse(argc, argv))
247  exit(-1);
248 
249 // parse the split mode, barf if you can't do it
250 split_mode split=parseSplit(split_by);
251 split_mode split2=parseSplit(split_by2);
252 
253 // Print the command line arguments
254 cout << "# " << hdr << endl;
255 
256 // Create the system and the trajectory file
257 // Note: The pTraj type is a Boost shared pointer, so we'll need
258 // to use pointer semantics to access it
259 AtomicGroup system = tropts->model;
260 pTraj traj = tropts->trajectory;
261 if (!(system.isPeriodic() || traj->hasPeriodicBox()))
262  {
263  cerr << "Error- Either the model or the trajectory must have periodic box information.\n";
264  exit(-1);
265  }
266 
267 // Attach trajectory to weights
268 if (wopts->has_weights)
269  {
270  wopts->weights.add_traj(traj);
271  }
272 
273 
274 double bin_width = (hist_max - hist_min)/num_bins;
275 
276 // Select the 2 groups, then split them appropriately
277 vector<AtomicGroup> g1_mols, g2_mols;
278 uint numgroups = doSplit(system, selection1, split, g1_mols);
279 if (numgroups == 0)
280  {
281  cerr << "No groups created by selection1" << endl;
282  exit(1);
283  }
284 numgroups = doSplit(system, selection2, split2, g2_mols);
285 if (numgroups == 0)
286  {
287  cerr << "No groups created by selection2" << endl;
288  exit(1);
289  }
290 
291 
292 
293 // read the initial coordinates into the system
294 vector<uint> framelist = tropts->frameList();
295 traj->readFrame(framelist[0]);
296 traj->updateGroupCoords(system);
297 
298 // Create the histogram and zero it out
299 vector<double> hist;
300 hist.reserve(num_bins);
301 hist.insert(hist.begin(), num_bins, 0.0);
302 
303 double min2 = hist_min*hist_min;
304 double max2 = hist_max*hist_max;
305 
306 // Precompute the overlap between the two groups (this can be an
307 // expensive operation, so it's better to have it outside the
308 // while-loop)
309 
310 unsigned long unique_pairs = 0;
311 Math::Matrix<int, Math::RowMajor> group_overlap(g1_mols.size(), g2_mols.size());
312 
313 for (uint j=0; j<g1_mols.size(); ++j)
314  {
315  for (uint i=0; i<g2_mols.size(); ++i)
316  {
317  bool b = (g1_mols[j] == g2_mols[i]);
318  group_overlap(j, i) = b;
319  if ( !b )
320  {
321  ++unique_pairs;
322  }
323  }
324  }
325 
326 
327 // loop over the frames of the trajectory
328 uint framecount = framelist.size();
329 double volume = 0.0;
330 for (uint index = 0; index<framecount; ++index)
331  {
332  traj->readFrame(framelist[index]);
333  // update coordinates and periodic box
334  traj->updateGroupCoords(system);
335 
336  double weight = 1.0;
337  if (wopts->has_weights)
338  {
339  weight = wopts->weights();
340  wopts->weights.accumulate();
341  }
342 
343 
344  GCoord box = system.periodicBox();
345  volume += weight*(box.x() * box.y() * box.z());
346 
347  // compute the distribution of g2 around g1
348  for (unsigned int j = 0; j < g1_mols.size(); j++)
349  {
350  GCoord p1 = g1_mols[j].centerOfMass();
351  for (unsigned int k = 0; k < g2_mols.size(); k++)
352  {
353  // skip "self" pairs -- in case selection1 and selection2 overlap
354  if (group_overlap(j, k))
355  {
356  continue;
357  }
358  GCoord p2 = g2_mols[k].centerOfMass();
359 
360  // Compute the distance squared, taking periodicity into account
361  double d2 = p1.distance2(p2, box);
362  if ( (d2 < max2) && (d2 > min2) )
363  {
364  double d = sqrt(d2);
365  int bin = int((d-hist_min)/bin_width);
366  hist[bin]+=weight;
367  }
368  }
369  }
370  }
371 
372  if (wopts->has_weights)
373  {
374  volume /= wopts->weights.totalWeight();
375  }
376  else
377  {
378  volume /= framecount;
379  }
380 
381 
382 double expected = unique_pairs / volume;
383 if (wopts->has_weights)
384  {
385  expected *= wopts->weights.totalWeight();
386  }
387 else
388  {
389  expected *= framecount;
390  }
391 
392 double cum1 = 0.0;
393 double cum2 = 0.0;
394 
395 // Output the results
396 cout << "# Dist\tRDF\tCumAround1\tCumAround2" << endl;
397 for (int i = 0; i < num_bins; i++)
398  {
399  double d = bin_width*(i + 0.5);
400 
401  double d_inner = bin_width*i;
402  double d_outer = d_inner + bin_width;
403  double norm = 4.0/3.0 * M_PI*(d_outer*d_outer*d_outer
404  - d_inner*d_inner*d_inner);
405 
406  double total = hist[i]/ (norm*expected);
407  if (wopts->has_weights)
408  {
409  cum1 += hist[i] / (wopts->weights.totalWeight()*g1_mols.size());
410  cum2 += hist[i] / (wopts->weights.totalWeight()*g2_mols.size());
411  }
412  else
413  {
414  cum1 += hist[i] / (framecount*g1_mols.size());
415  cum2 += hist[i] / (framecount*g2_mols.size());
416  }
417 
418  cout << d << "\t" << total << "\t"
419  << cum1 << "\t" << cum2 << endl;
420 
421  }
422 }
