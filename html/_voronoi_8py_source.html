49 class VoronoiWrapper:
50  """
51  Wrap the scipy Voronoi class, which in turn is a wrapper around QHull.
52  So, it's a wrapper wrapper, and if this program is called from a script,
53  then we'd have a wrapper wrapper wrapper. Now I just need m4 and Tod will
54  approve.
55 
56  atomicgroup is a LOOS AtomicGroup
57  pad is a float specifying how far out we will generate padding atoms
58  (fake "image" atoms used to emulate periodicity). 15 ang is a good
59  default if you're using all atoms or all heavy atoms, but you may
60  need to go farther if you're using a sparser selection (e.g. just
61  lipid phosphates)
62  """

83  def generate_padding_atoms(self):
84  """
85  Build the list of atoms in the periodic images surrounding the central
86  image.
87 
88  This is necessary because QHull doesn't know anything about periodicity,
89  so we need to do fake it; otherwise, you'd have bizarre or infinite
90  areas for atoms near the edge of the box
91  """
92  if not self.isPeriodic():
93  raise VoronoiError("Periodic boundaries are required")
94 
95  box = self.atoms.periodicBox()
96  half_box = 0.5 * box
97  xbox = loos.GCoord(box.x(), 0.0, 0.0)
98  ybox = loos.GCoord(0.0, box.y(), 0.0)
99 
100  for a in self.atoms:
101  c = a.coords()
102  # generate the square neighbors
103  if (c.x() < -half_box.x() + self.pad):
104  self.padding_atoms.append(c + xbox)
105  if (c.x() > half_box.x() - self.pad):
106  self.padding_atoms.append(c - xbox)
107  if (c.y() < -half_box.y() + self.pad):
108  self.padding_atoms.append(c + ybox)
109  if (c.y() > half_box.y() - self.pad):
110  self.padding_atoms.append(c - ybox)
111 
112  # generate the diagonal neighbors
113  if ((c.x() < -half_box.x() + self.pad) and
114  (c.y() < -half_box.y() + self.pad)):
115  self.padding_atoms.append(c + xbox + ybox)
116 
117  if ((c.x() > half_box.x() - self.pad) and
118  (c.y() > half_box.y() - self.pad)):
119  self.padding_atoms.append(c - xbox - ybox)
120 
121  if ((c.x() < -half_box.x() + self.pad) and
122  (c.y() > half_box.y() - self.pad)):
123  self.padding_atoms.append(c + xbox - ybox)
124 
125  if ((c.x() > half_box.x() - self.pad) and
126  (c.y() < -half_box.y() + self.pad)):
127  self.padding_atoms.append(c - xbox + ybox)
128 
129  return len(self.padding_atoms)

137  def generate_voronoi(self):
138  # make the 2D list of points
139  points = []
140  for a in self.atoms:
141  points.append([a.coords().x(), a.coords().y()])
142 
143  numpad = self.generate_padding_atoms()
144  for p in self.padding_atoms:
145  points.append([p.x(), p.y()])
146 
147  points = numpy.array(points)
148  self.voronoi = Voronoi(points)
149 
150  # read in all of the ridges
151  for r in self.voronoi.ridge_vertices:
152  self.edges.append(Edge(r[0], r[1]))
153 
154  # now build the regions
155  # the scipy voronoi object has an array point_region, which
156  # maps the index of the input point to its associated Voronoi region
157  v = []
158  for vert in self.voronoi.vertices:
159  c = loos.GCoord(vert[0], vert[1], 0.0)
160  v.append(c)
161  for i in range(self.num_atoms()):
162  index = self.voronoi.point_region[i]
163  if index == -1:
164  raise ValueError("point %d (atomId = %d) from voronoi decomposition isn't associated with a voronoi region; you may need to increase the padding value" % i).with_traceback(
165  self.atoms[i].id())
166  r = self.voronoi.regions[index]
167  self.regions.append(Region(v, r, self.atoms[i]))
168  self.atoms_to_regions[self.atoms[i].id()] = self.regions[i]

204  def area(self):
205  area = 0.0
206  for i in range(self.num_indices()):
207  if ((i + 1) == self.num_indices()):
208  j = 0
209  else:
210  j = i + 1

212  p1 = self.vertices[self.indices[i]]
213  p2 = self.vertices[self.indices[j]]
214  area += p1.x() * p2.y() - p2.x() * p1.y()
