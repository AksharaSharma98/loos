42 string fullHelpMessage(void) {
43  string s =
44  "\n"
45  " SYNOPSIS\n"
46  "\n"
47  " Read a single structure and translate a specific selection to the origin\n"
48  "\n"
49  " DESCRIPTION\n"
50  "\n"
51  " This tool reads a file with coordinates and translates and reimages it\n"
52  " such that a selected set of atoms are at the origin. The user can\n"
53  " specify a subset of atoms to be written out. The resulting PDB file\n"
54  " is written to stdout.\n"
55  "\n"
56  " Options\n"
57  "\n"
58  " --center Selection specifying the atoms that would be moved to the \n"
59  " origin by the translation. Defaults to 'all'.\n"
60  " --apply Selection specifying the atoms to which the translation\n"
61  " is applied. Defaults to 'all'.\n"
62  " --write Selection specifying which atoms should be be written \n"
63  " out. Defaults to 'all'.\n"
64  " --reimage If specified, the system will be reimaged, first by \n"
65  " segment, then by molecule. If this flag is\n"
66  " specified, the system must contain connectivity and \n"
67  " periodicity information.\n"
68  " --center_xy Apply the translation only in the x-y plane\n"
69  " --bonds Specify a second file to use to find the connectivity \n"
70  " information, e.g. a PSF file to complement a PDB file \n"
71  " containing cooordinates.\n"
72  " --translate=(x,y,z) Translate center to this coordinate\n"
73  "\n"
74  "\n"
75  " EXAMPLES\n"
76  "\n"
77  "\tcenter-molecule model.pdb >centered.pdb\n"
78  "This centers based on all atoms in the model, transforming all atoms\n"
79  "\n"
80  "\tcenter-molecule --center 'name == \"CA\"' model.pdb >centered.pdb\n"
81  "This centers the model based on all alpha-carbons, transforming all atoms.\n"
82  "\n"
83  "\tcenter-molecule --center 'name == \"CA\"' --write '!hydrogen' model.pdb >centered.pdb\n"
84  "This centers the model based on all alpha-carbons, transforming all atoms, but\n"
85  "only writing out non-hydrogen atoms.\n"
86  "\n"
87  "NOTES\n"
88  "\n"
89  "\tThis tool is largely redundant with other tools, such as \n"

146 void copyBonds(AtomicGroup& target, AtomicGroup& source) {
147 
148  if (target.size() != source.size()) {
149  cerr << "ERROR- centering model and connectivity model have different number of atoms.\n";
150  exit(-1);

160 int main(int argc, char *argv[]) {
161  string hdr = invocationHeader(argc, argv);
162  opts::BasicOptions* bopts = new opts::BasicOptions;
163  opts::ModelWithCoords* mopts = new opts::ModelWithCoords;
164  ToolOptions* topts = new ToolOptions;
165  opts::AggregateOptions options;
166  options.add(bopts).add(mopts).add(topts);
167 
168  if (!options.parse(argc, argv))
169  exit(-1);
170 
171  AtomicGroup model = mopts->model;
172 
173  if (topts->reimage) {
174  if (!model.isPeriodic()) {
175  cerr << "WARNING- Reimaging requested, but the model has no periodic box information\n";
176  } else {
177  if (!topts->bonds_name.empty()) {
178  AtomicGroup bonds = createSystem(topts->bonds_name);
179  copyBonds(model, bonds);
180  }
181 
182  if (!model.hasBonds()) {
183  cerr << "WARNING- The model has no connectivity. Assigning bonds based on distance.\n";
184  model.findBonds();
185  }
186  }
187  }
188 
189  AtomicGroup center_mol = selectAtoms(model, topts->center_sel);
190  GCoord center = center_mol.centroid();
191 
192  AtomicGroup apply_mol = selectAtoms(model, topts->apply_sel);
193  GCoord offset = topts->translate - center;
194  if (topts->center_xy)
195  offset.z() = 0.0;
196 
197  for (AtomicGroup::iterator atom = apply_mol.begin(); atom != apply_mol.end(); ++atom)
198  (*atom)->coords() += offset;
199 
200  if (topts->reimage && model.isPeriodic()) {
201  vGroup molecules = model.splitByMolecule();
202  vGroup segments = model.splitByUniqueSegid();
203 
204  for (vGroup::iterator seg = segments.begin(); seg != segments.end(); ++seg)
205  seg->reimage();
206 
207  for (vGroup::iterator mol =molecules.begin(); mol != molecules.end(); ++mol)
208  mol->reimage();
209  }
210 
211  AtomicGroup write_mol = selectAtoms(model, topts->write_sel);
212  PDB pdb = PDB::fromAtomicGroup(write_mol);
213  pdb.remarks().add(hdr);
214  cout << pdb;
215 }
