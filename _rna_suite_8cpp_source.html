60  size_t RnaSuite::assignDDGIndex(double dihedral, vector<double> &min,
61  vector<double> &max, uint increment, uint &ddg_index) {
62 
63  size_t i = 0;
64  while (i < min.size()) {
65 
66  if (dihedral >= min[i] && dihedral <= max[i]) {
67 
68  ddg_index += i * increment;
69  return i;
70 
71  }
72 
73  ++i;
74 
75  }
76 
77  return i;
78 
79  } // assignDDGIndex()

81  void RnaSuite::assignSuitenameSuites() {
82 
83  size_t N_delta = delta_min.size();
84  size_t N_gamma = gamma_min.size();
85  size_t N_dg = N_delta * N_gamma;
86  vector<double> suite(7);
87 
88  // Index into delta(i-1), delta, gamma clusters
89  uint ddg_index;
90 
91  // Scaled 4D hyperellipsoid distance in epsilon, zeta, alpha, beta
92  double dist_ezab;
93 
94  // Closest scaled 4D hyperellipsoid distance to a cluster and index of
95  // the associated cluster
96  double min_dist_ezab;
97  size_t min_index;
98 
99  // Closest scaled 4D hyperellipsoid distance to a dominant cluster
100  double dom_min_dist_ezab;
101  size_t dom_min_index;
102 
103  // Closest scaled 4D hyperellipsoid distance to a non-dominant cluster
104  double sat_min_dist_ezab;
105  size_t sat_min_index;
106 
107  // Index into vector of widths for pair of dominant-satellite clusters
108  size_t dom_sat_index;
109 
110  // Number of clusters this dinucleotide could belong to
111  uint candidates;
112 
113  // Scaled 7D hyperellipsoid distance
114  double dist_7;
115 
116  // Index of the assigned suite
117  size_t assigned_suite_index;
118 
119  // Goodness-of-fit for assigned suite
120  double suiteness_score;
121 
122  if (suite_dihedrals.empty()) {
123 
124  cerr << "Warning: backbone dihedrals are empty" << endl;
125  return;
126 
127  }
128 
129  // Initialize vectors of suite names and suiteness scores
130  suite_names.clear();
131  suite_ddgs.clear();
132  suiteness.clear();
133  suite_names.reserve(N_suite);
134  suite_ddgs.reserve(N_suite);
135  suiteness.reserve(N_suite);
136 
137  for (size_t i = 0; i < N_suite; ++i) {
138 
139  // Assign delta(j-1), delta, gamma index. These 3 dihedrals have
140  // 12 clusters that are independent of the other 4 dihedrals.
141  ddg_index = 0;
142 
143  // Filter on 5' delta. Values outside of this range are
144  // indicative of incorrect stereochemistry in the ribose.
145  if (assignDDGIndex(suite_dihedrals[i][0], delta_min, delta_max,
146  N_dg, ddg_index) == N_delta) {
147 
148  suite_names.push_back("!d");
149  suite_ddgs.push_back("!!!");
150  suiteness.push_back(0.0);
151  continue;
152 
153  }
154 
155  // Filter on 3' delta
156  if (assignDDGIndex(suite_dihedrals[i][6], delta_min, delta_max,
157  N_gamma, ddg_index) == N_delta) {
158 
159  suite_names.push_back("!d");
160  suite_ddgs.push_back("!!!");
161  suiteness.push_back(0.0);
162  continue;
163 
164  }
165 
166  // Filter on gamma
167  if (assignDDGIndex(suite_dihedrals[i][5], gamma_min, gamma_max, 1,
168  ddg_index) == N_gamma) {
169 
170  suite_names.push_back("!g");
171  suite_ddgs.push_back("!!!");
172  suiteness.push_back(0.0);
173  continue;
174 
175  }
176 
177  // Filter on epsilon. Values outside of this range are
178  // indicative of a misfit sugar pucker.
179  if (suite_dihedrals[i][1] < ezab_min[0]
180  || suite_dihedrals[i][1] > ezab_max[0]) {
181 
182  suite_names.push_back("!e");
183  suite_ddgs.push_back("!!!");
184  suiteness.push_back(0.0);
185  continue;
186 
187  }
188 
189  // Filter on zeta
190  if (suite_dihedrals[i][2] < ezab_min[1]
191  || suite_dihedrals[i][2] > ezab_max[1]) {
192 
193  suite_names.push_back("!z");
194  suite_ddgs.push_back("!!!");
195  suiteness.push_back(0.0);
196  continue;
197 
198  }
199 
200  // Filter on alpha
201  if (suite_dihedrals[i][3] < ezab_min[2]
202  || suite_dihedrals[i][3] > ezab_max[2]) {
203 
204  suite_names.push_back("!a");
205  suite_ddgs.push_back("!!!");
206  suiteness.push_back(0.0);
207  continue;
208 
209  }
210 
211  // Filter on beta
212  if (suite_dihedrals[i][4] < ezab_min[3]
213  || suite_dihedrals[i][4] > ezab_max[3]) {
214 
215  suite_names.push_back("!b");
216  suite_ddgs.push_back("!!!");
217  suiteness.push_back(0.0);
218  continue;
219 
220  }
221 
222  // If there are no clusters associated with this ddg_index, then
223  // this is an outlier
224  if (N_reference_suite[ddg_index] == 0) {
225 
226  suite_names.push_back("!!");
227  suite_ddgs.push_back(reference_ddgs[ddg_index]);
228  suiteness.push_back(0.0);
229  continue;
230 
231  }
232 
233  // Find closest cluster in epsilon, zeta, alpha, beta
234  // Largest distance in 7D is 688.66^3, so 10^9 should be safe
235  min_dist_ezab = 999999999.0;
236  dom_min_dist_ezab = 999999999.0;
237  sat_min_dist_ezab = 999999999.0;
238  min_index = N_reference_suite[ddg_index];
239  dom_min_index = N_reference_suite[ddg_index];
240  sat_min_index = N_reference_suite[ddg_index];
241  candidates = 0;
242 
243  for (size_t j = 0; j < N_reference_suite[ddg_index]; ++j) {
244 
245  // Get 4D scaled hyperellipsoid distance
246  dist_ezab = hyperellipsoidDist(suite_dihedrals[i],
247  reference_dihedrals[ddg_index][j], dihedral_width, 1, 4);
248 
249  // Get closest cluster
250  if (dist_ezab < min_dist_ezab) {
251 
252  min_dist_ezab = dist_ezab;
253  min_index = j;
254 
255  }
256 
257  // Get closest non-dominant cluster
258  if (dominant_suites[ddg_index][j] != j
259  && dist_ezab < sat_min_dist_ezab) {
260 
261  sat_min_dist_ezab = dist_ezab;
262  sat_min_index = j;
263 
264  }
265 
266  // If 4D distance < 1, this reference suite is a candidate
267  if (dist_ezab < 1) {
268 
269  ++candidates;
270 
271  // Is this candidate a dominant cluster?
272  if (dominant_suites[ddg_index][j] == j) {
273 
274  dom_min_dist_ezab = dist_ezab;
275  dom_min_index = j;
276 
277  }
278 
279  }
280 
281  } // loop over reference suites
282 
283  // Assign membership to a reference suite
284 
285  // If there are multiple candidates, and the two canidates are
286  // a dominant-satellite pair, then reweight distances
287  if (candidates > 1 && dom_min_index != N_reference_suite[ddg_index]
288  && sat_min_index != N_reference_suite[ddg_index]
289  && dominant_suites[ddg_index][sat_min_index] == dom_min_index) {
290 
291  // Is the DNMP in between the dominant and satellite suites?
292  if (isBetweenDomSatPair(suite_dihedrals[i],
293  reference_dihedrals[ddg_index][dom_min_index],
294  reference_dihedrals[ddg_index][sat_min_index])) {
295 
296  // Rescale distances from point to dominant and satellite
297  // suites by ratio of distances from suite centers to
298  // boundary plane and assign to closest of the two
299  dom_sat_index = dom_sat_pair_index[ddg_index][sat_min_index];
300  if (hyperellipsoidDist(suite_dihedrals[i],
301  reference_dihedrals[ddg_index][sat_min_index],
302  satellite_width[dom_sat_index], 1, 4)
303  <= hyperellipsoidDist(suite_dihedrals[i],
304  reference_dihedrals[ddg_index][dom_min_index],
305  dominant_width[dom_sat_index], 1, 4))
306 
307  assigned_suite_index = sat_min_index;
308 
309  else assigned_suite_index = dom_min_index;
310 
311 
312  }
313 
314  else {
315 
316  // Assign to closer of dominant or satellite suite
317  if (sat_min_dist_ezab <= dom_min_dist_ezab)
318  assigned_suite_index = sat_min_index;
319  else assigned_suite_index = dom_min_index;
320 
321  }
322 
323  }
324 
325  // If there is zero or one candidate or multiple candidates but no
326  // dominant-satellite pair, then assign to the closest suite
327  else assigned_suite_index = min_index;
328 
329  // Make a final decision on whether this is an outlier using 7D
330  // hyperellipsoid distance
331  dist_7 = hyperellipsoidDist(suite_dihedrals[i],
332  reference_dihedrals[ddg_index][assigned_suite_index],
333  dihedral_width, 0, 6);
334 
335  if (dist_7 < 1) {
336 
337  suite_names.push_back(
338  reference_names[ddg_index][assigned_suite_index]);
339  suite_ddgs.push_back(reference_ddgs[ddg_index]);
340  suiteness_score = (1 + cos(M_PI * cbrt(dist_7))) / 2.0;
341  if (suiteness_score < suiteness_cutoff)
342  suiteness_score = suiteness_cutoff;
343  suiteness.push_back(suiteness_score);
344 
345  } else {
346 
347  suite_names.push_back("!!");
348  suite_ddgs.push_back(reference_ddgs[ddg_index]);
349  suiteness.push_back(0.0);
350 
351  }
352 
353  } // loop over suites
354 
355  } // assignSuitenameSuites()

357  void RnaSuite::calculateBackboneDihedrals() {
358 
359  double prev_delta;
360  vector<double> suite(7);
361 
362  // Clear vector of doubles for suite backbone dihedrals
363  suite_dihedrals.clear();
364 
365  for (size_t i = 0; i < N_continuous_group; ++i) {
366 
367  prev_delta = calculateDihedral(delta_atoms[i][0]);
368 
369  for (size_t j = 0; j < N_residue[i]; ++j) {
370 
371  suite[0] = prev_delta;
372  suite[1] = calculateDihedral(epsilon_atoms[i][j]);
373  suite[2] = calculateDihedral(zeta_atoms[i][j]);
374  suite[3] = calculateDihedral(alpha_atoms[i][j]);
375  suite[4] = calculateDihedral(beta_atoms[i][j]);
376  suite[5] = calculateDihedral(gamma_atoms[i][j]);
377  prev_delta = calculateDihedral(delta_atoms[i][j + 1]);
378  suite[6] = prev_delta;
379  suite_dihedrals.push_back(suite);
380 
381  }
382 
383  }
384 
385  } // calculateBackboneDihedrals()

387  double RnaSuite::calculateDihedral(const AtomicGroup &group) {
388 
389  double dihedral = Math::torsion(group[0], group[1], group[2], group[3]);
390  if (dihedral < 0.0) dihedral += 360.0;
391  return dihedral;
392 
393  } // calculateDihedral()

426  void RnaSuite::defineSuites(const string& suite_definition) {
427 
428  // Clear vectors for reference suites
429  reference_dihedrals.clear();
430  reference_names.clear();
431  reference_ddgs.clear();
432  dihedral_width.clear();
433  dominant_suites.clear();
434  dom_sat_pair_index.clear();
435  dominant_width.clear();
436  satellite_width.clear();
437  delta_min.clear();
438  delta_max.clear();
439  gamma_min.clear();
440  gamma_max.clear();
441  ezab_min = vector<double>(4);
442  ezab_max = vector<double>(4);
443  N_reference_ddg = 0;
444  N_reference_suite.clear();
445 
446  // Temporary variables for parsing lines from the definition file
447  size_t ddg_index;
448  size_t dom_index;
449  size_t sat_index;
450  size_t position;
451  string field;
452  string line;
453  string record;
454  vector<double> dihedrals(7);
455 
456  // Store dominant-satellite pairs
457  vector<size_t> domsat_ddg;
458  vector<size_t> domsat_dom;
459  vector<size_t> domsat_sat;
460  vector<vector<size_t> > domsat_dihedral;
461  vector<vector<double> > domsat_dom_width;
462  vector<vector<double> > domsat_sat_width;
463 
464  // Read file contents
465  ifstream ifs(suite_definition.c_str());
466  if (!ifs) throw(FileOpenError(suite_definition));
467 
468  while (getline(ifs, line)) {
469 
470  record = parseStringAs<std::string>(line, 0, 8);
471 
472  if (record.empty() || record[0] == '#') continue;
473 
474  else if (record == "suite") {
475 
476  // Define a reference suite
477 
478  // Get delta delta gamma cluster
479  field = parseStringAs<std::string>(
480  line, 16, min((size_t) 8, line.size() - 16));
481  ddg_index = N_reference_ddg;
482  for (size_t i = 0; i < N_reference_ddg; ++i)
483  if (field == reference_ddgs[i]) {
484  ddg_index = i;
485  break;
486  }
487 
488  // This is a new DDG cluster
489  if (ddg_index == N_reference_ddg) {
490  reference_ddgs.push_back(field);
491  reference_dihedrals.push_back(vector<vector<double> >());
492  reference_names.push_back(vector<string>());
493  ++N_reference_ddg;
494  N_reference_suite.push_back(0);
495  }
496 
497  // Get suite name
498  field = parseStringAs<std::string>(line, 8, 8);
499  if (field.empty()) continue;
500  reference_names[ddg_index].push_back(field);
501 
502  // Get reference suite dihedrals
503  dihedrals[0] = parseStringAs<double>(line, 24, 8);
504  dihedrals[1] = parseStringAs<double>(line, 32, 8);
505  dihedrals[2] = parseStringAs<double>(line, 40, 8);
506  dihedrals[3] = parseStringAs<double>(line, 48, 8);
507  dihedrals[4] = parseStringAs<double>(line, 56, 8);
508  dihedrals[5] = parseStringAs<double>(line, 64, 8);
509  dihedrals[6] = parseStringAs<double>(line, 72, 8);
510  reference_dihedrals[ddg_index].push_back(dihedrals);
511 
512  ++N_reference_suite[ddg_index];
513 
514  } else if (record == "width") {
515 
516  // Get default widths for hyperellipsoid distance
517  dihedral_width.push_back(parseStringAs<double>(line, 8, 8));
518  dihedral_width.push_back(parseStringAs<double>(line, 16, 8));
519  dihedral_width.push_back(parseStringAs<double>(line, 24, 8));
520  dihedral_width.push_back(parseStringAs<double>(line, 32, 8));
521  dihedral_width.push_back(parseStringAs<double>(line, 40, 8));
522  dihedral_width.push_back(parseStringAs<double>(line, 48, 8));
523  dihedral_width.push_back(parseStringAs<double>(line, 56, 8));
524 
525  } else if (record == "domsat") {
526 
527  // Define a dominant-satellite pair
528 
529  // Get index of dominant suite
530  field = parseStringAs<std::string>(line, 16, 8);
531  ddg_index = N_reference_ddg;
532  for (size_t i = 0; i < N_reference_ddg; ++i) {
533  for (size_t j = 0; j < N_reference_suite[i]; ++j)
534  if (field == reference_names[i][j]) {
535  ddg_index = i;
536  dom_index = j;
537  break;
538  }
539  if (ddg_index != N_reference_ddg) break;
540  }
541 
542  if (ddg_index == N_reference_ddg) {
543  cerr << boost::format(
544  "Warning: dominant suite %s was not defined in file %s")
545  % field % suite_definition << endl;
546  continue;
547  }
548 
549  // Get index of satellite suite
550  field = parseStringAs<std::string>(line, 8, 8);
551  sat_index = N_reference_suite[ddg_index];
552  for (size_t j = 0; j < N_reference_suite[ddg_index]; ++j)
553  if (field == reference_names[ddg_index][j]) {
554  sat_index = j;
555  break;
556  }
557 
558  if (sat_index == N_reference_suite[ddg_index]) {
559  cerr << boost::format(
560  "Warning: satellite suite %s was not defined in file %s")
561  % field % suite_definition << endl;
562  continue;
563  }
564 
565  domsat_ddg.push_back(ddg_index);
566  domsat_dom.push_back(dom_index);
567  domsat_sat.push_back(sat_index);
568 
569  // Loop over dihedrals with alternate widths
570  vector<size_t> dihedral_indices;
571  vector<double> dom_width;
572  vector<double> sat_width;
573  position = 24;
574  while (position < line.size()) {
575  dihedral_indices.push_back(
576  parseStringAs<size_t>(line, position, 8));
577  sat_width.push_back(
578  parseStringAs<double>(line, position + 8, 8));
579  dom_width.push_back(
580  parseStringAs<double>(line, position + 16, 8));
581  position += 24;
582  }
583  domsat_dihedral.push_back(dihedral_indices);
584  domsat_dom_width.push_back(dom_width);
585  domsat_sat_width.push_back(sat_width);
586 
587  } else if (record == "delta") {
588 
589  delta_min.push_back(parseStringAs<double>(line, 8, 8));
590  delta_max.push_back(parseStringAs<double>(line, 16, 8));
591 
592  } else if (record == "epsilon") {
593 
594  ezab_min[0] = parseStringAs<double>(line, 8, 8);
595  ezab_max[0] = parseStringAs<double>(line, 16, 8);
596 
597  } else if (record == "zeta") {
598 
599  ezab_min[1] = parseStringAs<double>(line, 8, 8);
600  ezab_max[1] = parseStringAs<double>(line, 16, 8);
601 
602  } else if (record == "alpha") {
603 
604  ezab_min[2] = parseStringAs<double>(line, 8, 8);
605  ezab_max[2] = parseStringAs<double>(line, 16, 8);
606 
607  } else if (record == "beta") {
608 
609  ezab_min[3] = parseStringAs<double>(line, 8, 8);
610  ezab_max[3] = parseStringAs<double>(line, 16, 8);
611 
612  } else if (record == "gamma") {
613 
614  gamma_min.push_back(parseStringAs<double>(line, 8, 8));
615  gamma_max.push_back(parseStringAs<double>(line, 16, 8));
616 
617  } else cerr << boost::format(
618  "Warning: Unrecognized record %s in suite definition from %s")
619  % record % suite_definition << endl;
620 
621  } // Loop over lines in file
622 
623  // Construct vectors for dominant-satellite pairs
624  for (size_t i = 0; i < N_reference_ddg; ++i) {
625  dominant_suites.push_back(
626  vector<size_t>(N_reference_suite[i], N_reference_suite[i]));
627  dom_sat_pair_index.push_back(
628  vector<size_t>(N_reference_suite[i], domsat_dihedral.size()));
629  }
630 
631  for (size_t i = 0; i < domsat_dihedral.size(); ++i) {
632  dominant_suites[domsat_ddg[i]][domsat_dom[i]] = domsat_dom[i];
633  dominant_suites[domsat_ddg[i]][domsat_sat[i]] = domsat_dom[i];
634  dom_sat_pair_index[domsat_ddg[i]][domsat_sat[i]] = i;
635  dominant_width.push_back(dihedral_width);
636  satellite_width.push_back(dihedral_width);
637  for (size_t j = 0; j < domsat_dihedral[i].size(); ++j) {
638  dominant_width[i][domsat_dihedral[i][j]] = domsat_dom_width[i][j];
639  satellite_width[i][domsat_dihedral[i][j]] = domsat_sat_width[i][j];
640  }
641  }
642 
643  } // defineSuites()

645  void RnaSuite::extractRnaBackboneAtoms(const AtomicGroup &group) {
646 
647  vector<AtomicGroup> continuous_alpha_atoms;
648  vector<AtomicGroup> continuous_beta_atoms;
649  vector<AtomicGroup> continuous_gamma_atoms;
650  vector<AtomicGroup> continuous_delta_atoms;
651  vector<AtomicGroup> continuous_epsilon_atoms;
652  vector<AtomicGroup> continuous_zeta_atoms;
653  AtomicGroup dihedral_atoms;
654  AtomicGroup residue_p;
655  AtomicGroup residue_o5p;
656  AtomicGroup residue_c5p;
657  AtomicGroup residue_c4p;
658  AtomicGroup residue_c3p;
659  AtomicGroup residue_o3p;
660  AtomicGroup prev_residue_c4p;
661  AtomicGroup prev_residue_c3p;
662  AtomicGroup prev_residue_o3p;
663  int current_resid = -2;
664  size_t residue_size;
665 
666  // True if this is the initial residue in a continuous group
667  bool first_res = true;
668 
669  // Clear vector of vectors of AtomicGroups for each backbone dihedral

780  prev_residue_c3p = residue_c3p;

887  bool RnaSuite::isBetweenDomSatPair(const vector<double> &dihedrals,
888  const vector<double> &dominant, const vector<double> &satellite) {
889 
890  double dom_to_sat;
891  double dom_dot_product = 0;
892  double sat_dot_product = 0;
893 
894  // If the point is in between the dominant and satellite reference
895  // suites, then the dot product between the vectors (point - dominant)
896  // and (satellite - dominant) and the dot product between the vectors
897  // (point - satellite) and (dominant - satellite) should both be
898  // positive, i.e. the cosine of the angles is positive.
899  for (uint i = 1; i <= 4; ++i) {
900 
901  dom_to_sat = satellite[i] - dominant[i];
902  dom_dot_product += (dihedrals[i] - dominant[i]) * dom_to_sat;
903  // sat_dot_product += (dihedrals[i] - satellite[i]) * sat_to_dom
904  // sat_to_dom = -dom_to_sat
905  sat_dot_product += (satellite[i] - dihedrals[i]) * dom_to_sat;
906 
907  }
908 
909  return dom_dot_product > 0 && sat_dot_product > 0;
910 
911  } // isBetweenDomSatPair()
