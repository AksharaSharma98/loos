63  struct RangeItem {
64  uint start;
65  uint stop;
66  int step;
67 
68 
69  bool validate() const {
70  if (start < stop && step < 0)
71  return false;
72  if (start > stop && step > 0)
73  return false;
74 
75  return true;
76  }
77 
78 
79  vector<uint> generate() const {
80  vector<uint> results;
81 
82  if (!validate())
83  throw(ParseError("Invalid range"));
84  if (step == 0)
85  results.push_back(start);
86  else if (step > 0) {
87  for (uint i=start; i<=stop; i += step)
88  results.push_back(i);
89  } else {
90  uint i;
91  for (i=start; i>= stop && i<=start; i += step)
92  results.push_back(i);
93  }
94 
95  return results;
96  }
97 
98  RangeItem() : start(0), stop(0), step(0) { }
99  RangeItem(const uint a) : start(a), stop(a), step(0) { }
100  RangeItem(const uint a, const uint b) : start(a), stop(b), step(1) { }
101  RangeItem(const uint a, const uint b, const int c) : start(a), stop(b), step(c) { }
102  };

109  template<typename Iterator>
110  struct RangeParser : qi::grammar<Iterator, RangeList(), ascii::space_type>
111  {
112  RangeParser() : RangeParser::base_type(start), maxsize(0) {
113  using qi::int_;
114  using qi::uint_;
115  using qi::_1;
116  using qi::_2;
117  using qi::_3;
118  using qi::_val;
119  using phoenix::ref;
120  using qi::eps;
121 
122 
123  endpoint = uint_ [ _val = _1 ]
124  | eps [ _val = ref(maxsize) ];
125 
126  ranger = (uint_ >> ':' >> int_ >> ':' >> endpoint ) [ _val = phoenix::new_<RangeItem>(_1,_3, _2) ]
127  | ( endpoint >> ':' >> int_ >> ':' >> uint_ ) [ _val = phoenix::new_<RangeItem>(_1,_3, _2) ]
128  | ( uint_ >> ':' >> endpoint ) [ _val = phoenix::new_<RangeItem>(_1, _2) ]
129  | ( endpoint >> ':' >> uint_ ) [ _val = phoenix::new_<RangeItem>(_1, _2, -1) ]
130  | ( uint_ ) [ _val = phoenix::new_<RangeItem>(_1) ];
131 
132  start = ranger % ',';
133  }

147  vector<uint> parseIndexRange(const std::string& input, const uint maxsize = 0) {
148  using boost::spirit::ascii::space;
149  typedef std::string::const_iterator iterator_type;
150  typedef internal::RangeParser<iterator_type> range_parser;
151 
152  range_parser parser;
153  parser.maxsize = maxsize;
154 
155  vector<internal::RangeItemPointer> rangelist;
156  string::const_iterator iter = input.begin();
157  string::const_iterator end = input.end();
158 
159  vector<uint> result;
160  bool ok = phrase_parse(iter, end, parser, space, rangelist);
161  if (ok && iter == end) {
162  for (uint i=0; i<rangelist.size(); ++i) {
163  vector<uint> r = rangelist[i]->generate();
164  copy(r.begin(), r.end(), back_inserter(result));
165  delete rangelist[i];
166  }
167 
168  } else {
169  throw(ParseError("Could not parse range: " + input));
170  }
171 
172  return result;
173  }
