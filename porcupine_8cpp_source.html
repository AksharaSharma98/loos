1 /*
2  porcupine
3 
4 
5  Creates "porcupine" plots by placing atoms at the endpoints of the
6  vectors and adding a bond between them. This is all written out a
7  PDB file with CONECT records.
8 
9  Notes:
10 
11  You can use a "map" file to map the eigenvectors back onto specific
12  atoms via atomid. The eigenvector matrix is arranged so that the
13  eigenvectors are stored as column-vectors, but each triplet of rows
14  is the eigenvector corresponding to an individual atom. The map
15  then consists of two columns: the left is the tripled index
16  (i.e. row/3) and the right column is the atomid of the corresponding
17  atom.
18 
19  0 30
20  1 42
21  2 57
22  3 66
23 
24  Alternatively, porcupine can infer the mapping of vectors to atoms
25  by giving it the same selection you used to compute the vectors
26  along with the same model...
27

325 double averageSubvectorLength(const Matrix& U, const uint col) {
326  double avg = 0.0;
327 
328  for (uint i=0; i<U.rows(); i += 3) {
329  double l = 0.0;
330  for (uint j=0; j<3; ++j)
331  l += U(i+j, col) * U(i+j, col);
332  avg += sqrt(l);
333  }
334 
335  return(avg / (U.rows()/3));
336 }

339 vector<double> determineScaling(const Matrix& U) {
340 
341  uint n = cols.size();
342  vector<double> scaling(n, 1.0);
343  vector<double> svals(n, 1.0);
344  vector<double> avgs(n, 0.0);
345 
346  // First, handle singular values, if given
347  if (!svals_file.empty()) {
348  Matrix S;
349  readAsciiMatrix(svals_file, S);
350  if (verbosity > 1)
351  cerr << "Read singular values from file " << svals_file << endl;
352  if (S.cols() != 1) {
353  cerr << boost::format("Error- singular value file is %d x %d, but it should be a %d x 1\n") % S.rows() % S.cols() % U.rows();
354  exit(-2);
355  }
356  for (uint i = 0; i<n; ++i) {
357  scaling[i] = S[cols[i]];
358  if (square)
359  scaling[i] *= scaling[i];
360  if (invert && scaling[i] != 0.0)
361  scaling[i] = 1.0 / scaling[i];
362  svals[i] = scaling[i];
363  }
364  }
365 
366  if (autoscale) {
367  // Find the largest avg subvector size...
368  double maxscale = 0.0;
369  for (uint i=0; i<n; ++i) {
370  double avg = averageSubvectorLength(U, cols[i]);
371  avgs[i] = avg;
372  double scale = avg * scaling[i];
373  if (scale > maxscale)
374  maxscale = scale;
375  }
376 
377  for (uint i=0; i<n; ++i)
378  scaling[i] *= autolength / maxscale;
379  }
380 
381  // Incorporate additional scaling...
382  if (verbosity > 1) {
383  cerr << boost::format("%4s %4s %15s %15s %15s\n") % "col" % "mode" % "sval" % "avg" % "scale";
384  cerr << boost::format("%4s %4s %15s %15s %15s\n") % "----" % "----" % "---------------" % "---------------" % "---------------";
385  }
386  for (uint i=0; i<n; ++i) {
387  scaling[i] *= scales[i] * global_scale;
388  if (verbosity > 1)
389  cerr << boost::format("%4d %4d %15.5f %15.5f %15.5f\n") % cols[i] % (cols[i]-offset) % svals[i] % avgs[i] % scaling[i];
390  else if (verbosity > 0)
391  cerr << boost::format("Scaling column %d by %f\n") % cols[i] % scaling[i];
392  }
393 
394  return(scaling);
395 }

400 int main(int argc, char *argv[]) {
401  string hdr = invocationHeader(argc, argv);
402 
403  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
404  opts::BasicSelection* sopts = new opts::BasicSelection("name == 'CA'");
405  opts::ModelWithCoords* mopts = new opts::ModelWithCoords;
406  ToolOptions* topts = new ToolOptions;
407  opts::RequiredArguments *ropts = new opts::RequiredArguments;
408  ropts->addArgument("lsv", "left-singular-vector-file");
409 
410  opts::AggregateOptions options;
411  options.add(bopts).add(sopts).add(mopts).add(topts).add(ropts);
412  if (!options.parse(argc, argv)) {
413  cerr << "***WARNING***\n";
414  cerr << "The interface to porcupine has changed significantly\n";
415  cerr << "and is not compatible with previous versions. See the\n";
416  cerr << "help info above, or the --fullhelp guide.\n";
417  exit(-1);
418  }
419 
420  verbosity = bopts->verbosity;
421 
422  // First, read in the LSVs
423  Matrix U;
424  readAsciiMatrix(ropts->value("lsv"), U);
425  uint m = U.rows();
426 
427  vector<double> scalings = determineScaling(U);
428 
429  // Read in the average structure...
430  AtomicGroup avg = mopts->model;
431 
432  vector<int> atomids;
433  if (map_name.empty()) {
434  if (sopts->selection.empty())
435  atomids = fakeMap(avg);
436  else
437  atomids = inferMap(avg, sopts->selection);
438 
439  } else
440  atomids = readMap(map_name);
441 
442  // Double check size of atomid map
443  if (atomids.size() * 3 != m) {
444  cerr << boost::format("Error - The vector-to-atom map (provided or inferred) has %d atoms, but expected %d.\n") %
445  atomids.size() % (m / 3);
446  exit(-1);
447  }
448 
449  int atomid = 1;
450  AtomicGroup spines;
451 
452  for (uint j=0; j<cols.size(); ++j) {
453  int resid = 1;
454  double k = scalings[j];
455  uint col = cols[j];
456 
457  string segid = generateSegid(col - offset);
458 
459  for (uint i=0; i<m; i += 3) {
460  GCoord v(U(i,col), U(i+1,col), U(i+2,col));
461  if (uniform)
462  v /= v.length();
463 
464  v *= k;
465  pAtom pa = avg.findById(atomids[i/3]);
466  GCoord c = pa->coords();
467 
468  pAtom atom2;
469  if (double_sided)
470  atom2 = pAtom(new Atom(atomid++, porcupine_tag, c-v));
471  else
472  atom2 = pAtom(new Atom(atomid++, porcupine_tag, c));
473 
474  atom2->resid(resid);
475  atom2->resname(porcupine_tag);
476  atom2->segid(segid);
477 
478 
479  if (tip_size == 0.0) {
480  pAtom atom1(new Atom(atomid++, porcupine_tag, c+v));
481  atom1->resid(resid);
482  atom1->resname(porcupine_tag);
483  atom1->segid(segid);
484 
485 
486  atom1->addBond(atom2);
487  atom2->addBond(atom1);
488 
489  spines.append(atom2);
490  spines.append(atom1);
491 
492  } else {
493 
494  GCoord base = c + v;
495  GCoord tip = base + (v / v.length()) * tip_size;
496 
497  pAtom atom1(new Atom(atomid++, porcupine_tag, base));
498  atom1->resid(resid);
499  atom1->resname(porcupine_tag);
500  atom1->segid(segid);
501  atom1->addBond(atom2);
502  atom2->addBond(atom1);
503 
504  pAtom atom0(new Atom(atomid++, tip_tag, tip));
505  atom0->resid(resid);
506  atom0->resname(porcupine_tag);
507  atom0->segid(segid);
508  atom1->addBond(atom0);
509  atom0->addBond(atom1);
510 
511  spines.append(atom2);
512  spines.append(atom1);
513  spines.append(atom0);
514  }
515  ++resid;
516  }
517 
518  }
519 
520  PDB outpdb = PDB::fromAtomicGroup(spines);
521  outpdb.remarks().add(hdr);
522  cout << outpdb;
523 }
