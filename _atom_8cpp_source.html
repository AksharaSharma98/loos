221  os << " ATOMICNUMBER='" << a._atomic_number <<"'";
222  os << " MASK='" << boost::format("%x") % a.mask << "'";

227  os << " <BOND>" << *i << "</BOND>\n";
228  os << "</ATOM>";
229  } else
230  os << "/>";
231  return(os);
