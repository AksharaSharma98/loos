936  // This function is to to remove code duplication in
937  // logisticContacts() and logisticContacts2D().
938  // Handle even and odd powers separately -- even can
939  // avoid the sqrt
940  // Sigh, this doesnt' seem to make it much faster...
941  double logisticFunc(const GCoord& cent, const GCoord& other, double radius, int sigma, const GCoord& box) const{
942  double prod;
943  if (sigma % 2 == 0) {
944  double distance2 = cent.distance2(other, box);
945  double ratio = distance2/(radius*radius);
946  prod = ratio;
947  for (int j=0; j<(sigma/2)-1; ++j) {
948  prod *= ratio;
949  }
950  }
951  else {
952  double distance = cent.distance(other, box);
953  double ratio = distance/radius;
954  prod = ratio;
955  for (int j=0; j < sigma-1; ++j) {
956  prod *= ratio;
957  }
958  }
959  double sum = 1./(1. + prod);
960  return(sum);
961  }

964  // Find all atoms in the current group that are within dist

970  template <typename DistanceCalc>
971  AtomicGroup within_private(const double dist, AtomicGroup& grp, const DistanceCalc& distance_functor) const {
972 
973  AtomicGroup res;
974  res.box = box;
975 
976  double dist2 = dist * dist;
977  std::vector<uint> indices;
978 
979  for (uint j=0; j<size(); j++) {
980  GCoord c = atoms[j]->coords();
981  for (uint i=0; i<grp.size(); i++) {
982  if (distance_functor(c, grp.atoms[i]->coords()) <= dist2) {
983  indices.push_back(j);
984  break;
985  }
986  }
987  }
988 
989  if (indices.size() == 0)
990  return(res);
991 
992  for (std::vector<uint>::const_iterator ci = indices.begin(); ci != indices.end(); ++ci)
993  res.addAtom(atoms[*ci]);
994 
995  return(res);
996  }

1000  bool contactwith_private(const double dist, const AtomicGroup& grp, const uint min_contacts, const DistanceCalc& distance_function) const {

1003

1006  for (uint i = 0; i<grp.size(); ++i)

1020  void findBondsImpl(const double dist, const DistanceCalc& distance_function) {
1021  iterator ij;
1022  double dist2 = dist * dist;
1023  double current_dist2;
1024 
1025  for (ij = begin(); ij != end() - 1; ++ij) {
1026  iterator ii;
1027  GCoord u = (*ij)->coords();
1028 
1029  for (ii = ij + 1; ii != end(); ++ii) {
1030  current_dist2 = distance_function(u, (*ii)->coords());
1031  if (current_dist2 < dist2) {
1032  (*ij)->addBond(*ii);
1033  (*ii)->addBond(*ij);
1034  }
1035  }
1036  }
