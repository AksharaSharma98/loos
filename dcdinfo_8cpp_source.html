49 void analyzeBoxes(DCD& dcd) {
50  dcd.rewind();
51  GCoord max(0,0,0), min(1e38, 1e38, 1e38), avg(0,0,0);
52  uint n = 0;
53 
54  while (dcd.readFrame()) {
55  GCoord b = dcd.periodicBox();
56 
57  avg += b;
58  ++n;
59 
60  if (b.length2() > max.length2())
61  max = b;
62  if (b.length2() < min.length2())
63  min = b;
64  }
65 
66  avg /= n;
67 
68  cout << "* Average box size is " << avg << ", min is " << min << ", and max is " << max << endl;
69 
70 }

100 int main(int argc, char *argv[]) {
101 
102  if (argc < 2 || argc > 3) {
103  cerr << "Usage - dcdinfo [-s] trajectory.dcd\n";
104  cerr << " -s scan the DCD for box information\n";
105  cerr << fullHelpMessage();
106  exit(-1);
107  }
108 
109  int opt = 1;
110  bool scan = false;
111  if (strcmp(argv[opt], "-s") == 0) {
112  scan = true;
113  ++opt;
114  }
115 
116  DCD::setSuppression(true);
117  DCD dcd(argv[opt]);
118 
119  if (!dcd.nativeFormat())
120  cout << "The DCD is not in a native binary format.\n";
121 
122  cout << boost::format("* DCD has %u atoms in %u frames with a timestep of %f.\n") % dcd.natoms() % dcd.nframes() % dcd.timestep();
123  uint n = countFrames(dcd);
124  if (n != dcd.nframes())
125  cout << "WARNING- Trajectory actually has " << n << " rather than what is given in the header!\n"
126  << " You can fix this using the fixdcd tool on the trajectory.\n";
127 
128  if (dcd.hasCrystalParams()) {
129  cout << "* DCD HAS box/crystal information.\n";
130  dcd.readFrame();
131  vector<double> xtal = dcd.crystalParams();
132  cout << "* DCD Crystal params (first frame): ";
133  copy(xtal.begin(), xtal.end(), ostream_iterator<double>(cout, " "));
134  cout << endl;
135  if (scan) {
136  cout << "Scanning trajectory for box information...\n";
137  analyzeBoxes(dcd);
138  }
139  } else
140  cout << "* DCD has no box/crystal information.\n";
141 
142  vector<string> titles = dcd.titles();
143  if (titles.empty())
144  cout << "* DCD has no titles\n";
145  else {
146  cout << "* DCD Titles (" << titles.size() << "):\n";
147  for (uint i=0; i<titles.size(); ++i)
148  cout << "[" << i << "] : " << titles[i] << endl;
149  }
150 
151  cout << "icntrl dump:\n";
152  for (int i=0; i<20; ++i)
153  cout << boost::format("\ticntrl[%d]\t= %d\n") % i % dcd.icntrl(i);
154 
155 }
