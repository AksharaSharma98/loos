287 struct ScanfFmt
288 {
289  ScanfFmt(const string& s) : fmt(s) { }
290 
291  uint operator()(const string& s) const
292  {
293  uint d;
294  if (sscanf(s.c_str(), fmt.c_str(), &d) != 1)
295  {
296  cerr << boost::format("Bad conversion of '%s' using format '%s'\n") % s % fmt;
297  exit(-20);
298  }
299 
300  return(d);
301  }
302 
303  string fmt;
304 };
305

307 
308 struct RegexFmt
309 {
310  RegexFmt(const string& s) : fmt(s)
311  {
312  regexp = boost::regex(s, boost::regex::perl);
313  }
314 
315  uint operator()(const string& s) const
316  {
317  boost::smatch what;
318  if (boost::regex_search(s, what, regexp))
319  {
320  for (uint i=0; i<what.size(); ++i)
321  {
322  string submatch(what.str(i));
323  char *p;
324  if (submatch.empty()) // Should never happen?
325  continue;
326 
327  uint val = strtoul(submatch.c_str(), &p, 10);
328  if (*p == '\0')
329  return(val);
330  }
331  }
332 
333  cerr << boost::format("Bad conversion of '%s' using regexp '%s'\n") % s % fmt;
334  exit(-20);
335  }
336 
337  string fmt;
338  boost::regex regexp;
339 };

362 // Given a vector of trajectory filenames, along with a functor for
363 // extracting the frame index from the filename, sorts it in numeric
364 // ascending order...
365 
366 template<class FmtOp>
367 vector<string> sortNamesByFormat(vector<string>& names, const FmtOp& op)
368 {
369  uint n = names.size();
370  vector<SortDatum> bound(n);
371  for (uint i=0; i<n; ++i)
372  {
373  bound[i] = SortDatum(names[i], op(names[i]));
374  }
375 
376  sort(bound.begin(), bound.end());
377 
378  vector<string> sorted(n);
379  for (uint i=0; i<n; ++i)
380  sorted[i] = bound[i].s;
381 
382  return(sorted);
383 }

388 {
389  string hdr = invocationHeader(argc, argv);
390  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
391  ToolOptions* topts = new ToolOptions;
392  opts::RequiredArguments* ropts = new opts::RequiredArguments;
393  ropts->addArgument("model", "model-filename");
394  ropts->addArgument("output_traj", "output-trajectory");
395  ropts->addVariableArguments("input_traj", "trajectory");
396 
397  opts::AggregateOptions options;
398  options.add(bopts).add(topts).add(ropts);
399  if (!options.parse(argc, argv))
400  exit(-1);
401 
402  model_name = ropts->value("model");
403  output_traj = ropts->value("output_traj");
404  input_dcd_list = ropts->variableValues("input_traj");
405 
406  if (topts->sort_flag)
407  {
408  if (!topts->scanf_spec.empty())
409  {
410  input_dcd_list = sortNamesByFormat(input_dcd_list, ScanfFmt(topts->scanf_spec));
411  }
412  else
413  {
414  input_dcd_list = sortNamesByFormat(input_dcd_list, RegexFmt(topts->regex_spec));
415  }
416  }
417 
418  cout << hdr << endl;
419  AtomicGroup system = createSystem(model_name);
420 
421  // We check for specifying both xy/z and full in the code
422  // that processes the command line options, so we don't
423  // have to do it here
424  bool full_recenter = false;
425  bool xy_recenter = false;
426  bool z_recenter = false;
427  bool post_recenter = false;
428  bool xy_post_recenter = false;
429  bool z_post_recenter = false;
430  if ( center_selection.length() != 0 )
431  {
432  full_recenter = true;
433  }
434  if ( xy_center_selection.length() != 0 )
435  {
436  xy_recenter = true;
437  }
438  if ( z_center_selection.length() != 0 )
439  {
440  z_recenter = true;
441  }
442  if (postcenter_selection.length() != 0)
443  {
444  post_recenter = true;
445  }
446  if (postcenter_xy_selection.length() != 0)
447  {
448  xy_post_recenter = true;
449  }
450  if (postcenter_z_selection.length() != 0)
451  {
452  z_post_recenter = true;
453  }
454 
455  pTrajectoryWriter output = createOutputTrajectory(output_traj, true);
456 
457  pTrajectoryWriter output_downsample;
458  bool do_downsample = (output_traj_downsample.length() > 0);
459  if (do_downsample)
460  {
461  output_downsample = createOutputTrajectory(output_traj_downsample, true);
462  }
463 
464  // Set up to do the recentering
465  vector<AtomicGroup> molecules;
466  AtomicGroup center, xy_center, z_center;
467  AtomicGroup post_center, xy_post_center, z_post_center;
468  vector<AtomicGroup>::iterator m;
469  if ( full_recenter )
470  {
471  center = selectAtoms(system, center_selection);
472  }
473  else
474  {
475  if ( xy_recenter )
476  {
477  xy_center = selectAtoms(system, xy_center_selection);
478  }
479  if ( z_recenter )
480  {
481  z_center = selectAtoms(system, z_center_selection);
482  }
483  }
484 
485  if ( full_recenter || xy_recenter || z_recenter || reimage_by_molecule )
486  {
487  if ( system.hasBonds() )
488  {
489  molecules = system.splitByMolecule();
490  }
491  else
492  {
493  molecules = system.splitByUniqueSegid();
494  }
495  }
496 
497  if (post_recenter)
498  {
499  post_center = selectAtoms(system, postcenter_selection);
500  }
501  else
502  {
503  if ( xy_post_recenter )
504  {
505  xy_post_center = selectAtoms(system, postcenter_xy_selection);
506  }
507  else if ( z_post_recenter )
508  {
509  z_post_center = selectAtoms(system, postcenter_z_selection);
510  }
511  }
512 
513  uint original_num_frames = output->framesWritten();
514  cout << "Target trajectory "
515  << output_traj
516  << " has "
517  << original_num_frames
518  << " frames."
519  << endl;
520 
521  uint previous_frames = 0;
522  vector<string>::iterator f;
523  for (f=input_dcd_list.begin(); f!=input_dcd_list.end(); ++f)
524  {
525  pTraj traj=createTrajectory(*f, system);
526  int nframes = traj->nframes();
527  if (skip_first_frame && nframes > 1)
528  {
529  nframes--;
530  }
531  cout << "File: " << *f << ": " << nframes;
532 
533  if ( previous_frames + nframes <= original_num_frames)
534  // all of this file is contained in the existing file, skip it
535  {
536  // increment the frame pointer
537  previous_frames += nframes;
538  cout << " ( " << previous_frames << " )"
539  << "\tSkipping trajectory "
540  << endl;
541 
542  }
543  else
544  // we need at least some of the data from this file
545  {
546  int frames_to_skip = original_num_frames - previous_frames;
547  if ( frames_to_skip > 0 )
548  {
549  traj->seekFrame(frames_to_skip-1);
550  }
551  else
552  {
553  frames_to_skip = 0;
554  }
555 
556  previous_frames += frames_to_skip;
557 
558  // if this is an xtc file, we need to skip 1 more frame
559  if (skip_first_frame)
560  {
561  traj->readFrame();
562  }
563 
564  cout << " ( " << previous_frames + nframes - frames_to_skip
565  << " ) "
566  << "\t Writing " << nframes - frames_to_skip
567  << " frames."
568  << endl;
569 
570  while ( traj->readFrame() )
571  {
572  traj->updateGroupCoords(system);
573 
574  // Find the smallest box dimension
575  GCoord box = system.periodicBox();
576  double smallest=1e20;
577  for (int i=0; i<3; i++)
578  {
579  if (box[i] < smallest)
580  {
581  smallest = box[i];
582  }
583  }
584 
585  smallest /=2.0;
586 
587 
588  // If molecules can be broken across image bondaries
589  // (eg GROMACS), then we may need 2 translations to
590  // fix them -- first, translate the whole molecule such
591  // that a single atom is at the origin, reimage the
592  // molecule, and put it back
593  if (reimage_by_molecule)
594  {
595  for (m=molecules.begin(); m != molecules.end(); ++m )
596  {
597  // This is relatively slow, so we'll skip the
598  // cases we know we won't need this -- 1 particle
599  // molecules and molecules with small radii
600  // Note: radius(true) computes the max distance between atom 0
601  // and all other atoms in the group. In certain perverse
602  // cases the centroid can be closer than 1/2 box to all atoms
603  // even when the molecule is split.
604  if ( (m->size() > 1) && (m->radius(true) > smallest) )
605  {
606  m->mergeImage();
607  m->reimage();
608  }
609  }
610  }
611 
612 
613  if ( full_recenter || xy_recenter || z_recenter)
614  {
615  // If the selection is split, then we effectively need to
616  // do the centering twice. First, we pick one atom from the
617  // centering selection, translate the entire system so it's
618  // at the origin, and reimage. This will get the selection
619  // region to not be split on the image boundary. At that
620  // point, we can just do regular imaging.
621  if (selection_split)
622  {
623  GCoord centroid;
624  if (full_recenter)
625  {
626  centroid = center[0]->coords();
627  }
628  else
629  {
630  if (xy_recenter)
631  {
632  centroid.x() = xy_center[0]->coords().x();
633  centroid.y() = xy_center[0]->coords().y();
634  }
635  if (z_recenter)
636  {
637  centroid.z() = z_center[0]->coords().z();
638  }
639  }
640 
641  system.translate(-centroid);
642 
643  for (m=molecules.begin(); m!=molecules.end(); m++)
644  {
645  m->reimage();
646  }
647  }
648  // Now, do the regular imaging. Put the system centroid
649  // at the origin, and reimage by molecule
650  GCoord centroid;
651  if (full_recenter)
652  {
