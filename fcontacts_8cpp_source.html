45 string fullHelpMessage() {
46  string s =
47  "\n"
48  "SYNOPSIS\n"
49  "Determine what fraction of contacts with a probe belong to the specified targets\n"
50  "\n"
51  "DESCRIPTION\n"
52  "\tfcontacts can be used to classify the fraction of contacts made with\n"
53  "a probe selection as belonging to different target selections.\n"
54  "For each atom in the probe selection, all atoms that are within outer radius\n"
55  "but farther than inner radius are counted as contacts. The number of contacting\n"
56  "atoms that belong to each target selection are then counted and the corresponding\n"
57  "fraction is written out. This is repeated for each time step, so the final\n"
58  "output is a matrix with time increasing along the rows and each column is the\n"
59  "fractional contact corresponding to the targets listed on the command line.\n"
60  "\n"
61  "\tThe autoself option splits the probe selection based on connectivity (if present)\n"
62  "or segid. The algorithm will iterate over each molecule and report the average\n"
63  "fractional contact.\n"
64  "\n"
65  "\tThe exclude option determines whether the entire probe molecule is excluded\n"
66  "from the total contacts or not. For example, if the probe is a side-chain\n"
67  "of a peptide and the outer radius is set long, then there will be additional\n"
68  "peptide contacts that you may not want to be included in the contact count.\n"
69  "Turning the exclude option on will tell fcontacts to find all atoms connected\n"
70  "to the probe selection (after splitting, if so requested) and ignore these atoms\n"
71  "when calculating the total number of contacts.\n"
72  "\n"
73  "The selection option in fcontacts is used as a 'pre-filter' for all subsequent\n"
74  "selections. This is useful for excluding hydrogens, for example.\n"
75  "\n"
76  "\tfcontacts --inner=0 --outer=4.5 model.psf traj.dcd 'segid == \"PEPT\"'\\\n"
77  "\t 'resname == \"PEGL\"' 'segid == \"BULK\"'\n"
78  "This example counts contacts as any atom with 4.5 angstroms and prints out\n"
79  "the fraction of contacts with atoms having segid PEPT and PEGL residues vs\n"
80  "bulk water (segid BULK)\n"
81  "\n"
82  "fcontacts --inner=2 --outer=5 model.psf traj.dcd 'resname == \"TRP\"'\\\n"
83  "\t 'resname == \"PEGL\"' 'resname == \"PGGL\"' 'segid == \"BULK\"'\n"
84  "This example counts contacts as any atom-atom distance greater than 2 angstroms\n"

193 // Return a list of target atoms that are in contact with probe
194 AtomicGroup contacts(const AtomicGroup& probe, const AtomicGroup& target, const double inner_radius, const double outer_radius, const bool symmetry) {
195 
196  double or2 = outer_radius * outer_radius;
197  double ir2 = inner_radius * inner_radius;
198 
199  GCoord box = target.periodicBox();
200  AtomicGroup contacting_atoms;
201 
202 
203  for (AtomicGroup::const_iterator j = probe.begin(); j != probe.end(); ++j) {
204  GCoord v = (*j)->coords();
205 
206  for (AtomicGroup::const_iterator i = target.begin(); i != target.end(); ++i) {
207  GCoord u = (*i)->coords();
208  double d = symmetry ? v.distance2(u, box) : v.distance2(u);
209  if (d >= ir2 && d <= or2)
210  contacting_atoms.append(*i);
211  }
212  }
213 
214  return(contacting_atoms);
215 }

218 AtomicGroup pickNearbyAtoms(const AtomicGroup& probe, const AtomicGroup& target, const double radius, const bool symmetry) {
219 
220  GCoord c = probe.centroid();
221  GCoord box = probe.periodicBox();
222  double maxrad = probe.radius() + radius;
223  maxrad *= maxrad;
224 
225 
226  AtomicGroup nearby;
227  nearby.periodicBox(probe.periodicBox());
228  for (AtomicGroup::const_iterator i = target.begin(); i != target.end(); ++i) {
229  double d = symmetry ? c.distance2((*i)->coords(), box) : c.distance2((*i)->coords());
230  if (d <= maxrad)
231  nearby.append(*i);
232  }
233 
234  return(nearby);
235 }

238 vector<double> fractionContactsToProbe(const AtomicGroup& probe,
239  const AtomicGroup& nearby,
240  const vGroup& targets,
241  const double inner_radius,
242  const double outer_radius,
243  const bool symmetry)
244 {
245 
246  // First, find which nearby atoms are actually in contact...
247  AtomicGroup nearby_contacts = contacts(probe, nearby, inner_radius, outer_radius, symmetry);
248 
249  vector<double> fracts(targets.size(), 0.0);
250  for (uint i=0; i<targets.size(); ++i) {
251  AtomicGroup target_nearby = nearby_contacts.intersect(targets[i], IdEquals());
252  if (nearby_contacts.empty())
253  fracts[i] = 0.0;
254  else
255  fracts[i] = static_cast<double>(target_nearby.size()) / nearby_contacts.size();
256  }
257  return(fracts);
258 }

261 FContactsList fractionContacts(const AtomicGroup& system,
262  const vGroup& probes,
263  const vGroup& excludeds,
264  const vGroup& targets,
265  const double inner_radius,
266  const double outer_radius,
267  const bool symmetry)
268 {
269 
270  FContactsList fclist;
271 
272  for (uint j=0; j<probes.size(); ++j) {
273  AtomicGroup nearby = pickNearbyAtoms(probes[j], excludeds[j], outer_radius, symmetry);
274  vector<double> v = fractionContactsToProbe(probes[j], nearby, targets,
275  inner_radius, outer_radius, symmetry);
276  fclist.push_back(v);
277  }
278 
279  return(fclist);
280 }

284 vector<double> average(const FContactsList& f)
285 {
286  vector<double> avgs(f[0].size(), 0.0);
287  for (uint i=0; i<avgs.size(); ++i) {
288  for (uint j=0; j<f.size(); ++j)
289  avgs[i] += f[j][i];
290  avgs[i] /= f.size();
291  }
292 
293  return(avgs);
294 }
295 
296 
297 vector<double> stddevs(const FContactsList& f, const vector<double>& avgs)
298 {
299  vector<double> stds(f[0].size(), 0.0);
300  for (uint i=0; i<stds.size(); ++i) {
301  for (uint j=0; j<f.size(); ++j) {
302  double d = f[j][i] - avgs[i];
303  stds[i] += d*d;
304  }
305  stds[i] /= (f.size()-1);
306  }
307 
308  return(stds);
309 }

313 int main(int argc, char *argv[]) {
314  string hdr = invocationHeader(argc, argv);
315 
316  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
317  opts::BasicSelection* sopts = new opts::BasicSelection();
318  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices();
319  ToolOptions* topts = new ToolOptions;
320 
321  opts::AggregateOptions options;
322  options.add(bopts).add(sopts).add(tropts).add(topts);
323 
324  if (!options.parse(argc, argv))
325  exit(-1);
326 
327  AtomicGroup model = tropts->model;
328  pTraj traj = tropts->trajectory;
329  vector<uint> indices = tropts->frameList();
330 
331  AtomicGroup system = selectAtoms(model, sopts->selection);
332  AtomicGroup probe = selectAtoms(system, topts->probe_selection);
333 
334 
335  // Build each of the requested targets...
336  vGroup targets;
337  for (vector<string>::iterator i = topts->target_selections.begin(); i != topts->target_selections.end(); ++i)
338  targets.push_back(selectAtoms(system, *i));
339 
340 
341  // If splitting, then split based on presence of connectivity...
342  vGroup myselves;
343  vGroup excludes;
344 
345  if (topts->auto_split) {
346  if (probe.hasBonds())
347  myselves = probe.splitByMolecule();
348  else
349  myselves = probe.splitByUniqueSegid();
350 
351  } else
352  myselves.push_back(probe);
353 
354 
355  if (topts->exclude_self) {
356  vGroup molecules;
357  if (system.hasBonds())
358  molecules = system.splitByMolecule();
359  else
360  molecules = system.splitByUniqueSegid();
361 
362  for (vGroup::iterator i = myselves.begin(); i != myselves.end(); ++i) {
363  AtomicGroup exclusive;
364  for (vGroup::iterator j = molecules.begin(); j != molecules.end(); ++j)
365  if (i->containsAny(*j, IdEquals()))
366  exclusive.append(*j);
367  excludes.push_back(exclusive);
368  }
369  } else
370  excludes = myselves;
371 
372  // This is system excluding requested probe atoms...
373  vGroup excludeds;
374  for (vGroup::iterator i = excludes.begin(); i != excludes.end(); ++i) {
375  AtomicGroup pruned = system;
376  pruned.remove(*i);
377  excludeds.push_back(pruned);
378  }
379 
380 
381  // Size of the output matrix
382  uint rows = indices.size();
383  uint cols = targets.size();
384  if (topts->report_stddev)
385  cols *= 2;
386  ++cols;
387 
388 
389  uint t = 0;
390  DoubleMatrix M(rows, cols);
391 
392  // Setup our progress counter since this can be a time-consuming
393  // program, but only if verbose output is requested.
394  PercentProgressWithTime watcher;
395  ProgressCounter<PercentTrigger, EstimatingCounter> slayer(PercentTrigger(0.1), EstimatingCounter(indices.size()));
396  slayer.attach(&watcher);
397  if (bopts->verbosity)
398  slayer.start();
399 
400  for (vector<uint>::iterator frame = indices.begin(); frame != indices.end(); ++frame) {
401  traj->readFrame(*frame);
402  traj->updateGroupCoords(model);
403 
404  if (topts->symmetry && !model.isPeriodic()) {
405  cerr << "ERROR - the trajectory must be periodic to use --reimage\n";
406  exit(-1);
407  }
408 
409  M(t, 0) = *frame;
410 
411  FContactsList fcl = fractionContacts(system, myselves, excludeds, targets, topts->inner_cutoff, topts->outer_cutoff, topts->symmetry);
412  vector<double> avg = average(fcl);
413  if (topts->report_stddev) {
414  vector<double> stds = stddevs(fcl, avg);
415  uint k = 0;
416  for (uint i=0; i<avg.size(); ++i) {
417  M(t, ++k) = avg[i];
418  M(t, ++k) = stds[i];
419  }
420 
421  } else {
422 
423  for (uint i=0; i<avg.size(); ++i)
424  M(t, i+1) = avg[i];
425 
426  }
427 
428  ++t;
429  if (bopts->verbosity)
430  slayer.update();
431  }
432 
433  if (bopts->verbosity)
434  slayer.finish();
435 
436  writeAsciiMatrix(cout, M, hdr);
437 }
