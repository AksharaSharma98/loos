124 typedef boost::tuple<GCoord, GCoord, GCoord, GCoord, GCoord> BoxInfo;

126 BoxInfo scanBoxes(pTraj& traj, const uint maxframe) {
127  GCoord min, max, avg;
128  GCoord mine, maxe;
129 
130  traj->rewind();
131  traj->readFrame();
132  mine = maxe = min = max = avg = traj->periodicBox();
133  double minsize = min[0] * min[1] * min[2];
134  double maxsize = max[0] * max[1] * max[2];
135 
136  for (uint t = 1; t<maxframe; ++t) {
137  traj->readFrame();
138  GCoord box = traj->periodicBox();
139  avg += box;
140  double size = box[0] * box[1] * box[2];
141  if (size < minsize) {
142  minsize = size;
143  min = box;
144  }
145  if (size > maxsize) {
146  maxsize = size;
147  max = box;
148  }
149  for (int i=0; i<3; ++i) {
150  if (mine[i] > box[i])
151  mine[i] = box[i];
152  if (maxe[i] < box[i])
153  maxe[i] = box[i];
154  }
155  }
156 
157  avg /= traj->nframes();
158  BoxInfo result(avg, min, max, mine, maxe);
159  return(result);
160 }

164 boost::tuple<GCoord, GCoord> scanCentroid(AtomicGroup& model, pTraj& traj, const uint maxframe) {
165  vector<GCoord> centers;
166  GCoord avg;
167 
168  traj->rewind();
169  for (uint t=0; t<maxframe; ++t) {
170  traj->readFrame();
171  traj->updateGroupCoords(model);
172  GCoord c = model.centroid();
173  centers.push_back(c);
174  avg += c;
175  }
176 
177  avg /= maxframe;
178 
179  GCoord std;
180  for (uint i=0; i<maxframe; ++i) {
181  GCoord c = centers[i] - avg;
182  std += c*c;
183  }
184 
185  std /= (maxframe - 1);
186  for (uint i=0; i<3; i++)
187  std[i] = sqrt(std[i]);
188 
189  boost::tuple<GCoord, GCoord> result(avg, std);
190  return(result);
191 }

194 uint verifyFrames(pTraj& traj) {
195  uint n = 0;
196 
197  traj->rewind();
198  bool flag = true;
199  while (flag) {
200  try {
201  flag = traj->readFrame();
202  }
203  catch (LOOSError& e) {
204  cerr << "*****************************************" << endl;
205  cerr << "At frame " << n << "...\n";
206  cerr << e.what() << endl;
207  cerr << "*****************************************" << endl;
208  return(n);
209  }
210  catch (exception& e) {
211  cerr << "An errror occured while reading frame " << n << endl;
212  cerr << e.what() << endl;
213  exit(-10);
214  }
215  if (flag)
216  ++n;
217  }
218 
219  return(n);
220 }

227 void verbInfo(AtomicGroup& model, pTraj& traj, AtomicGroup& center, const bool centroid = false) {
228  cout << boost::format(fldpre + "%s\n") % "Model name" % model_name;
229  cout << boost::format(fldpre + "%s\n") % "Trajectory name" % traj_name;
230  cout << boost::format(fldpre + "%s\n") % "Format" % traj->description();
231  cout << boost::format(fldpre + "%d\n") % "Number of atoms" % traj->natoms();
232  cout << boost::format(fldpre + "%d\n") % "Number of frames" % traj->nframes();
233  uint n = verifyFrames(traj);
234  cout << boost::format(fldpre + "%d\n") % "Actual frames" % n;
235 
236  cout << boost::format(fldpre + "%f\n") % "Timestep" % traj->timestep();
237  if (traj->hasPeriodicBox()) {
238  cout << boost::format(fldpre + "%s\n") % "Periodic box" % "yes";
239  if (box_info) {
240  BoxInfo box = scanBoxes(traj, n);
241  cout << boost::format(fldpre + "%s\n") % "Average box" % boost::get<0>(box);
242  cout << boost::format(fldpre + "%s\n") % "Smallest box" % boost::get<1>(box);
243  cout << boost::format(fldpre + "%s\n") % "Largest box" % boost::get<2>(box);
244  cout << boost::format(fldpre + "%s x %s\n") % "Box extents" % boost::get<3>(box) % boost::get<4>(box);
245  }
246  } else
247  cout << boost::format(fldpre + "%s\n") % "Periodic box" % "no";
248 
249  if (centroid) {
250  boost::tuple<GCoord, GCoord> res = scanCentroid(center, traj, n);
251  cout << boost::format(fldpre + "%s +- %s\n") % "Average Centroid" % boost::get<0>(res) % boost::get<1>(res);
252  }
253  cout << boost::format(fldpre + "%s\n") % "Velocities" % (traj->hasVelocities() ? "yes" : "no");
254 
255  if (n != traj->nframes()) {
256  cout << endl;
257  cout << "***WARNINGS***\n";
258  cout << "* Frame count mismatch between header and trajectory.\n";
259  if (box_info || centroid)
260  cout << "* Centroid and box information uses actual frames (may be fewer than listed in header)\n";
261  if (n < traj->nframes()) {
262  cout << "* The trajectory has fewer frames than expected.\n";
263  cout << "* Try using subsetter to extract the first " << n << " valid frames.\n";
264  }
265 
266  if (traj->description() == "CHARMM/NAMD DCD")
267  cout << "* If the trajectory is not corrupted, try fixdcd to correct the header.\n";
268  }
269 }

296 
297  box_info = topts->box_info;
298  centroid = !topts->centroid_selection.empty();
299 
300  AtomicGroup model = tropts->model;
301  pTraj traj = tropts->trajectory;
302 
303  if (model.size() != traj->natoms())
304  cerr << boost::format("WARNING- the trajectory has %d atoms but the system defines %d\n") % traj->natoms() % model.size();
305 
306  // Export names for subsequent functions
307  model_name = tropts->model_name;
308  traj_name = tropts->traj_name;
309 
310  AtomicGroup center;
311  if (centroid)
312  center = selectAtoms(model, topts->centroid_selection);
313 
314  if (!topts->brief)
315  verbInfo(model, traj, center, !(topts->centroid_selection.empty()));
316  else
317  briefInfo(traj, topts->verify);
