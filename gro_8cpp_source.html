34  void Gromacs::read(std::istream& ifs) {
35  std::string buf;
36 
37  getline(ifs, title_);
38 
39  // Get the # of atoms;;;
40  getline(ifs, buf);
41  int natoms = parseStringAs<int>(buf);
42  while (natoms-- > 0) {
43  getline(ifs, buf);
44 
45  int resid = parseStringAs<int>(buf, 0, 5);
46  std::string resname = parseStringAs<std::string>(buf, 5, 5);
47  std::string name = parseStringAs<std::string>(buf, 10, 5);
48  int atomid = parseStringAs<int>(buf, 15, 5);
49  float x = parseStringAs<float>(buf, 20, 8) * 10.0;
50  float y = parseStringAs<float>(buf, 28, 8) * 10.0;
51  float z = parseStringAs<float>(buf, 36, 8) * 10.0;
52 
53 
54 
55  pAtom pa(new Atom);
56  pa->index(_max_index++);
57  pa->resid(resid);
58  pa->id(atomid);
59  pa->resname(resname);
60  pa->name(name);
61  pa->coords(GCoord(x,y,z));
62 
63  if (buf.length() > 44) {
64  _has_velocities = true;
65  float vx = parseStringAs<float>(buf, 44, 8) * 10.0;
66  float vy = parseStringAs<float>(buf, 52, 8) * 10.0;
67  float vz = parseStringAs<float>(buf, 60, 8) * 10.0;
68  pa->velocities(GCoord(vx, vy, vz));
69  }
70 
71 
72  append(pa);
73  }
74 
75  // Now process box...
76  getline(ifs, buf);
77  std::istringstream iss(buf);
78  GCoord box;
79  if (!(iss >> box[0] >> box[1] >> box[2]))
80  throw(FileReadError(_filename, "Cannot parse box '" + buf + "'"));
81  periodicBox(box * 10.0);
82 
83  // Since the atomic field in .gro files is only 5-chars wide, it can
84  // overflow. if there are enough atoms to cause an overflow, manually
85  // renumber everything...
86  if (atoms.size() >= 100000)
87  renumber();
88  }

90  std::string Gromacs::atomAsString(const pAtom p) const {
91  std::ostringstream s;
92 
93  // Float formatter for coords
94  Fmt crdfmt(3);
95  crdfmt.width(8);
96  crdfmt.right();
97  crdfmt.trailingZeros(true);
98  crdfmt.fixed();
99 
100  // Float formatter for velocities
101  Fmt velfmt(4);
102  velfmt.width(8);
103  velfmt.right();
104  velfmt.trailingZeros(true);
105  velfmt.fixed();
106 
107  s << std::setw(5) << p->resid();
108  s << std::left << std::setw(5) << p->resname();
109  s << std::right << std::setw(5) << p->name();
110  s << std::setw(5) << p->id();
111  s << crdfmt(p->coords().x()/10.);
112  s << crdfmt(p->coords().y()/10.);
113  s << crdfmt(p->coords().z()/10.);
114  s << velfmt(0.0);
115  s << velfmt(0.0);
116  s << velfmt(0.0);
117 
118  return(s.str());
119 
120  }
