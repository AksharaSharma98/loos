216 // Parcels out work to the compute threads... Work is given to the threads
217 // one row at a time.

219 class Master {
220 public:
221 
222  Master(const uint nr, const bool tr, const bool b) : _toprow(0), _maxrow(nr), _updatefreq(500), _triangle(tr),
223  _verbose(b), _start_time(time(0))
224  {
225  if (_triangle)
226  _total = _maxrow*(_maxrow-1) / 2;
227  else
228  _total = _maxrow;
229  }
230 
231  // Checks whether there are any columns left to work on
232  // and places the column index into the passed pointer.
233 
234  bool workAvailable(uint* ip)
235  {
236 
237  _mtx.lock();
238  if (_toprow >= _maxrow) {
239  _mtx.unlock();
240  return(false);
241  }
242  *ip = _toprow++;
243 
244  if (_verbose)
245  if (_toprow % _updatefreq == 0)
246  updateStatus();
247 
248  _mtx.unlock();
249  return(true);
250  }
251 
252 
253  void updateStatus() {
254  time_t dt = elapsedTime();
255  uint work_done = _triangle ? (_toprow * (_toprow-1) / 2) : (_toprow);
256  uint work_left = _total - work_done;
257  uint d = work_left * dt / work_done; // rate = work_done / dt; d = work_left / rate;
258 
259  uint hrs = d / 3600;
260  uint remain = d % 3600;
261  uint mins = remain / 60;
262  uint secs = remain % 60;
263 
264  cerr << boost::format("Row %5d /%5d, Elapsed = %5d s, Remaining = %02d:%02d:%02d\n")
265  % _toprow % _maxrow % dt % hrs % mins % secs;
266  }
267 
268 
269 
270  time_t elapsedTime() const
271  {
272  return (time(0) - _start_time);
273  }
274 
275 
276 private:
277  uint _toprow, _maxrow;
278  uint _updatefreq;
279  bool _triangle;
280  bool _verbose;
281  time_t _start_time;
282  uint _total;
283  boost::mutex _mtx;
284 
285 };

316  void calc(const uint i)
317  {
318  for (uint j=0; j<_maxcol; ++j) {

387 template<class W>
388 class Threader
389 {
390 public:
391  Threader(W* wrkr, const uint np) :
392  worker(wrkr),
393  threads(vector<Thread>(np))
394  {
395 
396  for (uint i=0; i<np; ++i) {
397  W w(*worker);
398  threads[i] = new boost::thread(w);
399  }
400 
401  }
402 
403 
404  void join()
405  {
406  for (uint i=0; i<threads.size(); ++i)
407  threads[i]->join();
408  }
409 
410 
411  ~Threader()
412  {
413  for (uint i=0; i<threads.size(); ++i)
414  delete threads[i];
415  }
416 
417  W* worker;
418  vector<Thread> threads;
419 };

425 void showStatsHalf(const RealMatrix& R) {
426  uint total = (R.rows() * (R.rows()-1)) / 2;
427 
428  double avg = 0.0;
429  double max = 0.0;
430  for (uint j=1; j<R.rows(); ++j)
431  for (uint i=0; i<j; ++i) {
432  avg += R(j, i);
433  if (R(j, i) > max)
434  max = R(j, i);
435  }
436 
437  avg /= total;
438  cerr << boost::format("Max rmsd = %.4f, avg rmsd = %.4f\n") % max % avg;
439 }

467 void checkMemoryUsage(long mem) {
468  if (!mem)
469  return;
470 
471  double used = static_cast<double>(used_memory) / mem;
472 
473  if (verbosity > 2)
474  cerr << boost::format("Memory: available=%d GB, estimated used=%.2f MB\n")
475  % (mem >> 30) % (static_cast<double>(used_memory) / (1lu<<20) );
476 
477  if (used >= cache_memory_fraction_warning) {
478  cerr << boost::format("***WARNING***\nThe estimated memory used is %.1f%% (%d MB) of your total memory (%d GB).\n")
479  % (used * 100.0)
480  % (used_memory >> 20)
481  % (mem >> 30);
482 
483  cerr << "If your machine starts swapping, try subsampling the trajectories\n";
484  }

490 int main(int argc, char *argv[]) {
491  string header = invocationHeader(argc, argv);
492 
493  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
494  ToolOptions* topts = new ToolOptions;
495 
496  opts::AggregateOptions options;
497  options.add(bopts).add(topts);
498  if (!options.parse(argc, argv))
499  exit(-1);
500 
501  verbosity = bopts->verbosity;
502  report_stats = (verbosity || topts->noop);
503  AtomicGroup model = createSystem(topts->model1);
504  pTraj traj = createTrajectory(topts->traj1, model);
505  AtomicGroup subset = selectAtoms(model, topts->sel1);
506  vector<uint> indices = assignTrajectoryFrames(traj, topts->range1, topts->skip1);
507 
508  long mem = availableMemory();
509  uint nthreads = topts->nthreads ? topts->nthreads : boost::thread::hardware_concurrency();
510 
511  if (verbosity > 1) {
512  cerr << "Using " << nthreads << " threads\n";
513  cerr << "Reading trajectory - " << topts->traj1 << endl;
514  }
515  vMatrix T = readCoords(subset, traj, indices, verbosity > 1);
516  used_memory += T.size() * T[0].size() * sizeof(vMatrix::value_type::value_type); // Coords matrix
517  used_memory += T.size() * T.size() * sizeof(RealMatrix::element_type); // RMSDS matrix
518  checkMemoryUsage(mem);
519  centerTrajectory(T);
520 
521  RealMatrix M;
522  if (topts->model2.empty()) {
523 
524  if (verbosity > 1)
525  cerr << "Calculating RMSD...\n";
526  M = RealMatrix(T.size(), T.size());
527  Master master(T.size(), true, verbosity);
528  SingleWorker worker(&M, &T, &master);
529  Threader<SingleWorker> threads(&worker, nthreads);
530  threads.join();
531  if (verbosity)
532  master.updateStatus();
533 
534  if (verbosity || topts->noop || topts->stats)
535  showStatsHalf(M);
536 
537  } else {
538  AtomicGroup model2 = createSystem(topts->model2);
539  pTraj traj2 = createTrajectory(topts->traj2, model2);
540  AtomicGroup subset2 = selectAtoms(model2, topts->sel2);
541  vector<uint> indices2 = assignTrajectoryFrames(traj2, topts->range2, topts->skip2);
542 
543  if (verbosity > 1)
544  cerr << "Reading trajectory - " << topts->traj2 << endl;
545  vMatrix T2 = readCoords(subset2, traj2, indices2, verbosity > 1);
546  used_memory += T2.size() * T2[0].size() * sizeof(double);
547  checkMemoryUsage(mem);
548  centerTrajectory(T2);
549 
550  if (verbosity > 1)
551  cerr << "Calculating RMSD...\n";
552  M = RealMatrix(T.size(), T2.size());
553  Master master(T.size(), false, verbosity);
554  DualWorker worker(&M, &T, &T2, &master);
555  Threader<DualWorker> threads(&worker, nthreads);
556  threads.join();
557 
558  if (verbosity)
559  master.updateStatus();
560 
561  if (verbosity || topts->noop || topts->stats)
562  showStatsWhole(M);
563  }
564 
565  if (!topts->noop) {
566  cout << "# " << header << endl;
567  cout << setprecision(topts->matrix_precision) << M;
568  }
569
