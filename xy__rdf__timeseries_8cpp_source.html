47 int main (int argc, char *argv[])
48 {
49 if ( (argc <= 1) ||
50  ( (argc >= 2) && (strncmp(argv[1], "-h", 2) == 0) ) ||
51  (argc < 11)
52  )
53  {
54  Usage();
55  exit(-1);
56  }
57 
58 cout << "# " << invocationHeader(argc, argv) << endl;
59 cout << "# This program is now deprecated in favor of "
60  << "xy_rdf, with the --timeseries argument"
61  << endl;
62 
63 // copy the command line variables to real variable names
64 AtomicGroup system = createSystem(argv[1]);
65 pTraj traj = createTrajectory(argv[2], system);
66 char *selection1 = argv[3];
67 char *selection2 = argv[4];
68 double hist_min = atof(argv[5]);
69 double hist_max = atof(argv[6]);
70 int num_bins = atoi(argv[7]);
71 int skip = atoi(argv[8]);
72 int interval = atoi(argv[9]);
73 char *dir_name = argv[10];
74 
75 double bin_width = (hist_max - hist_min)/num_bins;
76 
77 AtomicGroup group1 = selectAtoms(system, selection1);
78 AtomicGroup group2 = selectAtoms(system, selection2);
79 
80 // The groups are presumed to be units like lipid headgroups. We want to
81 // treat each headgroup as a single unit. This could reasonably be done
82 // with splitByUniqueSegid() or splitByMolecule(), depending on the details.
83 // For lipids and lipopeptides, it's the same (although splitByMolecule will
84 // have more overhead), but if you want water, then you need splitbyMolecule.
85 // If you get weird-looking behavior, you might want to change this to use
86 // splitByMolecule.
87 vector<AtomicGroup> g1_mols = group1.splitByUniqueSegid();
88 vector<AtomicGroup> g2_mols = group2.splitByUniqueSegid();
89 
90 // Skip the initial frames
91 traj->readFrame(skip);
92 
93 // read the initial coordinates into the system
94 traj->updateGroupCoords(system);
95 
96 // Now that we have some real coordinates, we need to subdivide the groups
97 // one more time, into upper and lower leaflets. This assumes that the
98 // coordinates are properly centered and imaged.
99 vector<AtomicGroup> g1_upper, g1_lower;
100 vector<AtomicGroup> g2_upper, g2_lower;
101 
102 for (unsigned int i = 0; i < g1_mols.size(); i++)
103  {
104  GCoord c = g1_mols[i].centerOfMass();
105  if (c.z() >=0.0)
106  {
107  g1_upper.push_back(g1_mols[i]);
108  }
109  else
110  {
111  g1_lower.push_back(g1_mols[i]);
112  }
113  }
114 
115 for (unsigned int i = 0; i < g2_mols.size(); i++)
116  {
117  GCoord c = g2_mols[i].centerOfMass();
118  if (c.z() >=0.0)
119  {
120  g2_upper.push_back(g2_mols[i]);
121  }
122  else
123  {
124  g2_lower.push_back(g2_mols[i]);
125  }
126  }
127 
128 
129 // Create 2 histograms -- one for top, one for bottom
130 vector<double> hist_lower, hist_upper;
131 hist_lower.reserve(num_bins);
132 hist_upper.reserve(num_bins);
133 hist_lower.insert(hist_lower.begin(), num_bins, 0.0);
134 hist_upper.insert(hist_upper.begin(), num_bins, 0.0);
135 
136 double min2 = hist_min*hist_min;
137 double max2 = hist_max*hist_max;
138 
139 // Set up the normalization
140 uint num_upper, num_lower;
141 // Normalization is slightly different is this is a self distribution
142 if (group1 == group2)
143  {
144  // the two selection groups are the same
145  num_upper = g1_upper.size()*(g1_upper.size()-1);
146  num_lower = g1_lower.size()*(g1_lower.size()-1);
147  }
148 else
149  {
150  // the two selection groups are different
151  num_upper = g1_upper.size() * g2_upper.size();
152  num_lower = g1_lower.size() * g2_lower.size();
153  }
154 
155 double upper_expected = interval * num_upper;
156 double lower_expected = interval * num_lower;
157 
158 
159 // loop over the frames of the traj file
160 int frame = 0;
161 double area = 0.0;
162 while (traj->readFrame())
163  {
164  // update coordinates and periodic box
165  traj->updateGroupCoords(system);
166  GCoord box = system.periodicBox();
167  area += box.x() * box.y();
168 
169  // compute the distribution of g2 around g1 for the lower leaflet
170  for (unsigned int j = 0; j < g1_lower.size(); j++)
171  {
172  GCoord p1 = g1_lower[j].centerOfMass();
173  for (unsigned int k = 0; k < g2_lower.size(); k++)
174  {
175  // skip "self" pairs
176  if (g1_lower[j] == g2_lower[k])
177  {
178  continue;
179  }
180  GCoord p2 = g2_lower[k].centerOfMass();
181  GCoord displ = (p2 - p1);
182  displ.reimage(box);
183  double d2 = (displ.x() * displ.x()) + (displ.y() * displ.y());
184  if ( (d2 < max2) && (d2 > min2) )
185  {
186  double d = sqrt(d2);
187  int bin = int((d-hist_min)/bin_width);
188  hist_lower[bin]++;
189  }
190  }
191  }
192 
193  // compute the distribution of g2 around g1 for the upper leaflet
194  for (unsigned int j = 0; j < g1_upper.size(); j++)
195  {
196  GCoord p1 = g1_upper[j].centerOfMass();
197  for (unsigned int k = 0; k < g2_upper.size(); k++)
198  {
199  // skip "self" pairs
200  if (g1_upper[j] == g2_upper[k])
201  {
202  continue;
203  }
204  GCoord p2 = g2_upper[k].centerOfMass();
205  GCoord displ = (p2 - p1);
206  displ.reimage(box);
207  double d2 = (displ.x() * displ.x()) + (displ.y() * displ.y());
208  if ( (d2 < max2) && (d2 > min2) )
209  {
210  double d = sqrt(d2);
211  int bin = int((d-hist_min)/bin_width);
212  hist_upper[bin]++;
213  }
214  }
215  }
216 
217  frame++;
218 
219  // periodically dump out the rdf
220  if (frame %interval == 0)
221  {
222  area /= interval;
223 
224  // create the output file
225  ostringstream outfilename;
226  outfilename << dir_name << "/" << "rdf_" << frame << ".dat";
227  ofstream out(outfilename.str().c_str());
228  if (out.fail())
229  {
230  cerr << "couldn't open " << outfilename.str() << " ... exiting" << endl;
231  exit(-1);
232  }
233  out << "# Dist\tTotal\tUpper\tLower" << endl;
234  float cum = 0.0;
235  for (int m = 0; m < num_bins; m++)
236  {
237  float d = bin_width*(m + 0.5);
238 
239  float d_inner = bin_width*m;
240  float d_outer = d_inner + bin_width;
241  float norm = M_PI*(d_outer*d_outer - d_inner*d_inner)/area;
242 
243  float upper = hist_upper[m]/(norm*upper_expected);
244  float lower = hist_lower[m]/(norm*lower_expected);
245  float total = (hist_upper[m] + hist_lower[m])/
246  (norm*(upper_expected + lower_expected) );
247  cum += (hist_upper[m] + hist_lower[m])/(group1.size()*interval);
248 
249  out << d << "\t"
250  << total << "\t"
251  << upper << "\t"
252  << lower << "\t"
253  << cum << endl;
254 
255  }
256 
257  out << endl; // blank line for gnuplot
258  out.close();
259  // rezero the histograms
260  hist_upper.assign(num_bins, 0.0);
261  hist_lower.assign(num_bins, 0.0);
262 
263  // zero out the area
264  area = 0;
265  }
266 
267  }
