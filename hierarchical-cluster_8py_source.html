40 def fullhelp():
41  print("""
42  Perform hierarchical clustering given a matrix of distances
43 
44  Summary of Arguments
45  rmsd_file : matlab-format text file containing "distances" between all
46  observations. For example, this could be the output of
47  the LOOS program rmsds
48  num_clusters: number of clusters to truncate the clustering
49 
50  Optional arguments
51  --index_file text file containing a list of filenames and which rows
52  of the rmsd_file they correspond to.
53  The format is:
54  START_FRAME END_FRAME FILENAME
55  using 0-based indexing. This could be useful if you
56  passed multiple trajectories to rmsds, but now want to
57  pull out the cluster populations for each individual
58  trajectory.
59  --prefix name to use for the outputted time series and histogram.
60  --prefix "foo" produces "foo.dat" and "foo.hist"
61  --link writes out the linkage matrix to "all.link" or the
62  equivalent name if you supplied a --prefix.
63 
64  Output
65  If you did not pass in an index file or a prefix, this program will produce
66  2 output files, "all.dat" and "all.hist". "all.dat" contains the time
67  series of cluster assignments, while "all.hist" is a normalized histogram of
68  cluster populations. If you supplied a "--prefix" argument, that string
69  will replace "all" in the .dat and .hist filenames.
70 
71  If you did supply an index file, it will additionally supply a .dat and
72  .hist for each file listed in the index file. It will strip off any
73  leading path and anything that looks like a filename extension, so that
74  "/foo/bar/baz.dcd" would produce "baz.dat" and "baz.hist".
75 
76  Technical Details
77 
78  This program is mostly a thin wrapper around scipy's hierarchical clustering
79  routines. It uses the UPGMA algorithm: when trying to decide whether to
80  merge 2 clusters, it takes the average of the distances between all pairs of
81  points in the 2 clusters. Because we supply a distance matrix instead of
82  the actual data points (which would be much bigger, and would make this
83  program much more complicated), we can't use some of the other algorithms,
84  like Ward's algorithm or the median or centroid algorithms.
85 
86  """)

113 num_clusters = args.num_clusters
114 if args.index_file is not None:
115  index_filename = args.index_file
116 else:
117  index_filename = None

119 # read the RMSD file and
120 # convert to condensed upper triangular
121 rmsd = numpy.loadtxt(args.rmsd_file)
122 upper = squareform(rmsd)
123 
124 
125 link = sch.average(upper)
126 
127 if args.link:
128  numpy.savetxt(args.prefix + ".link", link, header=cmd_args)
129 
130 assignments = sch.fcluster(link, num_clusters, criterion='maxclust')
131 
132 # Read the index file if one was supplied
133 indices = {}
134 if index_filename:
135  with(open(index_filename)) as index_file:
136  for line in index_file.readlines():
137  (first, last, filename) = line.split()
138  first = int(first)
139  last = int(last)
140  t = basename(filename)
141  sim, ext = splitext(t)
142  indices[sim] = (first, last)
143 
144 # always produce a histogram of all of the data
145 indices[args.prefix] = (0, len(rmsd)-1)
146 
147 for sim in list(indices.keys()):
148  filename = sim + "_" + str(num_clusters) + ".dat"
149  histname = sim + "_" + str(num_clusters) + ".hist"
150  first, last = indices[sim]
151  sim_slice = assignments[first:last+1]
152  numpy.savetxt(filename, sim_slice.astype(int), fmt="%d", header=cmd_args)
153  sim_hist = numpy.bincount(sim_slice, minlength=num_clusters+1)
154  sim_hist = sim_hist.astype(float)
155  sim_hist /= numpy.add.reduce(sim_hist)
156  numpy.savetxt(histname, sim_hist, header=cmd_args)
