43 string fullHelpMessage(void) {
44  string msg =
45  "\n"
46  "SYNOPSIS\n"
47  "\n"
48  "Extract a frame from a trajectory, writing it out as a PDB\n"
49  "\n"
50  "DESCRIPTION\n"
51  "\n"
52  "Given a model, a trajectory, and a frame number, this tool will extract that\n"
53  " frame and write it out as a PDB. Optionally, a subset of the model can be \n"
54  "extracted. Any LOOS supported model and trajectory type may be used. Note that\n"
55  "frame numbers are zero-based. Negative frame numbers are relative to the end\n"
56  "of the trajectory. Note that you will need to put '--' on the command line\n"
57  "*after* any options to tell the options parse that the negative frame number\n"
58  "is not another command line option.\n"
59  "\n"
60  "EXAMPLES\n"
61  "\n"
62  "\tframe2pdb model.psf simulation.dcd 42 >frame.pdb\n"
63  "Extracts the 43rd frame from the simulation.\n"
64  "\n"
65  "\tframe2pdb -- model.psf simulation.dcd -1 >frame.pdb\n"
66  "Extracts the last frame from the simulation.\n"
67  "\n"
68  "\tframe2pdb --selection 'resid <= 100' model.psf simulation.dcd 13 >frame.pdb\n"
69  "Extracts the 14th frame, only writing out the first 100 residues.\n";
70  return(msg);
71 }

99 int main(int argc, char *argv[]) {
100  string hdr = invocationHeader(argc, argv);
101 
102  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
103  opts::BasicSelection* sopts = new opts::BasicSelection;
104  opts::BasicTrajectory* tropts = new opts::BasicTrajectory;
105  opts::RequiredArguments* ropts = new opts::RequiredArguments;
106  ToolOptions* topts = new ToolOptions;
107  ropts->addArgument("frameno", "frame-number");
108 
109  opts::AggregateOptions options;
110  options.add(bopts).add(sopts).add(tropts).add(topts).add(ropts);
111  if (!options.parse(argc, argv))
112  exit(-1);
113 
114  if (tropts->skip)
115  cerr << "WARNING- --skip is ignored by this tool\n";
116 
117  long frameno = parseStringAs<long>(ropts->value("frameno"));
118  uint frame_index;
119  if (frameno < 0)
120  frame_index = tropts->trajectory->nframes() + frameno;
121  else
122  frame_index = static_cast<uint>(frameno);
123 
124  bool b = tropts->trajectory->readFrame(frame_index);
125  if (!b) {
126  cerr << "Could not read frame " << frame_index << " from trajectory " << tropts->traj_name << endl;
127  exit(-2);
128  }
129  AtomicGroup subset = selectAtoms(tropts->model, sopts->selection);
130  if (!topts->use_bonds)
131  subset.clearBonds();
132 
133  tropts->trajectory->updateGroupCoords(subset);
134  PDB pdb = PDB::fromAtomicGroup(subset);
135  if (sopts->selection != "all")
136  pdb.clearBonds();
137  pdb.remarks().add(hdr);
138  cout << pdb;
139 }
