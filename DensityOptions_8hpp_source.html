55  class BasicWater : public OptionsPackage {
56  /*
57  Grids used as masks are integer grids. They should only contain values from
58  [0,max_grid_mask_value].
59  */
60  static const int max_grid_mask_value = 0xff;
61  public:
62  BasicWater() :
63  pad(1.0),
64  radius(10.0),
65  contacts(5),
66  water_string("name == 'OH2'"),
67  prot_string("name == 'CA'"),
68  grid_name(""),
69  filter_mode("axis"),
70  bulked_spec(""),
71  zrange_spec(""),
72  filter_func(0)
73  { }

75  void addGeneric(po::options_description& opts) {
76  opts.add_options()
77  ("water,W", po::value<std::string>(&water_string)->default_value(water_string), "Water selection")
78  ("prot,P", po::value<std::string>(&prot_string)->default_value(prot_string), "Protein selection")
79  ("pad", po::value<double>(&pad)->default_value(pad), "Pad (for bounding box)")
80  ("bulked", po::value<std::string>(&bulked_spec)->default_value(bulked_spec), "Add bulk water (z-slices between cutoff and bounding box) [pad,zmin:zmax]")
81  ("radius,R", po::value<double>(&radius)->default_value(radius), "Radius (for principal axis filter and radius filter)")
82  ("zrange", po::value<std::string>(&zrange_spec)->default_value(zrange_spec), "Clamp the volume to integrate over in Z (min:max)")
83  ("grid,G", po::value<std::string>(&grid_name)->default_value(grid_name), "Name of grid to use in grid-mode (for internal waters)")
84  ("contacts,C", po::value<uint>(&contacts)->default_value(contacts), "Minimum number of contacts required for contacts filter")
85  ("mode,M", po::value<std::string>(&filter_mode)->default_value(filter_mode), "Mode (axis|box|radius|contacts|core|grid)");
86  }

88  bool postConditions(po::variables_map& map) {
89  if (filter_mode == "axis") {
90  filter_func = new DensityTools::WaterFilterAxis(radius);
91  } else if (filter_mode == "box") {
92  filter_func = new DensityTools::WaterFilterBox(pad);
93  } else if (filter_mode == "radius") {
94  filter_func = new DensityTools::WaterFilterRadius(radius);
95  } else if (filter_mode == "contacts") {
96  filter_func = new DensityTools::WaterFilterContacts(radius, contacts);
97  } else if (filter_mode == "core") {
98  filter_func = new DensityTools::WaterFilterCore(radius);
99  } else if (filter_mode == "grid") {
100  if (grid_name.empty()) {
101  std::cerr << "ERROR - you must specify a grid to use when using grid-mode\n";
102  return(false);
103  }
104 
105  std::ifstream ifs(grid_name.c_str());
106  if (!ifs) {
107  std::cerr << "Error- cannot open grid file '" << grid_name << "' for reading." << std::endl;
108  return(false);
109  }
110 
111  ifs >> the_grid;
112  std::cerr << "Read in grid with size " << the_grid.gridDims() << std::endl;
113  if (!validateGridMask(the_grid)) {
114  std::cerr << "\n***WARNING***WARNING***WARNING***\n\n"
115  << "The grid '" << grid_name
116  << "' does not appear to be a grid mask. Your output will be suspect!\n"
117  << "Make sure the grid is an integer grid (e.g. from blobid or pick_block).\n"
118  << "If there are many unique blobs in the grid, then you may see this warning\n"
119  << "in which case you should reconsider how you ran blobid.\n"
120  << "Use --fullhelp for more information.\n"
121  << "\n***WARNING***WARNING***WARNING***\n\n";
122  }
123 
124 
125 
126  filter_func = new DensityTools::WaterFilterBlob(the_grid);
127 
128  } else {
129  std::cerr << "ERROR - unknown mode " << filter_mode << std::endl;
130  return(false);
131  }
132 
133  // Handle "decoration"
134  if (!zrange_spec.empty()) {
135  double zmin, zmax;
136  int i = sscanf(zrange_spec.c_str(), "%lf:%lf", &zmin, &zmax);
137  if (i != 2) {
138  std::cerr << boost::format("ERROR - unable to parse range '%s'\n") % zrange_spec;
139  return(false);
140  }
141 
142  filter_func = new DensityTools::ZClippedWaterFilter(filter_func, zmin, zmax);
143  }
144 
145  if (!bulked_spec.empty()) {
146  double zmin, zmax, pad;
147  int i = sscanf(bulked_spec.c_str(), "%lf,%lf:%lf", &pad, &zmin, &zmax);
148  if (i != 3) {
149  std::cerr << boost::format("ERROR - unable to parse bulk range '%s'\n") % bulked_spec;
150  return(false);
151  }
152 
153  filter_func = new DensityTools::BulkedWaterFilter(filter_func, pad, zmin, zmax);
154  }
155 
156 
157  return(true);
158  }
159 
160  std::string print() const {
161  std::ostringstream oss;
162 
163  oss << boost::format("water='%s', prot='%s', pad=%f, bulked='%s', radius=%f, contacts=%d, zrange='%s', grid='%s', mode='%s'")
164  % water_string
165  % prot_string
166  % pad
167  % (bulked_spec.empty() ? "undef" : bulked_spec)
168  % radius
169  % contacts
170  % (zrange_spec.empty() ? "undef" : zrange_spec)
171  % (grid_name.empty() ? "undef" : grid_name)
172  % filter_mode;
173 
174  return(oss.str());
175  }

202  bool validateGridMask(const DensityTools::DensityGrid<int>& grid) {
203  for (long i=0; i<grid.size(); ++i)
204  if (grid(i) < 0 || grid(i) > max_grid_mask_value)
205  return(false);
206 
207  return(true);
208  }
