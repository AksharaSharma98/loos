47 string fullHelpMessage(void) {
48  string msg =
49  "\n"
50  "SYNOPSIS\n"
51  "\tCalculate a contact \"heat-map\" between residues in a simulation.\n"
52  "\n"
53  "DESCRIPTION\n"
54  "\n"
55  "\tThis tool will break apart the selection into residues. At each time point\n"
56  "in the trajectory, it will determine if any residues are in contact with each\n"
57  "other. This will be accumulated over the trajectory and a matrix representing\n"
58  "the fractional contacts will be written out. This matrix can be visualized as\n"
59  "a \"heat-map\" using octave/matlab or gnuplot.\n"
60  "\tA contact can be defined in two different ways. It can be defined as occuring when\n"
61  "the distance between any two atoms less than or equal to the\n"
62  "threshold given on the command line. Alternatively, it can be defined as occuring when\n"
63  "the distance between the centers of mass of the two residues is less than or equal\n"
64  "to the threshold.\n"
65  "\n"
66  "EXAMPLES\n"
67  "\n"
68  "\tresidue-contact-map --selection 'segid == \"PROT\"' \\\n"
69  "\t model.pdb simulation.dcd 4.0 >contacts.asc\n"
70  "This example defines a contact when any pair of atoms between a given two residues is\n"
71  "less than or equal to the 4.0 Angstroms. Only residues with segid \"PROT\" are used.\n"
72  "\n"
73  "\tresidue-contact-map --selection 'resid <= 100' --centers 1 \\\n"
74  "\t model.pdb simulation.dcd 6.5 >contacts.asc\n"
75  "This example defines a contact when the centers of mass between two residues is less than\n"
76  "or equal two 6.5 Angstroms. Only the first 100 residues are used.\n"
77  "\n"
78  "SEE ALSO\n"
79  "\trmsds\n";
80 
81  return(msg);
82 }

111 void accumulateFrameUsingCenters(DoubleMatrix& M, const vGroup& residues, const double threshold) {
112 
113  for (uint j=1; j<residues.size(); ++j) {
114  GCoord v(residues[j].centerOfMass());
115 
116  for (uint i=0; i<j; ++i)
117  if (v.distance2(residues[i].centerOfMass()) <= threshold) {
118  M(j, i) += 1;
119  M(i, j) += 1;
120  }
121  }
122 
123  for (uint i=0; i<residues.size(); ++i)
124  M(i, i) += 1;
125 }

128 void accumulateFrameUsingAllAtoms(DoubleMatrix& M, const vGroup& residues, const double threshold) {
129 
130  for (uint j=1; j<residues.size(); ++j) {
131 
132  for (uint i=0; i<j; ++i) {
133  bool flag = true;
134  for (AtomicGroup::const_iterator a = residues[j].begin(); a != residues[j].end() && flag; ++a)
135  for (AtomicGroup::const_iterator b = residues[i].begin(); b != residues[i].end(); ++b)
136  if ((*a)->coords().distance2((*b)->coords()) <= threshold) {
137  flag = false;
138  break;
139  }
140  if (!flag) {
141  M(j, i) += 1;
142  M(i, j) += 1;
143  }
144  }
145  }
146 
147  for (uint i=0; i<residues.size(); ++i)
148  M(i, i) += 1;
149 }

153 int main(int argc, char *argv[]) {
154  string hdr = invocationHeader(argc, argv);
155 
156  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
157  opts::BasicSelection* sopts = new opts::BasicSelection;
158  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
159  ToolOptions* topts = new ToolOptions;
160  opts::RequiredArguments* ropts = new opts::RequiredArguments("threshold", "Distance threshold for contacts");
161 
162  opts::AggregateOptions options;
163  options.add(bopts).add(sopts).add(tropts).add(topts).add(ropts);
164  if (!options.parse(argc, argv))
165  exit(-1);
166 
167  AtomicGroup model = tropts->model;
168  pTraj traj = tropts->trajectory;
169  vector<uint> indices = tropts->frameList();
170 
171  double thresh = parseStringAs<double>(ropts->value("threshold"));
172  thresh *= thresh;
173 
174  AtomicGroup subset = selectAtoms(model, sopts->selection);
175  vGroup residues = subset.splitByResidue();
176 
177  DoubleMatrix M(residues.size(), residues.size());
178  for (vector<uint>::iterator i = indices.begin(); i != indices.end(); ++i) {
179  traj->readFrame(*i);
180  traj->updateGroupCoords(model);
181  if (topts->use_centers)
182  accumulateFrameUsingCenters(M, residues, thresh);
183  else
184  accumulateFrameUsingAllAtoms(M, residues, thresh);
185  }
186 
187  for (ulong i=0; i<residues.size() * residues.size(); ++i)
188  M[i] /= indices.size();
189 
190  writeAsciiMatrix(cout, M, hdr);
191 }
