27  uint Weights::read_weights(const std::string& filename)
28  {
29  std::ifstream ifs(filename.c_str());
30  if (!ifs) {
31  std::cerr << "Cannot open weights file: "
32  << filename
33  << std::endl;
34  throw(FileOpenError(filename));
35  }
36 
37  std::string input;
38  while (getline(ifs, input)) {
39  // skip blank lines and comments beginning with "#"
40  if ( (input.length() == 0) || (input[0] == '#' ) ) {
41  // do nothing
42  }
43  // TODO: we should really let it be any column
44  else {
45  double value = parseStringAs<double>(input);
46  _weights.push_back(value);
47  }
48  }
49  return _weights.size();
50  }

53  uint Weights::read_weights_list(const std::string &filename) {
54  uint num_weights_files = 0;
55  _has_list = true;
56 
57  std::ifstream ifs(filename.c_str());
58  if (!ifs) {
59  throw(FileOpenError(filename));
60  }
61 
62  std::string line, traj_file, weights_file;
63  std::istringstream iss;
64  while (getline(ifs, line)) {
65  // skip blank lines and comments beginning with "#"
66  if ( (line.length() == 0) || (line[0] == '#' ) ) {
67  // do nothing
68  }
69  else {
70  num_weights_files++;
71  iss.str(line);
72  iss >> traj_file >> weights_file;
73  _weights_files[traj_file] = weights_file;
74  }
75  }
76  return num_weights_files;
77  }
