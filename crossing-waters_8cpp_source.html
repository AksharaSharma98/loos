47 class InternalWater
48  {
49  public:
50  InternalWater(pAtom at, int first, double z)
51  : atom(at),
52  entry_frame(first),
53  exit_frame(-1),
54  exited_to_positive(false)
55  {
56  set_entered_side(z);
57  }
58 
59 
60  int entered() const {return(entry_frame);}
61 
62  void set_entered_side(double z)
63  {
64  if (z>0)
65  {
66  entered_from_positive = true;
67  }
68  else
69  {
70  entered_from_positive = false;
71  }
72  }
73 
74  void set_exit_side(double z)
75  {
76  if (z>0)
77  {
78  exited_to_positive = true;
79  }
80  else
81  {
82  exited_to_positive= false;
83  }
84  }
85 
86  bool from_pos() const {return(entered_from_positive);}
87  int exited() const {return(exit_frame);}
88  void exit(const int frame) {exit_frame = frame;}
89 
90  bool crossed()
91  {
92  if (exit_frame > 0) // we have exited the membrane
93  {
94  if ( (entered_from_positive && !exited_to_positive) ||
95  (!entered_from_positive && exited_to_positive) )
96  {
97  return(true);
98  }
99  }
100  return(false);
101  }
102 
103  int lifetime()
104  {
105  if (exit_frame > 0)
106  {
107  return( exited() - entered() );
108  }
109  return(-1);
110  }
111 
112  pAtom get_atom()
113  {
114  return(atom);
115  }
116 
117  int atom_id()
118  {
119  return(atom->id());
120  }
121 
122  const int exit_side() const
123  {
124  if (exited_to_positive)
125  {
126  return +1;
127  }
128  else
129  {
130  return -1;
131  }
132  }
133 
134  private:
135  pAtom atom;
136  int entry_frame;
137  int exit_frame;
138  bool entered_from_positive;
139  bool exited_to_positive;
140  };

144 string fullHelpMessage(void)
145  {
146  string s =
147 "\n"
148 "SYNOPSIS\n"
149 "\n"
150 "Track the rate at which water molecules cross the membrane.\n"
151 "\n"
152 "DESCRIPTION\n"
153 "\n"
154 "This tool measures the rate at which water molecules cross the lipid membrane. \n"
155 "To simplify matters, the tool assumes that the membrane is centered at z=0, \n"
156 "and keeps track of water molecules passing through z=0. To differentiate \n"
157 "between waters passing through the membrane and those that simply pass through \n"
158 "the periodic boundary, the user specifies 2 values, inner_threshold and \n"
159 "outer_threshold, which specify a distance from the membrane center at which \n"
160 "the waters are determined to have entered and exited the membrane, \n"
161 "respectively. The rationale for using 2 thresholds is that we only want to \n"
162 "track waters with a reasonable chance of crossing the membrane, so we use a \n"
163 "restrictive threshold there (the water has to really be in the membrane before \n"
164 "we pay attention), but on the other hand we don't want to say it's out until \n"
165 "it's safely outside the membrane, so we use a larger threshold there. The \n"
166 "optimum choice for these values depends on the thickness of the membrane, but \n"
167 "10 and 20 are a reasonable start.\n"
168 "\n"
169 "There's an additional optional flag to control how water is selected, in\n"
170 "case your naming conventions are different from ours. Just add a selection\n"
171 "string after the outer_threshold. The code will internally remove any\n"
172 "hydrogens, so you don't have to put that in your selection unless you want.\n"
173 "\n"
174 "The output is a list of waters that crossed the membrane, how many frames \n"
175 "each water spent inside the membrane, and the frames it entered and left. The \n"
176 "final column is either 1 or -1; the former indicated that the water exited on \n"
177 "the +z side of the membrane, the latter the -z side.\n"
178 "\n"
179 "EXAMPLE\n"
180 "\n"
181 "crossing-waters system.psf traj.dcd 10.0 20.0\n"
182 "\n"
183 "This will read system.psf and the trajectory file traj.dcd, and use 10 and 20 \n"
184 "angstroms as the inner and outer threshold. The output will look like this:\n"
185 "\n"
186 "# crossing-waters 'system.psf' 'traj.dcd' '10' '20' - alan (Tue Mar 13 14:32:49 2012) {/directory/you/were/working/in} [2.0.0 120313]\n"
187 "# Total frames = 719\n"
188 "#AtomID Lifetime Entered Exited ExitedPositive\n"
189 "38546 2 0 2 -1\n"
190 "25136 2 4 6 1\n"
191 "25856 1 5 6 -1\n"
192 "35909 1 8 9 -1\n"
193 "39665 1 8 9 -1\n"
194 "\n"
195  ;
196  return(s);
197  }

200 int main (int argc, char *argv[])
201 {
202 if ( (argc >=2) )
203  {
204  if (string(argv[1]) == string("-h"))
205  {
206  Usage();
207  exit(-1);
208  }
209  else if (string(argv[1]) == string("--fullhelp"))
210  {
211  cerr << fullHelpMessage() << endl;
212  exit(-1);
213  }
214  }
215 if (argc < 5)
216  {
217  Usage();
218  exit(-1);
219  }
220 
221 cout << "# " << invocationHeader(argc, argv) << endl;
222 
223 AtomicGroup system = createSystem(argv[1]);
224 pTraj traj = createTrajectory(argv[2], system);
225 greal inner_threshold = atof(argv[3]);
226 greal outer_threshold = atof(argv[4]);
227 
228 bool manual_water = false;
229 char *water_selection;
230 if (argc > 5)
231  {
232  manual_water = true;
233  water_selection = argv[5];
234  }
235 
236 // Select the water oxygens
237 HeavySolventSelector water_heavy; // select atoms which are both water
238  // and not hydrogens
239 
240 AtomicGroup water;
241 if (manual_water) // apply the supplied selection, but ensure no hydrogens
242  {
243  water = selectAtoms(system, water_selection);
244  water = selectAtoms(water, "!hydrogen");
245  }
246 else
247  {
248  water = system.select(water_heavy); // apply the selection
249  }
250 
251 
252 // Set up a map to store the waters
253 map<pAtom,InternalWater> internal_waters;
254 vector<InternalWater> exited_waters;
255 
256 int frame = 0;
257 while (traj->readFrame())
258  {
259  traj->updateGroupCoords(system);
260  bool inside_inner, inside_outer;
261 
262  // loop over water oxygens
263  pAtom w;
264  AtomicGroup::Iterator iter(water);
265  while ( (w=iter()) )
266  {
267  double z = w->coords().z();
268  if (fabs(z) < inner_threshold)
269  {
270  inside_inner = true;
271  inside_outer = true;
272  }
273  else if (fabs(z) < outer_threshold)
274  {
275  inside_inner = false;
276  inside_outer = true;
277  }
278  else
279  {
280  inside_inner = false;
281  inside_outer = false;
282  }
283 
284  // is this a water which is currently inside?
285  map<pAtom,InternalWater>::iterator wat = internal_waters.find(w);
286  if (wat != internal_waters.end())
287  {
288  // Yes, the water is listed as internal
289  if (inside_outer)
290  {
291  // do nothing -- it's still inside
292  }
293  else
294  {
295  // the water has left -- mark it so and clean up
296  InternalWater &i = wat->second;
297  i.exit(frame);
298  i.set_exit_side(z);
299  exited_waters.push_back(i);
300  internal_waters.erase(w);
301  // TODO: debugging code, don't leave in
302  //if (internal_waters.find(w) != internal_waters.end())
303  // {
304  // throw(runtime_error("Erase didn't erase"));
305  // }
306  }
307  }
308  else
309  // No, the water wasn't previously inside
310  {
311  if (inside_inner)
312  {
313  // create a new InternalWater and store it
314  internal_waters.insert(
315  pair<pAtom,InternalWater>(w, InternalWater(w, frame, z)));
316  }
317  else
318  {
319  // not inside: do nothing
320  }
321  }
322  }
323  frame++;
324  }
325 
326 vector<InternalWater>::iterator wat;
327 cout << "# Total frames = " << frame << endl;
328 cout << "# Number of waters = " << water.size() << endl;
329 cout << "#AtomID\tLifetime\tEntered\tExited\tExitedPositive" << endl;
330 for (wat = exited_waters.begin(); wat != exited_waters.end(); wat++)
331  {
332  if (wat->crossed())
333  {
334  cout << wat->atom_id() << "\t"
335  << wat->lifetime() << "\t"
336  << wat->entered() << "\t"
337  << wat->exited() << "\t"
338  << wat->exit_side()
339  << endl;
340  }
341  }
342 
343 }
