119 string fullHelpMessage(void)
120  {
121  string s =
122 "\n"
123 " SYNOPSIS\n"
124 "\n"
125 " Report the fraction of native contacts found over the course of \n"
126 " a trajectory.\n"
127 "\n"
128 " DESCRIPTION\n"
129 "\n"
130 " The purpose of this tool is to compute the fraction of native contacts\n"
131 " found on average over the course of trajectory. This is intended for\n"
132 " use in protein or RNA systems, as a way of tracking the degree to which\n"
133 " the molecule is folded. \n"
134 "\n"
135 " By default, the model file provided on the command line has coordinates, then \n"
136 " those coordinates are used to define \"native\" contacts. \n"
137 " If the model file doesn't have coordinates, then the first frame of the\n"
138 " trajectory is used.\n"
139 "\n"
140 " Alternatively, you can supply a separate structure containing reference\n"
141 " coordinates (e.g. a pdb file with the original crystal coordinates).\n"
142 " The only restriction is that the same selection string that picks out\n"
143 " the residues of interest from the system file must also apply to the \n"
144 " reference file.\n"
145 "\n"
146 " The set of atoms to be analyzed is specified on the\n"
147 " command line, which is then split by residue. If the centers of mass\n"
148 " of two residues are within the cutoff distance specified on the command\n"
149 " line, then those two residues are a native contact. The same criterion\n"
150 " is applied at each successive frame.\n"
151 "\n"
152 " Note: This code does not take periodicity into account by default,\n"
153 " because in most cases (e.g. a protein or RNA) the molecule will be \n"
154 " in a single unit cell. If you want periodicity, add the flag \n"
155 " '--periodic' on the command line. If you give this flag and supply an \n"
156 " initial structure that does not have box information, you will get a \n"
157 " warning, and the initial identification of contacts will be done without\n"
158 " using the periodic image. If this is not the desired behavior, you'll \n"
159 " need to add the box information to the initial structure by hand first,\n"
160 " or use the first frame of the trajectory as the reference.\n"
161 "\n"
162 " The --exclude-consecutive option causes the code to ignore residues\n"
163 " consecutive in sequence when computing the list of native contacts.\n"
164 " Note: this is done in a naive way, without checking that the consecutive\n"
165 " residues are part of the same chain. "
166 "\n"
167 " EXAMPLE\n"
168 "\n"
169 " native_contacts model.psf traj.dcd 5 --selection 'segname == \"PROT\"'\n"
170 "\n"
171 " This uses model.psf as the system file, traj.dcd as the trajectory,\n"
172 " sets the cutoff for a native contact at 5 angstroms, and operates on \n"
173 " the segment called PROT. Since PSF files don't have coordinates, the \n"
174 " first frame of the trajectory will be used to define which contacts \n"
175 " are native.\n"
176 "\n"
177 " If no selection string is provided, then the default is to use\n "
178 " 'name == \"CA\"'."
179 "\n"
180 " In addition, one can select just the sidechains using the\n "
181 " --exclude-backbone flag; this can be combined with other selections.\n"
182 " Turn it one with --exclude-backbone 1 \n"
183 "\n"
184 " If you supply the \"--outfile\" option, you will also get a time series for \n"
185 " all of the individual pairs of residues.\n"
186 "\n"
187 " If you supply the \"--per-residue FILENAME\", the program will output \n"
188 " the average fractional native contacts for each residue to FILENAME.\n"
189 " Residues with no native contacts will have a value of -1.\n"
190 "\n"
191  ;
192  return(s);
193  }

195 int main (int argc, char *argv[])
196 {
197 
198 string header = invocationHeader(argc, argv);
199 
200 opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
201 opts::BasicTrajectory* tropts = new opts::BasicTrajectory;
202 opts::BasicSelection* sopts = new opts::BasicSelection("name == \"CA\"");
203 opts::RequiredArguments* ropts = new opts::RequiredArguments;
204 ropts->addArgument("cut", "cutoff");
205 
206 ToolOptions* topts = new ToolOptions;
207 
208 opts::AggregateOptions options;
209 options.add(bopts).add(tropts).add(ropts).add(sopts).add(topts);
210 
211 if (!options.parse(argc, argv))
212  {
213  exit(-1);
214  }
215 
216 cout << "# " << header << endl;
217 
218 AtomicGroup system = tropts->model;
219 pTraj traj = tropts->trajectory;
220 
221 double cutoff = parseStringAs<double>(ropts->value("cut"));
222 double cut2 = cutoff*cutoff;
223 
224 
225 AtomicGroup sel = selectAtoms(system, sopts->selection);
226 
227 if (topts->exclude_backbone)
228  {
229  BackboneSelector backbone;
230  NotSelector sidechains(backbone);
231  sel = sel.select(sidechains);
232  }
233 
234 vector<AtomicGroup> residues = sel.splitByResidue();
235 
236 // If they asked for output of individual contacts, set it up
237 ofstream output;
238 if (topts->do_output)
239  {
240  output.open(topts->outfile.c_str());
241  if (!output.is_open() )
242  {
243  throw(runtime_error("couldn't open output file"));
244  }
245  }
246 
247 // Figure out what to use as a reference structure
248 // --If one was supplied on the command line use that, and
249 // use the same selections applied to the main system to
250 // select out the equivalent group.
251 // --If there was no reference structure, check to see if the
252 // model file has coordinates, and use those.
253 // --If neither is true, use frame 0 of the trajectory.
254 if (topts->use_reference)
255  {
256  AtomicGroup reference = createSystem(topts->reference);
257  AtomicGroup ref_sel = selectAtoms(reference, sopts->selection);
258  if (topts->exclude_backbone)
259  {
260  BackboneSelector backbone;
261  NotSelector sidechains(backbone);
262  ref_sel = ref_sel.select(sidechains);
263  }
264  // copy the coordinates from ref_sel to sel, after sanity checking
265  if (ref_sel.size() != sel.size())
266  {
267  cerr << "Selection from the reference file wasn't the same size as\n"
268  << "the selection from the main system. You must be able to use\n"
269  << "the same selection string on both systems."
270  << endl;
271  exit(1);
272  }
273  sel.copyCoordinatesFrom(ref_sel);
274  }
275 else if ( !(sel[0]->checkProperty(Atom::coordsbit)) )
276  {
277  traj->readFrame(0);
278  traj->updateGroupCoords(system);
279  }
280 
281 bool use_periodicity_for_reference = topts->use_periodicity;
282 if (topts->use_periodicity && !system.isPeriodic())
283  {
284  use_periodicity_for_reference = false;
285  cerr << "Warning: you requested periodicty, but the reference structure is not periodic" << endl;
286  cerr << "Periodicity will _not_ be used when computing the reference contacts, " << endl;
287  cerr << "but _will_ be used for the trajectory frames." << endl;
288  }
289 
290 // Compute the centers of mass of the selections
291 uint num_residues = residues.size();
292 vector<GCoord> centers_of_mass(num_residues);
293 for (uint i=0; i<num_residues; i++)
294  {
295  centers_of_mass[i] = residues[i].centerOfMass();
296  }
297 
298 vector<vector<uint> > contacts;
299 vector<uint>total_contacts_per_residue(num_residues);
300 vector<uint>contacts_per_residue(num_residues);
301 
302 GCoord box = system.periodicBox();
303 
304 int step = 1;
305 if (topts->exclude_consecutive)
306  {
307  step = 2;
308  }
309 // Find contacts within the threshold distance
310 for (uint i=0; i<num_residues-step; i++)
311  {
312  for (uint j=i+step; j< num_residues; j++)
313  {
314  GCoord diff = centers_of_mass[j] - centers_of_mass[i];
315  if (use_periodicity_for_reference)
316  {
317  diff.reimage(box);
318  }
319  if (diff.length2() <= cut2)
320  {
321  vector<uint> v(2);
322  v[0] = i;
323  v[1] = j;
324  contacts.push_back(v);
325  cout << "# " << (residues[i][0])->resid() << "\t"
326  << (residues[j][0])->resid() << endl;
327  if (topts->do_output)
328  {
329  output << "# " << (residues[i][0])->resid() << "\t"
330  << (residues[j][0])->resid() << endl;
331 
332  }
333 
334  // Store the total number of contacts for each residue
335  if (topts->do_per_residue)
336  {
337  total_contacts_per_residue[i] += 1;
338  total_contacts_per_residue[j] += 1;
339  }
340  }
341  }
342  }
343 
344 // Number of native contacts, as a float because we'll need
345 // to do floating point arithmatic with it later anyway
346 float num_native_contacts = (float) contacts.size();
347 cout << "# Total native contacts: " << num_native_contacts << endl;
348 
349 bool is_periodic;
350 if (topts->use_periodicity && traj->hasPeriodicBox())
351  {
352  is_periodic = true;
353  }
354 else if (topts->use_periodicity && !(traj->hasPeriodicBox()))
355  {
356  cerr << "Warn: you requested periodicity, but your trajectory isn't periodic." << endl;
357  cerr << "The calculation will proceed _ignoring_ periodicity." << endl;
358  is_periodic = false;
359  }
360 
361 // Loop over structures in the trajectory
362 vector<vector<uint> >::iterator p;
363 int frame = 0;
364 while (traj->readFrame())
365  {
366  traj->updateGroupCoords(system);
367  box = system.periodicBox();
368 
369  // Loop over contacts from the native structure
370  int num_contacts = 0;
371  for (p=contacts.begin(); p!= contacts.end(); p++)
372  {
373  uint r1 = p->at(0);
374  uint r2 = p->at(1);
375  GCoord c1 = residues[r1].centerOfMass();
376  GCoord c2 = residues[r2].centerOfMass();
377  GCoord diff = c2 - c1;
378  if (is_periodic)
379  {
380  diff.reimage(box);
381  }
382  if (diff.length2() <= cut2)
383  {
384  num_contacts++;
385  if (topts->do_output) output << "1\t";
386  if (topts->do_per_residue)
387  {
388  contacts_per_residue[r1]++;
389  contacts_per_residue[r2]++;
390  }
391  }
392  else
393  {
394  if (topts->do_output) output << "0\t";
395  }
396  }
397  float fraction = num_contacts / num_native_contacts;
398  cout << frame << "\t" << fraction << endl;
399  if (topts->do_output) output << endl;
400  frame++;
401  }
402 
403 // Output total contacts per residue
404 if (topts->do_per_residue)
405  {
406  ofstream per_residue_stream;
407  per_residue_stream.open(topts->per_residue_filename.c_str());
408  if (!per_residue_stream.is_open())
409  {
410  throw(runtime_error("couldn't open per-residue output file"));
411  }
412  per_residue_stream << "# Residue\tAveContacts\tTotalContacts" << endl;
413  for (uint i=0; i < num_residues; ++i)
414  {
415  int real_residue = (residues[i][0])->resid();
416  per_residue_stream << real_residue << "\t";
417  if (total_contacts_per_residue[i])
418  {
419  double ave = static_cast<double>(contacts_per_residue[i])/
420  (total_contacts_per_residue[i] * frame);
421  per_residue_stream << ave;
422  }
423  else
424  {
425  per_residue_stream << "-1";
426  }
427  per_residue_stream << "\t" << total_contacts_per_residue[i] << endl;
428  }
429  per_residue_stream.close();
430  }
431 
432 }
