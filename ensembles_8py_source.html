14 def averageStructure(traj):
15  """Returns the average structure for a trajectory.
16  >>> avg = loos.pyloos.averageStructure(traj)
17  """
18  avg = numpy.zeros((len(traj.frame()), 3))
19  for frame in traj:
20  coords = frame.getCoords()
21  avg += coords
22  avg /= len(traj)
