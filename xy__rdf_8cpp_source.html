150 split_mode parseSplit(const string &split_by)
151  {
152  split_mode split;
153  // figure out the splitting mode
154  if (!split_by.compare("by-residue"))
155  {
156  split = BY_RESIDUE;
157  }
158  else if (!split_by.compare("by-segment"))
159  {
160  split = BY_SEGMENT;
161  }
162  else if (!split_by.compare("by-molecule"))
163  {
164  split = BY_MOLECULE;
165  }
166  else
167  {
168  cerr << "--split-mode must be: by-residue|by-segment|by-molecule"
169  << endl;
170  cerr << "If unset, defaults to by-molecule";
171  cerr << endl;
172  exit(-1);
173  }
174  return (split);
175  }

177 string fullHelpMessage(void)
178 {
179 string s =
180  "\n"
181  "SYNOPSIS\n"
182  "\n"
183  "Compute a radial distribution function in the x-y plane\n"
184  "\n"
185  "DESCRIPTION\n"
186  "\n"
187  "This tool is intended primarily for analyzing lateral structure of\n"
188  "membrane systems. As with rdf, it operates primarily on groups of atoms.\n"
189  "There are 3 ways to group the atoms, controlled by the\n"
190  "arguments to --split-mode: \n"
191  "\n"
192  " by-residue: the selection is split into unique residues\n"
193  " by-molecule: the selection is split into unique molecules (only \n"
194  " available if the system file contains connectivity \n"
195  " information)\n"
196  " by-segment: the selection is split using the segid (this is present \n"
197  " in CHARMM/NAMD/XPLOR derived files, and some PDB files)\n"
198  "\n"
199  "The default mode if --split-mode isn't set is \"by-molecule\".\n"
200  "In each case, the splitting is performed _before_ the selection is \n"
201  "performed. \n"
202  "\n"
203  "The distance is then computed between the centers of mass of the grouped\n"
204  "objects, only considering the x and y coordinates. The program treats\n"
205  "the two leaflets of the separately, based on the sign of the z-coordinate\n"
206  "of the center of mass of the selection in the first frame; this can \n"
207  "cause problems if the membrane has not already been centered at the \n"
208  "origin (the merge-traj tool can do this for you).\n"
209  "\n"
210  "If one or both of the selections should be included in both leaflets\n"
211  "(e.g. a transmembrane helix or protein), the user should specify \n"
212  "the --sel1-spans or --sel2-spans flags (applying to the first and \n"
213  "second selections respectively). Otherwise, the selection will \n"
214  "be included in only one leaflet, which could lead to non-sensical \n"
215  "results. Note: this flag is applied to _all_ of the components \n"
216  "in the selection --- they are each assumed to span the membrane.\n"
217  "\n"
218  "The --timeseries flag lets you track the evolution of the RDF over time.\n"
219  "\n"
220  "The --reselect flag causes the program to recompute which leaflet\n"
221  "each molecule is in at each time step. This will impose a small \n"
222  "overhead, but is necessary if you're dealing with molecules that \n"
223  "can flip from one leaflet to the other.\n"
224  "\n"
225  "EXAMPLE\n"
226  "\n"
227  "To look at the distribution of PE lipid headgroups in a lipid\n"
228  "bilayer, you might use a command line like\n"
229  "\n"
230  "xy_rdf model-file traj-file 'resname == \"PEGL\"' 'resname == \"PEGL\"' 0 40 40\n"
231  " --split-mode=by-molecule\n"
232  "\n"
233  "Assuming the CHARMM27-style lipid naming, the headgroup would be its own\n"
234  "residue with name \"PEGL\", and the result would be the lateral RDF for \n"
235  "the headgroup centers of mass. \n"
236  "\n"
237  "As with the other rdf tools (rdf, atomic-rdf), histogram-min,\n"
238  "histogram-max, and histogram-bins specify the range over which the\n"
239  "radial distribution function is computed and the number of bins used. \n"
240  "\n"
241  "The --timeseries flag lets you track the evolution of the rdf over time, by\n"
242  "writing out a windowed average as it is accumulated. So, the adding the \n"
243  "flags\n"
244  "\n"
245  " --timeseries 100 --timeseries-directory \"foo\"\n"
246  "\n"
247  "to the end of the command above would cause the program to write out a \n"
248  "new average every 100 frames considering only the frames in that interval.\n"
249  "The files will appear in the directory \"foo\", with names rdf_0.dat, \n"
250  "rdf_1.dat, etc. The program does not attempt to create \"foo\" if it \n"
251  "doesn't exist, and instead will simply exit.\n"
252  "\n"
253  "Note: the 5th column (\"Cum\") is not a density like the other values, \n"
254  "but rather the absolute number of molecules of the second selection \n"
255  "found around the first selection.\n"
256  ;

258  return (s);

261 void assign_leaflet(vector<AtomicGroup> &molecules,
262  vector<AtomicGroup> &upper,
263  vector<AtomicGroup> &lower,
264  const bool spans
265  )
266  {
267  upper.clear();
268  lower.clear();
269  if (spans)
270  {
271  upper = molecules;
272  lower = molecules;
273  return;
274  }
275 
276  for (unsigned int i = 0; i < molecules.size(); i++)
277  {
278  GCoord c = molecules[i].centerOfMass();
279  if (c.z() >=0.0)
280  {
281  upper.push_back(molecules[i]);
282  }
283  else
284  {
285  lower.push_back(molecules[i]);
286  }
287  }
288  }

290 int main (int argc, char *argv[])
291 {
292 
293 // parse the command line options
294 string hdr = invocationHeader(argc, argv);
295 opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
296 opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
297 opts::WeightsOptions* wopts = new opts::WeightsOptions;
298 ToolOptions* topts = new ToolOptions;
299 
300 opts::AggregateOptions options;
301 options.add(bopts).add(tropts).add(topts).add(wopts);
302 if (!options.parse(argc, argv))
303  exit(-1);
304 
305 
306 
307 // parse the split mode, barf if you can't do it
308 split_mode split=parseSplit(split_by);
309 
310 cout << "# " << invocationHeader(argc, argv) << endl;
311 
312 // copy the command line variables to real variable names
313 AtomicGroup system = tropts->model;
314 pTraj traj = tropts->trajectory;;
315 if (!(system.isPeriodic() || traj->hasPeriodicBox()))
316  {
317  cerr << "Error- Either the model or the trajectory must have periodic box information.\n";
318  exit(-1);
319  }
320 
321 // Attach trajectory to weights
322 if (wopts->has_weights)
323  {
324  wopts->weights.add_traj(traj);
325  }
326 
327 double bin_width = (hist_max - hist_min)/num_bins;
328 
329 AtomicGroup group1 = selectAtoms(system, selection1);
330 if (group1.empty())
331  {
332  cerr << "Error- no atoms selected by '" << selection1 << "'\n";
333  exit(-1);
334  }
335 group1.pruneBonds();
336 
337 AtomicGroup group2 = selectAtoms(system, selection2);
338 if (group2.empty())
339  {
340  cerr << "Error- no atoms selected by '" << selection2 << "'\n";
341  exit(-1);
342  }
343 group2.pruneBonds();
344 
345 // Split the groups into chunks, depending on how the user asked
346 // us to.
347 vector<AtomicGroup> g1_mols, g2_mols;
348 if (split == BY_MOLECULE)
349  {
350  g1_mols = group1.splitByMolecule();
351  g2_mols = group2.splitByMolecule();
352  }
353 else if (split == BY_RESIDUE)
354  {
355  g1_mols = group1.splitByResidue();
356  g2_mols = group2.splitByResidue();
357  }
358 else if (split == BY_SEGMENT)
359  {
360  g1_mols = group1.splitByUniqueSegid();
361  g2_mols = group2.splitByUniqueSegid();
362  }
363 
364 // read the initial coordinates into the system
365 traj->updateGroupCoords(system);
366 
367 // Now that we have some real coordinates, we need to subdivide the groups
368 // one more time, into upper and lower leaflets. This assumes that the
369 // coordinates are properly centered and imaged.
370 vector<AtomicGroup> g1_upper, g1_lower;
371 vector<AtomicGroup> g2_upper, g2_lower;
372 
373 assign_leaflet(g1_mols, g1_upper, g1_lower, sel1_spans);
374 assign_leaflet(g2_mols, g2_upper, g2_lower, sel2_spans);
375 
376 
377 // Create 2 histograms -- one for top, one for bottom
378 // Also create 2 histograms to store the total
379 vector<double> hist_lower, hist_upper;
380 vector<double> hist_lower_total, hist_upper_total;
381 hist_lower.reserve(num_bins);
382 hist_upper.reserve(num_bins);
383 hist_lower_total.reserve(num_bins);
384 hist_upper_total.reserve(num_bins);
385 hist_lower.insert(hist_lower.begin(), num_bins, 0.0);
386 hist_upper.insert(hist_upper.begin(), num_bins, 0.0);
387 hist_lower_total.insert(hist_lower_total.begin(), num_bins, 0.0);
388 hist_upper_total.insert(hist_upper_total.begin(), num_bins, 0.0);
389 
390 
391 double min2 = hist_min*hist_min;
392 double max2 = hist_max*hist_max;
393 
394 // loop over the frames of the traj file
395 double area = 0.0;
396 double interval_area = 0.0;
397 double cum_upper_pairs = 0.0;
398 double cum_lower_pairs = 0.0;
399 double interval_upper_pairs = 0.0;
400 double interval_lower_pairs = 0.0;
401 
402 vector<uint> framelist = tropts->frameList();
403 uint framecnt = framelist.size();
404 
405 
406 for (uint index = 0; index<framecnt; ++index)
407  {
408  // update coordinates and periodic box
409  traj->readFrame(framelist[index]);
410  traj->updateGroupCoords(system);
411 
412  // If this is a reweighting calculation, get the weight.
413  // Otherwise, use 1.0
414  double weight = 1.0;
415  if (wopts->has_weights)
416  {
417  weight = wopts->weights();
418  wopts->weights.accumulate();
419  }
420 
421  GCoord box = system.periodicBox();
422  area += weight*(box.x() * box.y());
423  interval_area += weight*(box.x() * box.y());
424 
425  if (reselect_leaflet)
426  {
427  assign_leaflet(g1_mols, g1_upper, g1_lower, sel1_spans);
428  assign_leaflet(g2_mols, g2_upper, g2_lower, sel2_spans);
429  }
430 
431  // compute the distribution of g2 around g1 for the lower leaflet
432  for (unsigned int j = 0; j < g1_lower.size(); j++)
433  {
434  GCoord p1 = g1_lower[j].centerOfMass();
435  for (unsigned int k = 0; k < g2_lower.size(); k++)
436  {
437  // skip "self" pairs
438  if (g1_lower[j] == g2_lower[k])
439  {
440  continue;
441  }
442 
443  cum_lower_pairs+=weight;
444  interval_lower_pairs+=weight;
445 
446  GCoord p2 = g2_lower[k].centerOfMass();
447  GCoord displ = (p2 - p1);
448  displ.reimage(box);
449  double d2 = (displ.x() * displ.x()) + (displ.y() * displ.y());
450  if ( (d2 < max2) && (d2 > min2) )
451  {
452  double d = sqrt(d2);
453  int bin = int((d-hist_min)/bin_width);
454  hist_lower[bin]+=weight;
455  }
456  }
457  }
458 
459  // compute the distribution of g2 around g1 for the upper leaflet
460  for (unsigned int j = 0; j < g1_upper.size(); j++)
461  {
462  GCoord p1 = g1_upper[j].centerOfMass();
463  for (unsigned int k = 0; k < g2_upper.size(); k++)
464  {
465  // skip "self" pairs
466  if (g1_upper[j] == g2_upper[k])
467  {
468  continue;
469  }
470 
471  cum_upper_pairs+=weight;
472  interval_lower_pairs+=weight;
473 
474  GCoord p2 = g2_upper[k].centerOfMass();
475  GCoord displ = (p2 - p1);
476  displ.reimage(box);
477  double d2 = (displ.x() * displ.x()) + (displ.y() * displ.y());
478  if ( (d2 < max2) && (d2 > min2) )
479  {
480  double d = sqrt(d2);
481  int bin = int((d-hist_min)/bin_width);
482  hist_upper[bin]+=weight;
483  }
484  }
485  }
486 
487 
488  // if requested, write out timeseries as well
489  if (timeseries_interval && (index % timeseries_interval == 0))
490  {
491  interval_area /= timeseries_interval;
492  double upper_expected = interval_upper_pairs / interval_area;
493  double lower_expected = interval_lower_pairs / interval_area;
494 
495 
496  // create the output file
497  ostringstream outfilename;
498  outfilename << output_directory << "/" << "rdf_" << index << ".dat";
499  ofstream out(outfilename.str().c_str());
500  if (out.fail())
501  {
502  cerr << "couldn't open " << outfilename.str() << " ... exiting" << endl;
503  exit(-1);
504  }
505  out << "# Dist\tTotal\tUpper\tLower\tCum" << endl;
506  double cum = 0.0;
507  for (int m = 0; m < num_bins; m++)
508  {
509  double d = bin_width*(m + 0.5);
510 
511  double d_inner = bin_width*m;
512  double d_outer = d_inner + bin_width;
513  double norm = M_PI*(d_outer*d_outer - d_inner*d_inner);
514 
515  double upper = 0.0;
516  if (interval_upper_pairs > 0)
517  {
518  upper = hist_upper[m]/(norm*upper_expected);
519  }
520 
521  double lower = 0.0;
522  if (interval_lower_pairs > 0)
523  {
524  lower = hist_lower[m]/(norm*lower_expected);
525  }
526 
527  double total = (hist_upper[m] + hist_lower[m])/
528  (norm*(upper_expected + lower_expected) );
529  cum += (hist_upper[m] + hist_lower[m])/(group1.size()*timeseries_interval);
530 
531  out << d << "\t"
532  << total << "\t"
533  << upper << "\t"
534  << lower << "\t"
535  << cum << endl;
536 
537  }
538 
539  out << endl; // blank line for gnuplot
540  out.close();
541 
542  // sum up the total histograms
543  for (int i=0; i<num_bins; ++i)
544  {
545  hist_upper_total[i] += hist_upper[i];
546  hist_lower_total[i] += hist_lower[i];
547  }
548 
549  // rezero the histograms
550  hist_upper.assign(num_bins, 0.0);
551  hist_lower.assign(num_bins, 0.0);
552 
553  // zero out the area
554  interval_area = 0.0;
555  interval_upper_pairs = 0;
556  interval_lower_pairs = 0;
557  }
558 
559  }
560 
561 // normalize the area
562 // Not necessary if we're doing reweighting, since they're already
563 // normalized
564 if (!wopts->has_weights) area /= framecnt;
565 
566 // If we didn't write timeseries, then we need to copy the interval histograms
567 // to the total ones. If we did, we need to add in the additional data points
568 // since the last time we wrote out a times series file
569 if (!timeseries_interval)
570  {
571  hist_lower_total = hist_lower;
572  hist_upper_total = hist_upper;
573  }
574 else if (framecnt % timeseries_interval != 0)
575  {
576  for (int i=0; i< num_bins; i++)
577  {
578  hist_lower_total[i] += hist_lower[i];
579  hist_upper_total[i] += hist_upper[i];
580  }
581  }
582 
583 double upper_expected = cum_upper_pairs / area;
584 double lower_expected = cum_lower_pairs / area;
585 
586 // Output the results
587 cout << "# Dist\tTotal\tUpper\tLower\tCum" << endl;
588 double cum = 0.0;
589 for (int i = 0; i < num_bins; i++)
590  {
591  double d = bin_width*(i + 0.5);
592 
593  double d_inner = bin_width*i;
594  double d_outer = d_inner + bin_width;
595  double norm = M_PI*(d_outer*d_outer - d_inner*d_inner);
596 
597  double upper = 0.0;
598  if (cum_upper_pairs > 0)
599  {
600  upper = hist_upper_total[i]/(norm*upper_expected);
601  }
602 
603  double lower = 0.0;
604  if (cum_lower_pairs > 0)
605  {
606  lower = hist_lower_total[i]/(norm*lower_expected);
607  }
608 
609  double total = (hist_upper_total[i] + hist_lower_total[i])/
610  (norm*(upper_expected + lower_expected));
611  if (wopts->has_weights)
612  {
613  upper /= wopts->weights.totalWeight();
614  lower /= wopts->weights.totalWeight();
615  total /= wopts->weights.totalWeight();
616  }
617 
618  double cum_increment = (hist_upper_total[i] + hist_lower_total[i]) /
619  group1.size();
620  if (wopts->has_weights)
621  {
622  cum_increment *= framecnt / wopts->weights.totalWeight();
623  }
624 
625  cum += cum_increment;
626 
627  cout << d << "\t"
628  << total << "\t"
629  << upper << "\t"
630  << lower << "\t"
631  << cum << endl;
632 
633  }
