38 string fullHelpMessage(void)
39  {
40  string s =
41 "\n"
42 " SYNOPSIS\n"
43 "\n"
44 " Read a trajectory and reimage it such that each molecule has its\n"
45 " centroid in the central box.\n"
46 "\n"
47 " DESCRIPTION\n"
48 "\n"
49 " This tool reads a trajectory and processes it to produce a new \n"
50 " trajectory in DCD format where each molecule has its centroid in\n"
51 " the central image.\n"
52 " \n"
53 " This operation does not make a lot of sense if the system file \n"
54 " does not contain connectivity information; it will warn you \n"
55 " if you invoke it without connectivity, but will run.\n"
56 "\n"
57 " If the trajectory has information on box size built in to it, that\n"
58 " box data is used for the reimaging. If not, the periodicity information\n"
59 " may be read from the model file (e.g. a CRYSTL line from a PDB file).\n"
60 " Alternatively, the user can provide box size information on the command \n"
61 " line by supplying 3 extra arguments. If this is done, the information\n"
62 " overrides anything supplied in the trajectory or model file.\n"
63 "\n"
64 " Note: this tool is largely redundant with merge-traj and recenter-traj \n"
65 " (which also have additional capabilities), and may at some point \n"
66 " be deprecated.\n"
67 "\n"
68 " EXAMPLE\n"
69 "\n"
70 " reimage-by-molecule model.psf input_traj.dcd output_traj.dcd \n"
71 "\n"
72 " This reads the system information from model.psf, operates on \n"
73 " input_traj.dcd (which presumably has periodicity information), and \n"
74 " writes output_traj.dcd, which does have periodicity information.\n"
75 "\n"
76 " reimage-by-molecule model.psf input_traj.dcd output_traj.xtc 55 77 100\n"
77 "\n"
78 " This does essentially the same thing, but asserts that the periodic\n"
79 " box is constant with x-dimension 55 angstrom, y-dimension 77 angstroms,\n"
80 " and z-dimension 100 angstroms. The trajectory is also converted to\n"
81 " the GROMACS XTC format."
82 "\n"
83  ;
84  return(s);
85  }

94 int main(int argc, char *argv[])
95 {
96  if ( (argc >=2) )
97  {
98  if (string(argv[1]) == string("-h"))
99  {
100  Usage();
101  exit(-1);
102  }
103  else if (string(argv[1]) == string("--fullhelp"))
104  {
105  cerr << fullHelpMessage() << endl;
106  Usage();
107  exit(-1);
108  }
109  }
110  if (argc < 4)
111  {
112  Usage();
113  exit(-1);
114  }
115 
116  string hdr = invocationHeader(argc, argv);
117 
118  AtomicGroup model = createSystem(argv[1]);
119  if (!model.hasBonds())
120  {
121  cerr << "***WARNING***\nThe model does not have connectivity,\nso your results may not be what you expect.\n";
122  }
123 
124  pTraj traj = createTrajectory(argv[2], model);
125  pTrajectoryWriter traj_out = createOutputTrajectory(argv[3]);
126 
127  bool box_override = false;
128  GCoord newbox;
129 
130  if (argc == 7) {
131  // use the supplied box, set it in model
132  double xbox = atof(argv[4]);
133  double ybox = atof(argv[5]);
134  double zbox = atof(argv[6]);
135  newbox = GCoord(xbox, ybox, zbox);
136  model.periodicBox(newbox);
137  box_override = true;
138  if (!traj->hasPeriodicBox())
139  {
140  cerr << "Adding box " << newbox << endl;
141  }
142  else
143  {
144  cerr << "WARNING - Overriding existing box(es) with " << newbox << endl;
145  }
146  }
147 
148  if (!traj->hasPeriodicBox() && !box_override)
149  {
150  // if the trajectory doesn't have periodic box info, we need it from
151  // the user
152  cerr << "ERROR - The trajectory has no box information. You must add it or supply it on the command-line.\n";
153  exit(-1);
154  }
155 
156  // duplicate the info from dcd in dcd_out
157  traj_out->setComments(hdr);
158 
159  // split the system by molecule
160  vector<AtomicGroup> molecules = model.splitByMolecule();
161  cerr << "Found " << molecules.size() << " molecules.\n";
162 
163  // split the system by segid
164  vector<AtomicGroup> segments = model.splitByUniqueSegid();
165  cerr << "Found " << segments.size() << " segments.\n";
166 
167  cerr << "Trajectory has " << traj->nframes() << " total frames.\n";
168 
169 
170  // Loop over the frames of the dcd and reimage each molecule
171  vector<AtomicGroup>::iterator m;
172  int frame_no = 0;
173  cerr << "Frames processed - ";
174  while (traj->readFrame())
175  {
176  if (++frame_no % update_frequency == 0)
177  {
178  cerr << frame_no << " ";
179  }
180 
181  traj->updateGroupCoords(model);
182 
183  if (box_override)
184  model.periodicBox(newbox);
185 
186  for (m = segments.begin(); m != segments.end(); m++)
187  {
188  m->reimage();
189  }
190  for (m = molecules.begin(); m != molecules.end(); m++)
191  {
192  m->reimage();
193  }
194 
195  traj_out->writeFrame(model);
196  }
197 
198  cerr << " - done\n";
199 
200 }
