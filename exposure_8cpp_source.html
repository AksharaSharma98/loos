61 string fullHelpMessage(void) {
62  string s =
63  "\n"
64  "SYNOPSIS\n"
65  "Calculate the density of various target selections about a probe selection\n"
66  "\n"
67  "DESCRIPTION\n"
68  "\tFor a given probe selection of atoms, exposure will calculate the count of target\n"
69  "atoms within a given shell (radius) of any probe atom. The output can be normalized by\n"
70  "the probe volume (i.e. a density), and can be averaged over all probe atoms.\n"
71  "\n"
72  "EXAMPLES\n"
73  "\n"
74  "\texposure simulation.pdb simulation.dcd 'segid == \"PROT\"'\n"
75  "Computes the solvent exposure for the molecule with segid \"PROT\".\n"
76  "\n"
77  "\texposure -P 'segid =~ \"^L\"' simulation.pdb simulation.dcd \\\n"
78  "\t 'resname == \"HEXO\" && segid == \"P1\"'\n"
79  "Computes the exposure of the residue HEXO with segid P1 to a\n"
80  "lipid membrane (assuming the lipids have segids begining with \"L\".\n"
81  "This could be used to determine the degree of insertion of the\n"
82  "residue into the membrane, for example.\n"
83  "\n"
84  "\texposure -R1 -P 'segid =~ \"^L\"' simulation.pdb simulation.dcd 'segid == \"P1\"'\n"
85  "Similar to above, except that it averages over the entire peptide\n"
86  "with segid P1 and considers periodic boundaries when determining\n"
87  "which atoms are within the probe shell.\n"
88  "\n"
89  "\texposure -R1 -I2 -P 'segid != \"BULK\"' simulation.pdb simulation.dcd 'segid == \"P1\"'\n"
90  "Computes the degree to which P1 is buried, i.e. the density of non-\n"
91  "water atoms about P1, excluding any atom that is within 2 A of an atom\n"
92  "in P1. Also considers periodic boundaries when computing distances.\n"
93  "\n"
94  "\texposure -P '!(segid == \"BULK\" || segid == \"P1\")' \\\n"
95  "\t simulation.pdb simulation.dcd 'segid == \"P1\"'\n"
96  "Computes the degree to which P1 is buried, ignoring the atoms from P1.\n"
97  "\n"
98  "NOTES\n"
99  "Exposure calculations can be quite lengthy for large systems/trajectories.\n"
100  "you may want to add '&& !hydrogen' to your selections if speed is an issue.\n"
101  "\n"
102  "This tool has been largely replaced by contact-time (which is faster)\n"
103  "\n"
104  "SEE ALSO\n"
105  "\tcontact-time\n";
106 
107  return(s);
108 }

144 double density(const AtomicGroup& target, const AtomicGroup& probe, const double inner_radius, const double outer_radius) {
145 
146  double or2 = outer_radius * outer_radius;
147  double ir2 = inner_radius * inner_radius;
148 
149  double vol_out = 4.0/3.0 * M_PI * or2 * outer_radius;
150  double vol_inn = 4.0/3.0 * M_PI * ir2 * inner_radius;
151  double vol = vol_out - vol_inn;
152 
153  GCoord box = target.periodicBox();
154  ulong contacts = 0;
155 
156  for (AtomicGroup::const_iterator j = target.begin(); j != target.end(); ++j) {
157  GCoord v = (*j)->coords();
158  ulong probed = 0;
159 
160  for (AtomicGroup::const_iterator i = probe.begin(); i != probe.end(); ++i) {
161  GCoord u = (*i)->coords();
162  double d = symmetry ? v.distance2(u, box) : v.distance2(u);
163  if (d >= ir2 && d <= or2) {
164  ++probed;
165  }
166  }
167  contacts += probed;
168  }
169 
170  double dens = static_cast<double>(contacts);
171  if (average)
172  dens /= target.size();
173  if (normalize)
174  dens /= vol;
175  return(dens);
176 }

181 int main(int argc, char *argv[]) {
182  string hdr = invocationHeader(argc, argv);
183 
184  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
185  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
186  ToolOptions* topts = new ToolOptions;
187  opts::RequiredArguments* ropts = new opts::RequiredArguments;
188  ropts->addVariableArguments("target", "target-selection");
189 
190  opts::AggregateOptions options;
191  options.add(bopts).add(tropts).add(topts).add(ropts);
192  if (!options.parse(argc, argv))
193  exit(-1);
194 
195  verbosity = bopts->verbosity;
196 
197  AtomicGroup model = tropts->model;
198  pTraj traj = tropts->trajectory;
199  indices = tropts->frameList();
200 
201  target_selections = ropts->variableValues("target");
202 
203  AtomicGroup probe = selectAtoms(model, probe_selection);
204 
205  vGroup targets;
206  for (vector<string>::iterator i = target_selections.begin(); i != target_selections.end(); ++i)
207  targets.push_back(selectAtoms(model, *i));
208 
209  cout << "# " << hdr << endl;
210  cout << "# frame ";
211  for (uint i=0; i < target_selections.size(); ++i)
212  cout << "Density_" << i << "\t";
213  cout << endl;
214 
215  ulong t = 0;
216 
217  PercentProgressWithTime watcher;
218  ProgressCounter<PercentTrigger, EstimatingCounter> slayer(PercentTrigger(0.1), EstimatingCounter(indices.size()));
219  slayer.attach(&watcher);
220  if (verbosity)
221  slayer.start();
222 
223  for (vector<uint>::iterator frame = indices.begin(); frame != indices.end(); ++frame) {
224  traj->readFrame(*frame);
225  traj->updateGroupCoords(model);
226 
227  if (symmetry && !model.isPeriodic()) {
228  cerr << "ERROR - the trajectory must be periodic to use --reimage\n";
229  exit(-1);
230  }
231 
232  cout << boost::format("%8d") % t;
233 
234  for (vGroup::const_iterator i = targets.begin(); i != targets.end(); ++i) {
235  double d;
236  d = density(*i, probe, inner_cutoff, outer_cutoff);
237  cout << boost::format(" %8.6f") % d;
238  }
239  cout << endl;
240  ++t;
241  if (verbosity)
242  slayer.update();
243  }
244 
245  if (verbosity)
246  slayer.finish();
247 }
