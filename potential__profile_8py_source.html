26 def read_file(filename):
27  file = open(filename)
28  v = []
29  z_vals = []
30  comments = []
31  for line in file.xreadlines():
32  if line.startswith("#"):
33  comments.append(line)
34  continue
35  fields = line.split()
36  z = float(fields[0])
37  temp = map(float, fields[1:])
38  v.append(temp)
39  z_vals.append(z)
40  v = numpy.array(v)
41  z_vals = numpy.array(z_vals)
42  return z_vals, v, comments

47  import sys
48 
49  if (len(sys.argv) > 1 and sys.argv[1] == "--fullhelp"):
50  print("""
51 SYNOPSIS
52 
53 Compute electrostatic potential along membrane normal
54 
55 DESCRIPTION
56 
57 This program generates the electrostatic potential profile for a membrane
58 system given a data file containing the charge density as a function of position
59 along the membrane normal. Intended to post-process the output of the
60 density-dist tool, it takes input in electrons/Ang^3 and outputs the potential
61 in volts.
62 
63 The algorithm used here is described by Sachs, et al, J Chem Phys, 2004, 121,
64 10847. In particular, the periodicity correction is applied, such that the top
65 and bottom of the periodic box must have the same electrostatic potential.
66 However, this correction is applied _only_ to the full charge density, not the
67 charge density of each individual component (assuming the user made selections
68 when running density-dist). This is because only the total potential needs to
69 be continuous at the potential, not that due to any given component.
70 
71 EXAMPLE
72 
73 potential_profile.py is intended to be used in combination with the LOOS
74 tool density-dist. For example:
75 
76 density-dist --type=charge -- path/to/model-file path/to/trajectory -38 38 76 > charge-density.dat
77 potential_profile.py charge-density.dat > potential.dat
78 
79 The "--" in the first line tells LOOS to stop interpreting arguments with
80 leading minus signs as flags; otherwise, it will choke on the "-38" (run
81 density-dist --fullhelp for more discussion).
82 
83 The first column of the output is the z-value, the second is the periodicity
84 correction (see above), and the subsequent columns are the electrostatic
85 potentials for the full system and any individual components selected when
86 density-dist was run.
87 
88  """)
89  sys.exit(1)
90  elif (len(sys.argv) != 2):
91  print("Usage: ", sys.argv[0], " filename")
92  print(" where filename is presumed to be output from density-dist")
93  sys.exit(1)
94 
95  datafilename = sys.argv[1]
96  z_vals, data, comments = read_file(datafilename)
97 
98  # assume input units are angstroms and electron charge,
99  # convert to V
100  eps0 = 8.85e-12 # C/V m
101  e_to_C = 1.60e-19 # C/e
102  ang_to_m = 1e-10
103 
104  # dielectric charge density integrated distance
105  units = 1/eps0 * e_to_C/(ang_to_m**3) * ang_to_m**2
106 
107  pot = numpy.zeros(data.shape, numpy.float)
108  # compute the potential, uncorrected for periodicity
109  for i in range(len(data)):
110  for j in range(i+1, len(data)):
111  dz = z_vals[i] - z_vals[j]
112  pot[j] += dz*data[i]
113 
114  # compute the periodicity correction for the first column only!!!
115  zrange = z_vals[-1] - z_vals[0]
116  corr = z_vals.copy()
117  delta_p = pot[-1][0] - pot[0][0]
118  corr *= -delta_p/zrange
119 
120  pot = pot.swapaxes(0, 1)
121  pot[0] += corr
122  # try recentering in the middle: can help if there are artifacts at the
123  # upper edge
124  pot[0] -= pot[0][len(pot[0])//2]
125 
126  pot = pot.swapaxes(0, 1)
127 
128  # convert to units of volts
129  pot *= units
130  corr *= units
131 
132  # output the result
133  print("# ", " ".join(sys.argv))
134  print("#Generated from: ")
135  print("#".join(comments),)
136  print("#z\tCorrection\tPotentials")
137 
138  for i in range(len(z_vals)):
139  s = " ".join(map(str, pot[i].tolist()))
140  print(z_vals[i], corr[i], s)
