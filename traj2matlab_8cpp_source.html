46 string fullHelpMessage(void) {

77 int main(int argc, char *argv[]) {
78  string hdr = invocationHeader(argc, argv);
79 
80  if (argc != 4) {
81  cerr << "Usage: " << argv[0] << " model trajectory selection" << endl;
82  cerr << fullHelpMessage();
83  exit(-1);
84  }
85 
86  AtomicGroup model = createSystem(argv[1]);
87  pTraj traj = createTrajectory(argv[2], model);
88  AtomicGroup subset = selectAtoms(model, argv[3]);
89 
90  Matrix A(subset.size() * 3, traj->nframes());
91  for (uint i = 0; i<traj->nframes(); ++i) {
92  traj->readFrame(i);
93  traj->updateGroupCoords(subset);
94 
95  for (uint j=0, k=0; j<subset.size(); ++j) {
96  GCoord c = subset[j]->coords();
97  A(k++,i) = c.x();
98  A(k++,i) = c.y();
99  A(k++,i) = c.z();
100  }
101  }
102 
103  writeAsciiMatrix(cout, A, hdr);
104 }
