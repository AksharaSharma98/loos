75 // Extractor is the base-class (interface) for how we extract atoms to
76 // compute torsions for. It uses NVI to make the extract function
77 // polymorphic while allowing pre- and post-conditions implemented in
78 // the base-class. For more details about NVI,
79 // see http://en.wikibooks.org/wiki/More_C%2B%2B_Idioms/Non-Virtual_Interface

83 class Extractor {
84 public:
85 
86  Extractor() : missing_atoms_warn(false), skip_when_missing(false), show_atoms(false), verbosity_(0) { }
87  virtual ~Extractor() { }
88 
89  void verbosity(uint v) { verbosity_ = v; }
90  uint verbosity() const { return(verbosity_); }
91 
92  // Configuration...
93  // Warn if atoms are missing and a torsion cannot be calculated...
94  void missingAtomsWarn() { missing_atoms_warn = true; }
95 
96  // If atoms are missing, return an empty vGroup so this residue will
97  // be skipped in future calculations...
98  void skipMissingResidues() { skip_when_missing = true; }
99 
100  // Show which atoms are being used for what torsion
101  void showAtoms() { show_atoms = true; }
102 
103  // Returns the names of the torsions
104  virtual vector<string> names(void) =0;
105 
106  // Utility function to make sure each AtomicGroup has only 4 atoms
107  // for a torsion
108  void checkAtoms(const string& s, const AtomicGroup& nascent, const AtomicGroup& residue) {
109  if (show_atoms && nascent.size() == 4) {
110  cerr << "Extracted residues for torsion " << s << endl;
111  cerr << nascent << endl;
112  }
113 
114  if (!missing_atoms_warn)
115  return;
116 
117  if (nascent.size() != 4) {
118  cerr << boost::format("Warning- unable to determine %s from resid %d, segid '%s'\n")
119  % s
120  % residue[0]->resid()
121  % residue[0]->segid();
122 
123  if (verbosity_ > 0) {
124  cerr << "Residue dump:\n";
125  cerr << residue << endl;
126  cerr << "Extracted:\n";
127  cerr << nascent << endl;
128  }
129  }
130 
131  return;
132  }
133 
134 
135  // Front-end to atom extraction. This is used to provide a
136  // post-condition where an empty vGroup is returned if not all
137  // torsions could be calculated for the ith residue (and the user
138  // wants us to skip it)...
139  vGroup extractAtoms(vGroup& residues, const int i) {
140  vGroup result = extractImpl(residues, i);
141 
142  if (skip_when_missing) {
143  vGroup::iterator vi;
144  for (vi = result.begin(); vi != result.end(); ++vi)
145  if ((*vi).size() != 4) {
146  if (show_atoms)
147  cerr << "***SKIPPING PREVIOUS RESIDUE***\n";
148  return(vGroup());
149  }
150  }
151 
152  return(result);
153  }
154 
155 
156 protected:
157 
158 
159  // This is used by the derived classes to make sure that going out
160  // of bounds on the vector index isn't fatal...
161 
162  AtomicGroup& getResidue(vGroup& residues, const int i) {
163 
164  if (i < 0 || i >= static_cast<int>(residues.size())) {
165  static AtomicGroup null_group;
166  return(null_group); // Yes, returning a handle to internal
167  // data is generally a bad idea, but we
168  // really need to return a ref, alors...
169  }
170 
171  return(residues[i]);
172  }
173 
174 private:
175 
176  // Polymorphic extractAtoms function... This is what changes to
177  // provide more types of torsions to extract.
178  virtual vGroup extractImpl(vGroup&, const int) =0;
179 
180 
181  bool missing_atoms_warn;
182  bool skip_when_missing;
183  bool show_atoms;
184  uint verbosity_;

191 class PhiPsi : public Extractor {
192 public:
193 
194  virtual ~PhiPsi() { }
195 
196  // Names, i.e. phi and psi
197  vector<string> names(void) {
198  vector<string> s;
199 
200  s.push_back("phi");
201  s.push_back("psi");
202  return(s);
203  }
204 
205 
206  // Implementation of function to extract the appropriate atoms for
207  // phi & psi...
208 
209  vGroup extractImpl(vGroup& residues, const int i) {
210 
211  // Select specific atom types
212  AtomNameSelector carbon("C");
213  AtomNameSelector nitrogen("N");
214  AtomNameSelector calpha("CA");
215 
216  // Pull out the atoms we'll use. Since AtomicGroup::select() only
217  // returns an AtomicGroup, we use this even though we're actually
218  // dealing with a single atom in the group...
219  AtomicGroup C_prev = getResidue(residues, i-1).select(carbon);
220  AtomicGroup N = getResidue(residues, i).select(nitrogen);
221  AtomicGroup CA = getResidue(residues, i).select(calpha);
222  AtomicGroup C = getResidue(residues, i).select(carbon);
223  AtomicGroup N_succ = getResidue(residues, i+1).select(nitrogen);
224 
225  // Now build an AtomicGroup corresponding to the atoms used to
226  // calculate phi
227  AtomicGroup phi = C_prev;
228  phi.append(N);
229  phi.append(CA);
230  phi.append(C);
231  checkAtoms("phi", phi, getResidue(residues, i));
232 
233  AtomicGroup psi = N;
234  psi.append(CA);
235  psi.append(C);
236  psi.append(N_succ);
237  checkAtoms("psi", psi, getResidue(residues, i));
238 
239  // Combine the two AtomicGroup's into a vector & return this...
240  vGroup dihe;
241  dihe.push_back(phi);
242  dihe.push_back(psi);
243 
244  return(dihe);
245  }
246 };

258 
259  vector<string> names(void) {

262  s.push_back("eta");
263  s.push_back("theta");
264  return(s);
265  }

268 
269  vGroup extractImpl(vGroup& residues, const int i) {
270 
271  // Select specific atom types
272  AtomNameSelector C4P("C4'");
273  AtomNameSelector P("P");
274 
275  // Extract all the atoms we'll use to build the psuedo-torsions
276  AtomicGroup c4p_prev = getResidue(residues, i-1).select(C4P);
277  AtomicGroup p = getResidue(residues, i).select(P);
278  AtomicGroup c4p = getResidue(residues, i).select(C4P);
279  AtomicGroup p_succ = getResidue(residues, i+1).select(P);
280  AtomicGroup c4p_succ = getResidue(residues, i+1).select(C4P);
281 
282  // Assemble the group representing eta
283  AtomicGroup eta = c4p_prev;
284  eta.append(p);
285  eta.append(c4p);
286  eta.append(p_succ);
287  checkAtoms("eta", eta, getResidue(residues, i));
288 
289  AtomicGroup theta = p;
290  theta.append(c4p);
291  theta.append(p_succ);
292  theta.append(c4p_succ);
293  checkAtoms("theta", theta, getResidue(residues, i));
294 
295  // Combine the groups into a vector
296  vGroup dihe;
297  dihe.push_back(eta);
298  dihe.push_back(theta);
299 
300  return(dihe);
301  }
302 };

432 char classifySecondaryStructure(double phi, double psi) {
433 
434  if (phi == missing_flag || psi == missing_flag)
435  return('?');
436 
437  if ( ( (psi < 0.0 && psi > -60) && phi <= -40 )
438  || ( (psi > 25 && psi <= 90) && (phi >= 45 && phi <= 65) ) )
439  return('H');
440 
441  if ( psi >= 90 && phi <= -45 )
442  return('S');
443 
444  return('O');
445 }

452 int main(int argc, char *argv[]) {
453 
454  string hdr = invocationHeader(argc, argv);
455 
456  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
457  opts::BasicSelection* sopts = new opts::BasicSelection;
458  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
459  ToolOptions* topts = new ToolOptions;
460 
461  opts::AggregateOptions options;
462  options.add(bopts).add(sopts).add(tropts).add(topts);
463  if (!options.parse(argc, argv))
464  exit(-1);
465 
466  extractor->verbosity(bopts->verbosity);
467 
468  // Read-in/setup objects to access data...
469  AtomicGroup model = tropts->model;
470  pTraj traj = tropts->trajectory;
471  AtomicGroup subset = selectAtoms(model, sopts->selection);
472 
473  vector<uint> indices = tropts->frameList();
474 
475 
476 
477  // Data-structure here is a vector of vectors of AtomicGroups. Each
478  // AtomicGroup holds the list of atoms to use for the torsion calc.
479  // Each inner vector of AtomicGroups describes the torsions to
480  // calculate for each residue/nucleotide. The outer-vector is the
481  // list of all residues/nucleotides to operate over...
482  vvGroup torsion_atoms;
483 
484  vGroup chains = subset.splitByUniqueSegid();
485  for (vGroup::iterator chain = chains.begin(); chain != chains.end(); ++chain) {
486 
487  // Split selection into individual residues to operate over...
488  vGroup residues = chain->splitByResidue();
489 
490  for (int i=0; i<static_cast<int>(residues.size()); ++i) {
491  vGroup atoms = extractor->extractAtoms(residues, i);
492  if (!atoms.empty())
493  torsion_atoms.push_back(atoms);
494  }
495  }
496 
497  // Verify SS option
498  vector<string> torsion_names = extractor->names();
499  if (torsion_names.size() != 2 ||
500  !(torsion_names[0] == "phi" && torsion_names[1] == "psi")) {
501  cerr << "Error- Secondary structure assignment can only be used with phi/psi torsions.\n";
502  exit(-10);
503  }
504 
505 
506  cout << "# " << hdr << endl;
507  if (topts->ss_flag)
508  cout << "# Secondary Structure Codes: H = Helix, S = Sheet, O = Other, ? = Undefined\n";
509 
510  cout << "# frame\t" << setw(10);
511 
512  // Construct the header of what torsions were computed...
513  copy(torsion_names.begin(), torsion_names.end(), ostream_iterator<string>(cout, "\t"));
514  if (topts->ss_flag)
515  cout << "SS";
516  cout << endl;
517 
518  uint t = 0;
519  // Iterate over the requested frames from the trajectory...
520  for (vector<uint>::iterator frameno = indices.begin(); frameno != indices.end(); ++frameno) {
521  traj->readFrame(*frameno);
522 
523  // Update ALL atoms...since atoms are shared between AtomicGroups,
524  // updating the parent group will update all of the groups we
525  // built to calculate torsions...
526  traj->updateGroupCoords(model);
527 
528  // Iterate over each residue...
529  vvGroup::iterator vvi;
530  for (vvi = torsion_atoms.begin(); vvi != torsion_atoms.end(); ++vvi) {
531  cout << t << " ";
532 
533  // Iterate over each group of atoms to use for torsions within
534  // the residue...
535  vGroup::iterator vi;
536  vector<double> torsions;
537 
538  for (vi = (*vvi).begin(); vi != (*vvi).end(); ++vi) {
539  double angle = missing_flag;
540  if ((*vi).size() == 4)
541  angle = Math::torsion((*vi)[0], (*vi)[1], (*vi)[2], (*vi)[3]);
542  cout << setw(10) << angle << " ";
543  torsions.push_back(angle);
544  }
545 
546  if (topts->ss_flag) {
547  // double-check
548  if (torsions.size() != 2) {
549  cerr << "Error- secondary structure requested but incorrect number of torsions found.\n";
550  exit(-10);
551  }
552  cout << classifySecondaryStructure(torsions[0], torsions[1]);
553  }
554  cout << endl;
555  }
556 
557  ++t;
558  }
