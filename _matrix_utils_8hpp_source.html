50  template<class T1, class P1, template<typename> class S1,

52  void copyMatrix(Matrix<T1,P1,S1>& A, const Matrix<T2,P2,S2>& M) {
53  uint m = M.rows();
54  uint n = M.cols();
55  Matrix<T1,P1,S1> B(m, n);
56 
57  for (uint j=0; j<m; ++j)
58  for (uint i=0; i<n; ++i)
59  B(j,i) = M(j, i);
60 
61  A = B;
62  }

72  template<class T1, class P1, class T2, class P2>
73  void copyMatrix(Matrix<T1,P1,SparseArray>& A, const Matrix<T2,P2,SparseArray>& M) {
74  uint m = M.rows();
75  uint n = M.cols();
76  static T2 null_value;
77  Matrix<T1,P1,SparseArray> B(m, n);
78 
79  for (uint j=0; j<m; ++j)
80  for (uint i=0; i<n; ++i) {
81  T2 t = M(j, i);
82  if (t != null_value)
83  B(j, i) = t;
84  }
85  A = B;
86  }
