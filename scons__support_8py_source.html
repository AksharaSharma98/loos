50 # Detect CYGWIN & canonicalize linux type, setting defaults...
51  if (re.search("(?i)cygwin", loos_build_config.host_type)):
52  loos_build_config.host_type = 'Cygwin'
53  loos_build_config.suffix = 'dll.a'
54  elif (loos_build_config.host_type == 'Linux'):
55  # Determine linux variant...
56  linux_type = platform.platform()
57 
58  if (re.search("(?i)ubuntu", linux_type)):
59  linux_type = 'ubuntu'
60  elif (re.search("(?i)suse", linux_type)):
61  linux_type = 'suse'
62  elif (re.search("(?i)debian", linux_type)):
63  linux_type = 'debian'
64  elif (re.search("(?i)centos", linux_type)):
65  linux_type = 'centos'
66  elif (re.search("(?i)fedora", linux_type)):
67  linux_type = 'fedora'
68 
69  loos_build_config.linux_type = linux_type
70 
71  # MacOS is special (of course...)
72  elif (loos_build_config.host_type == 'Darwin'):
73  loos_build_config.suffix = 'dylib'
74 
75  typemsg = loos_build_config.host_type
76  if (typemsg == 'Linux'):

264 # Check for existince of boost library with various naming variants
265 # Will return a tuple containing the correct name and a flag indicating
266 # whether this is the threaded or non-threaded version.
267 # This will only search specified paths, not the built-in paths for g++,
268 # so some libraries may be missed...
269 def CheckForBoostLibrary(conf, name, path, suffix):
270  conf.Message('Checking for Boost library %s...' % name)
271  name = 'boost_' + name
272 
273  def sortByLength(w1,w2):
274  return len(w1)-len(w2)
275 
276  # Now check for names lib libboost_regex-gcc43-mt.so ...
277  files = glob.glob(os.path.join(path, 'lib%s*-mt.%s' % (name, suffix)))
278  files.sort(cmp=sortByLength)
279  if files:
280  conf.Result(name + '-mt')
281  name = os.path.basename(files[0])[3:-(len(suffix)+1)]
282  return(name, 1)
283 
284  files = glob.glob(os.path.join(path, 'lib%s*.%s' % (name, suffix)))
285  files.sort(cmp=sortByLength)
286  if files:
287  conf.Result(name)
288  name = os.path.basename(files[0])[3:-(len(suffix)+1)]
289  return(name, 0)
290 
291 
292  conf.Result('missing')
293  return('', -1)

594 def AutoConfiguration(env):
595  global default_lib_path
596 
597  conf = env.Configure(custom_tests = { 'CheckForSwig' : CheckForSwig,
598  'CheckBoostHeaders' : CheckBoostHeaders,
599  'CheckForBoostLibrary' : CheckForBoostLibrary,
600  'CheckBoostHeaderVersion' : CheckBoostHeaderVersion,
601  'CheckDirectory' : CheckDirectory,
602  'CheckAtlasRequires' : CheckAtlasRequires,
603  'CheckForIEC559' : CheckForIEC559,
604  'CheckSystemType' : CheckSystemType,
605  'CheckNumpy' : CheckNumpy
606  })
607 
608  use_threads = int(env['threads'])
609 
610  # Get system information
611  conf.CheckSystemType()
612 
613  conf.env['host_type'] = loos_build_config.host_type
614  conf.env['linux_type'] = loos_build_config.linux_type
615 
616 
617  if env.GetOption('clean') or env.GetOption('help'):
618  env['HAS_NETCDF'] = 1
619  else:
620  has_netcdf = 0
621 
622 
623  # Some distros use /usr/lib, others have /usr/lib64.
624  # Check to see what's here and prefer lib64 to lib
625  default_lib_path = '/usr/lib'
626  if not conf.CheckDirectory('/usr/lib64'):
627  if not conf.CheckDirectory('/usr/lib'):
628  print('Fatal error- cannot find your system library directory')
629  conf.env.Exit(1)
630  else:
631  # /usr/lib64 is found, so make sure we link against this (and not against any 32-bit libs)
632  default_lib_path = '/usr/lib64'
633  conf.env.Append(LIBPATH = default_lib_path)
634  # Only setup ATLAS if we're not on a Mac...
635  if loos_build_config.host_type != 'Darwin':
636  atlas_libpath = ''
637  ATLAS_LIBPATH = env['ATLAS_LIBPATH']
638  ATLAS_LIBS = env['ATLAS_LIBS']
639  if not ATLAS_LIBPATH:
640  # Some distros may have atlas in /atlas-base, so must check for that...
641  if conf.CheckDirectory(default_lib_path + '/atlas-base'):
642  atlas_libpath = default_lib_path + '/atlas-base'
643  elif conf.CheckDirectory(default_lib_path + '/atlas'):
644  atlas_libpath = default_lib_path + '/atlas'
645  else:
646  print('Warning: Could not find an atlas directory! Winging it...')
647  else:
648  atlas_libpath = ATLAS_LIBPATH
649  loos_build_config.user_libdirs['ATLAS'] = atlas_libpath
650 
651  if atlas_libpath:
652  conf.env.Prepend(LIBPATH = [atlas_libpath])
653 
654  # Now that we know the default library path, setup Boost, NetCDF, and ATLAS
655  # based on the environment or custom.py file
656  SetupBoostPaths(conf.env)
657  SetupNetCDFPaths(conf.env)
658 
659 
660 
661  # Check for standard typedefs...
662  if not conf.CheckType('ulong','#include <sys/types.h>\n'):
663  conf.env.Append(CCFLAGS = '-DREQUIRES_ULONG')
664  if not conf.CheckType('uint','#include <sys/types.h>\n'):
665  conf.env.Append(CCFLAGS = '-DREQUIRES_UINT')
666 
667 
668  # Check for floating point format...
669  if not conf.CheckForIEC559():
670  print('Error- your system must use the IEC559/IEEE754 floating point')
671  print(' format for Gromacs support in LOOS. Check your compiler')
672  print(' options or contact the LOOS developers at')
673  print(' loos.maintainer@gmail.com')
674  conf.env.Exit(1)
675 
676  # --- NetCDF Autoconf
677  has_netcdf = 0
