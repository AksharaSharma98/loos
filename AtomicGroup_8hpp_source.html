953  // This function is to to remove code duplication in
954  // logisticContacts() and logisticContacts2D().
955  // Handle even and odd powers separately -- even can
956  // avoid the sqrt
957  // Sigh, this doesnt' seem to make it much faster...
958  double logisticFunc(const GCoord &cent, const GCoord &other, double radius, int sigma, const GCoord &box) const
959  {
960  double prod;
961  if (sigma % 2 == 0)
962  {
963  double distance2 = cent.distance2(other, box);
964  double ratio = distance2 / (radius * radius);
965  prod = ratio;
966  for (int j = 0; j < (sigma / 2) - 1; ++j)
967  {
968  prod *= ratio;
969  }
970  }
971  else
972  {
973  double distance = cent.distance(other, box);
974  double ratio = distance / radius;
975  prod = ratio;
976  for (int j = 0; j < sigma - 1; ++j)
977  {
978  prod *= ratio;
979  }
980  }
981  double sum = 1. / (1. + prod);
982  return (sum);
983  }
