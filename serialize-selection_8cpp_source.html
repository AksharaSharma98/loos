34 string fullHelpMessage(void) {
35  string msg =
36  "\n"
37  "SYNOPSIS\n"
38  "Convert a trajectory with N copies of a molecule into a longer one with only one copy\n"
39  "\n"
40  "DESCRIPTION\n"
41  "\tGiven a trajectory that is L frames long and with N copies of a molecule (selection),\n"
42  "this tool will write out a trajectory that is N*L frames long but with only one copy\n"
43  "of the selection. Each L-sized chunk of the trajectory contains a different copy\n"
44  "of the same molecule. For example, if a system has 100 frames and 4 identical peptides,\n"
45  "then the output trajectory will be 400 frames long. The first 100 frames will be\n"
46  "the first peptide, the next 100 frames will be the second peptide, etc. By default,\n"
47  "the different molecules will not be centered. The --center option will recenter\n"
48  "the entire trajectory.\n"
49  "\n"
50  "\tserialize-selection can also be used to make a library of conformations from a\n"
51  "trajectory. The --pdbout option turns this on. The prefix uses a printf format\n"
52  "for setting the output filenames. The default will be 'output-%06d.pdb'.\n"
53  "For membrane systems, it is convenient to canonicalize the orientation of the\n"
54  "resulting library. The --canon option will flip the selection depending on which\n"
55  "leaflet it came frame (assuming the membrane normal points along the Z-axis).\n"
56  "\n"
57  "EXAMPLES\n"
58  "\n"
59  "\tserialize-selection --selection 'segid =~ \"PEP.\"' --prefix serial model.psf sim.dcd\n"
60  "Extracts peptides with segid PEP0, PEP1, etc, into serial.pdb and serial.dcd\n"
61  "\n"
62  "\tserialize-selection --selection 'resname == \"POPC\"' --pdbout 'popc-%03d.pdb' --canon 1 model.psf sim.dcd\n"
63  "Extracts POPC residues, centering them (by default, using all atoms in the molecule)\n"
64  "and flipping them if they come from the lower leaflet. The output is a library of PDB\n"
65  "files named popc-000.pdb, popc-001.pdb, popc-002.pdb, etc.\n"
66  "\n"
67  "NOTES\n"
68  "\tThe selection will be split into molecules either using connectivity (if present)\n"
69  "or the segid.\n";
70 
71  return(msg);
72 }

138 /*
139  * Writes output as a set of PDBs (one per frame). The name of each
140  * PDB is determined by the output prefix. This object tracks the
141  * number of frames written and uses this to generate the output name.
142  */

144 class PDBOutput : public Outputter
145 {
146 public:
147  PDBOutput(const string& prefix, const string& hdr)
148  : Outputter(prefix, hdr), _count(0)
149  {
150 
151  // Check to see if the format string really generates unique names...
152  ostringstream ossa, ossb;
153  ossa << boost::format(prefix) % 1;
154  ossb << boost::format(prefix) % 2;
155 
156 
157  if (ossa.str() == ossb.str()) {
158  cerr << "Error- output prefix needs to be a printf-style format string\n"
159  " when using pdb output mode, e.g. 'foo%05d.pdb'.\n";
160  exit(-10);
161  }
162 
163  }
164 
165  void writeFrame(const AtomicGroup& structure)
166  {
167  PDB pdb = PDB::fromAtomicGroup(structure);
168  pdb.remarks().add(_hdr);
169 
170  ostringstream oss;
171  oss << boost::format(_prefix) % _count;
172  ofstream ofs(oss.str().c_str());
173  if (ofs.fail()) {
174  cerr << "Error- failed to open '" << oss.str() << "' for output.\n";
175  exit(-11);
176  }
177 
178  ofs << pdb;
179 
180  ++_count;
181  }
182 
183 private:
184  uint _count;
185 };

189 /*
190  * Writes the output as a Trajectory. The first frame that is passed is used
191  * to generate a PDB model. The prefix name is used to name the
192  * output files.
193  */
194 
195 class TrajOutput : public Outputter
196 {
197 public:
198 
199  TrajOutput(const string& prefix, const string& type, const bool append, const string& hdr, const bool renum)
200  : Outputter(prefix, hdr), _first_frame(true), _renum(renum)
201  {
202  _traj = createOutputTrajectory(prefix + "." + type, append);
203  _traj->setComments(hdr);
204  }
205 
206  void writeFrame(const AtomicGroup& structure)
207  {
208  if (_first_frame) {
209  AtomicGroup output_model = structure.copy();
210  if (_renum)
211  output_model.renumber();
212 
213  PDB pdb = PDB::fromAtomicGroup(output_model);
214  pdb.remarks().add(_hdr);
215  string pdbname = _prefix + ".pdb";
216  ofstream ofs(pdbname.c_str());
217  if (ofs.fail()) {
218  cerr << "Error- failed to open '" << pdbname << "' for output.\n";
219  exit(-11);
220  }
221 
222  ofs << pdb;
223 
224  _first_frame = false;
225  }
226  _traj->writeFrame(structure);
227  }
228 
229 private:
230  bool _first_frame;
231  bool _renum;
232  pTrajectoryWriter _traj;
233 };

241 int main(int argc, char *argv[])
242 {
243 
244  string hdr = invocationHeader(argc, argv);
245 
246  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
247  opts::OutputPrefix* popts = new opts::OutputPrefix;
248  opts::BasicSelection* sopts = new opts::BasicSelection("all");
249  opts::TrajectoryWithFrameIndices* tropts = new opts::TrajectoryWithFrameIndices;
250  opts::OutputTrajectoryTypeOptions* otopts = new opts::OutputTrajectoryTypeOptions;
251  ToolOptions* topts = new ToolOptions;
252 
253  opts::AggregateOptions options;
254  options.add(bopts).add(popts).add(sopts).add(tropts).add(otopts).add(topts);
255  if (!options.parse(argc, argv))
256  exit(-1);
257 
258  // Handle case where pdbout is on, but no prefix was given (i.e. using the default)
259  // Relies on knowledge of what the default value in OutputPrefix is, which we will
260  // want to use a better solution
261  if (topts->pdb_output && popts->prefix == "output")
262  popts->prefix = "output-%06d.pdb";
263 
264  AtomicGroup subset = selectAtoms(tropts->model, sopts->selection);
265 
266  vector<AtomicGroup> molecules;
267  if (tropts->model.hasBonds())
268  molecules = subset.splitByMolecule();
269  else
270  molecules = subset.splitByUniqueSegid();
271 
272  // Simple safety check for bad selections/connectivity
273  for (uint i=1; i<molecules.size(); ++i)
274  if (molecules[i].size() != molecules[0].size()) {
275  cerr << "Error- molecule #" << i << " has a different size than the first one.\n";
276  cerr << " Check your selection and try again.\n";
277  exit(-10);
278  }
279 
280  AtomicGroup outgroup = molecules[0].copy();
281 
282  // Figure out how to center
283  bool do_center = !topts->center_selection.empty();
284  AtomicGroup centering_subset = outgroup;
285  if (do_center)
286  centering_subset = selectAtoms(outgroup, topts->center_selection);
287 
288  // Set output type
289  Outputter* output = 0;
290  if (topts->pdb_output)
291  output = new PDBOutput(popts->prefix, hdr);
292  else
293  output = new TrajOutput(popts->prefix, otopts->type, otopts->append, hdr, topts->renum);
294 
295  vector<uint> frames = tropts->frameList();
296 
297 
298  for (uint k=0; k<molecules.size(); ++k) {
299  for (uint j=0; j<frames.size(); ++j) {
300  tropts->trajectory->readFrame(frames[j]);
301  tropts->trajectory->updateGroupCoords(tropts->model);
302 
303  for (uint i=0; i<outgroup.size(); ++i)
304  outgroup[i]->coords(molecules[k][i]->coords());
305 
306  if (topts->canonicalize) {
307  GCoord c = outgroup.centroid();
308  if (c.z() < 0.0) {
309  outgroup.translate(-c);
310  outgroup.rotate(GCoord(1,0,0), 180.0);
311  }
312  }
313 
314  if (do_center) {
315  GCoord c = centering_subset.centroid();
316  outgroup.translate(-c);
317  }
318 
319  output->writeFrame(outgroup);
320  }
321  }
322 
323 }
