104 string fullHelpMessage(void) {
105  string msg =
106  "\n"
107  "SYNOPSIS\n"
108  "\tApply arbitrary clipping planes to a model\n"
109  "\n"
110  "DESCRIPTION\n"
111  "\n"
112  "\tClipper implements a set of arbitrary clipping planes that can be\n"
113  "applied to a selection or to the entire model. When a selection is\n"
114  "used, only the selection is clipped--all other atoms are retained in\n"
115  "the output. Clipping planes are specified by providing three\n"
116  "coordinates. The normal to the plane is determined using the\n"
117  "right-hand rule (i.e. assuming the points define the plane in a\n"
118  "counter-clockwise fashion). Atoms that lie on the normal side of the\n"
119  "plane are clipped. Alternatively, if the --byres flag is given, then\n"
120  "if an atom is clipped, the entire residue that contains that atom is\n"
121  "also clipped regardless of where it lies with respect to the clipping\n"
122  "plane. Finally, any number of clipping planes can be specified on the\n"
123  "command line.\n"
124  "\n"
125  "EXAMPLES\n"
126  "\n"
127  "\tclipper model.pdb '(0,0,0)' '(1,0,0)' '(0,1,0)' >clipped.pdb\n"
128  "This defines a clipping plane at z=0 with the normal pointing\n"
129  "along the positive z-axis.\n"
130  "\n"
131  "\tclipper model.pdb '(0,4,0)' '(1,4,0)' '(0,4,1)' >clipped.pdb\n"
132  "This defines a clipping plane at y=4 with the normal pointing\n"
133  "along the positive y-axis\n"
134  "\n"
135  "\tclipper --byres --selection 'segid==\"BULK\"' model.pdb '(0,0,0)' '(1,0,0)' '(0,1,0)' >clipped.pdb\n"
136  "This defines a clipping plane at z=0 with the normal pointing\n"
137  "along the positive z-axis, but only waters are clipped and if any\n"
138  "water atom is clipped, then the entire water molecule is also\n"
139  "clipped.\n";
140 
141  return(msg);
142 }

147 void generateClippingPlanes(vector<GCoord>& planes, AtomicGroup& model, const string& sel) {
148 
149  AtomicGroup subset = selectAtoms(model, sel);
150  vector<GCoord> axes = subset.principalAxes();
151  GCoord center = subset.centroid();
152 
153  planes.push_back(center);
154  planes.push_back(center + axes[0]);
155  planes.push_back(center + axes[1]);
156 
157  cerr << "Automatically adding the following clipping plane:\n\t" << center << endl;
158  cerr << "\t" << center + axes[0] << endl;
159  cerr << "\t" << center + axes[1] << endl;
160 }

164 int main(int argc, char *argv[]) {
165 
166  string hdr = invocationHeader(argc, argv);
167  opts::BasicOptions* bopts = new opts::BasicOptions(fullHelpMessage());
168  opts::BasicSelection* sopts = new opts::BasicSelection;
169  opts::ModelWithCoords* mopts = new opts::ModelWithCoords;
170  ToolOptions* topts = new ToolOptions;
171 
172  opts::AggregateOptions options;
173  options.add(bopts).add(sopts).add(mopts).add(topts);
174  if (!options.parse(argc, argv))
175  exit(-1);
176 
177 
178 
179  AtomicGroup model = mopts->model;
180  AtomicGroup subset = selectAtoms(model, sopts->selection);
181 
182  if (!topts->auto_selection.empty())
183  generateClippingPlanes(planes, model, topts->auto_selection);
184  else
185  planes = topts->planes;
186 
187  // First, make sure all atoms are unflagged
188  for (AtomicGroup::iterator i = model.begin(); i != model.end(); ++i)
189  (*i)->clearProperty(Atom::flagbit);
190 
191  for (vector<GCoord>::iterator vi = planes.begin(); vi != planes.end();) {
192  GCoord x1 = *vi++;
193  GCoord x2 = *vi++;
194  GCoord x3 = *vi++;
195  GCoord n = (x2-x1) ^ (x3-x1);
196  n /= n.length();
197 
198  for (AtomicGroup::iterator i = subset.begin(); i != subset.end(); ++i) {
199  double d = n * ((*i)->coords() - x1);
200  if (d >= 0) {
201  if (topts->byresidue) {
202  AtomicGroup residue = subset.getResidue(*i);
203  for (AtomicGroup::iterator j = residue.begin(); j != residue.end(); ++j)
204  (*j)->setProperty(Atom::flagbit);
205  } else {
206  (*i)->setProperty(Atom::flagbit);
207  }
208  }
209  }
210  }
211 
212  AtomicGroup clipped;
213  if (topts->cliponly) {
214  for (AtomicGroup::iterator i = subset.begin(); i != subset.end(); ++i)
215  if (!(*i)->checkProperty(Atom::flagbit))
216  clipped.append(*i);
217 
218  } else {
219  for (AtomicGroup::iterator i = model.begin(); i != model.end(); ++i)
220  if (!(*i)->checkProperty(Atom::flagbit))
221  clipped.append(*i);
222  }
223 
224  PDB pdb = PDB::fromAtomicGroup(clipped);
225  pdb.remarks().add(hdr);
226  cout << pdb;
227 }
