35 
36 fullhelp = """
37 
38 Compute the frame-frame distance matrix for a trajetory in contact space.
39 
40 This program is analogous to rmsds, except the distance is computed using a
41 logistic contact function for each pair of residues. For each pair of residues,
42 the degree of contact is computed using a logistic switching function
43 C(r) = 1/(1 + (r/radius)^sigma
44 
45 So, each structure with N residues is represented by a vector with N*(N-1)/2
46 entries, each storing CC = 1-C(r). The distance between 2 structures is
47 computed as sqrt(sum CC). We then write out the full square matrix, suitable
48 for consumption by a clustering program such as our cluster-kgs.
49 
50 Note: the final matrix is MxM, where M is the number of frames in the trajectory
51 or trajetories, so it will get very big very fast (eg ~2.5 GB for 1000 frames).
52 So, if you've written a lot of frames, you'll probably want to downsample the
53 calculation using the --stride option.
54 
55 You can control the nature of the contact calculation using these flags:
56  --sigma: the exponent in the contact equation. Must be an integer,
57  defaults to 6. Larger values make the 1->0 transition sharper.
58  --radius: mid-point of the switching distance. The default is 6 ang,
59  which seems reasonable for sidechain-sidechain distances.
60 
61 The program uses the selection string supplied on the command line to decide
62 which residues to look at. For your convenience, we've added 2 additional
63 flags:
64  --skip_backbone: adds "!backbone" to the selection to remove the backbone of
65  protein, RNA, or DNA
66  --include_h: by default, the calculation uses only the heavy atoms, since
67  hydrogens don't generally shift the location of the centroid
68  appreciably. If you want the hydrogens, use this flag.
69 
70 By default, the matrix of distances is written to stdout, so you could in
71 principle stream it into something like cluster-kgs without saving it.
72 If you want to save it to a file, use the --outfile flag.
73 
74 TODO: add an option to write out the contact structures themselves, so someone
75 could input them into PCA, tICA, or something like that.
76 
77 """
78 
79 lo = options.LoosOptions("Compute distance matrix using contacts", fullhelp)
80 lo.modelSelectionOptions()
81 lo.trajOptions()
82 lo.add_argument('--sigma',
83  default=6,
84  type=int,
85  help="exponent for the logistic contact")
86 lo.add_argument('--radius',
87  default=5.0,
88  type=float,
89  help="midpoint distance for the logistic contact")
90 lo.add_argument('--skip_backbone',
91  default=False,
92  action='store_true',
93  help="Consider only sidechains")
94 lo.add_argument('--include_h',
95  default=False,
96  action='store_true',
97  help="Include hydrogens")
98 lo.add_argument('--outfile',
99  help="Name of outputted file ")
100 
101 args = lo.parse_args()
102 
103 system = loos.createSystem(args.model)
104 
105 chain = loos.selectAtoms(system, args.selection)
106 if args.skip_backbone:
107  chain = loos.selectAtoms(chain, '!backbone')
108 if not args.include_h:
109  chain = loos.selectAtoms(chain, '!hydrogen')
110 
111 residues = chain.splitByResidue()
112 num_residues = len(residues)
113 
114 # if there's 1 trajectory, args.traj will be a string, otherwise
115 # it's a list
116 if type(args.traj) == str:
117  t = loos.pyloos.Trajectory(args.traj, system,
118  skip=args.skip, stride=args.stride)
119  traj = loos.pyloos.VirtualTrajectory(t)
120 else:
121  t = loos.pyloos.Trajectory(args.traj[0], system,
122  skip=args.skip, stride=args.stride)
123  traj = loos.pyloos.VirtualTrajectory(t)
124  for t in args.traj[1:]:
125  t = loos.pyloos.Trajectory(t, system,
126  skip=args.skip, stride=args.stride)
127  traj.append(t)
128 
129 num_pairs = len(residues) * (len(residues)-1) // 2
130 contacts = numpy.zeros((len(traj), num_pairs), numpy.float)
131 
132 default_box = loos.GCoord(10000., 10000., 10000.)
133 frame_number = 0
134 for frame in traj:
135  # checking each frame is wasteful, but shouldn't have a measurable
136  # effect on performance
137  if frame.isPeriodic():
138  box = frame.periodicBox()
139  else:
140  box = default_box
141 
142  # compute residue-residue contacts and store
143  index = 0
144  for i in range(num_residues - 1):
145  r1 = residues[i]
146  for j in range(i+1, num_residues):
147  contact = r1.logisticContact(residues[j],
148  args.radius,
149  args.sigma,
150  box)
151  # High contact -> low distance
152  # However, since we're taking distances between structures, you
153  # get pretty much the same answer either way.
154  contacts[frame_number, index] = 1. - contact
155  index += 1
156  frame_number += 1
157 
158 
159 dists = pdist(contacts, 'euclidean')
160 square_dists = squareform(dists)
161 
162 if args.outfile:
163  outfile = open(args.outfile, "w")
164 else:
165  outfile = sys.stdout
166 
167 frame_boundaries = traj.frameBoundaries()
168 
169 traj_header = [" traj start end filename"]
170 if (type(args.traj) == str):
171  s = ["0", "0", str(len(traj)-1), args.traj]
172  traj_header.append("\t".join(s))
173 else:
174  for i in range(len(args.traj)-1):
175  s = [str(i), str(frame_boundaries[i]), str(frame_boundaries[i+1]-1),
176  args.traj[i]]
177  traj_header.append("\t".join(s))
178  i = len(args.traj) - 1
179  s = [str(i), str(frame_boundaries[i]), str(len(traj)-1),
180  args.traj[i]]
181  traj_header.append("\t".join(s))
182 traj_header = "\n".join(traj_header)
183 
184 header = "\n".join(str(x) for x in [lo.header(), traj_header])
185 
186 numpy.savetxt(outfile, square_dists, header=header)
