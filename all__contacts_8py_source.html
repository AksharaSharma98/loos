39  all_contacts.py: compute the probability of residue-residue contact

41 
42  Mandatory arguments:
43  model: file describing system contents, e.g. a psf or pdb
44  selection: selection string for which residues to look at
45  out_file: name for the average contact map, written in matlab format
46  traj: 1 or more trajectory files
47 
48  Options
49  --cutoff: distance for atom-atom contacts, defaults to 4.0 Ang
50  --no-hydrogens: ignore hydrogens when looking for contacts.
51  --no-backbone: ignore the protein or RNA backbone. If this flag isn't
52  given, you'll have a contact probability of 1 between
53  consecutive residues in a chain
54  --individual: if more than 1 trajectory was given, write probability
55  files for each one as well as the total. The files will
56  be written in the working directory, with the file name
57  the same as the trajectory file, replacing the extension with
58  ".dat", so "foo/bar/baz.dcd" would produce "./baz.dat".
59  This flag has no effect if only one trajectory was given
60  on the command line.
61  --fullhelp: produce this message
62 
63  This program does not explicitly handle periodicity; it assumes you've
64  already fixed any periodicity issues before you ran it.

68 lo = options.LoosOptions("Compute probability of residue-residue contacts",
69  fullhelp)
70 lo.modelSelectionOptions()
71 lo.trajOptions()
72 
73 
74 lo.add_argument('--out_file',
75  default='outfile',
76  help="File with the average contact occupancies")
77 lo.add_argument('--cutoff', type=float,
78  help="Cutoff distance for contact", default=4.0)
79 # TODO: add a number of contacts option

81  help="Don't include hydrogens")
82 lo.add_argument('--no_backbone', action='store_true',
83  help="Don't include the backbone")
84 lo.add_argument('--individual', action='store_true',
85  help="Write contact maps for each trajectory")
86 args = lo.parse_args()
87

89 header = lo.header()

91 
92 system = loos.createSystem(args.model)
93 all_trajs = []
94 out_names = []
95 num_trajs = len(args.traj)
96 for t in args.traj:
97  traj = loos.pyloos.Trajectory(t, system)
98  all_trajs.append(traj)
99  if (num_trajs > 1) and args.individual:
100  t_base = basename(t)
101  core, ext = splitext(t_base)
102  out_names.append(core + ".dat")
103 
104 if args.no_hydrogens:
105  no_hydrogens = loos.selectAtoms(system, "!hydrogen")
106  target = loos.selectAtoms(no_hydrogens, args.selection)
107 else:
108  target = loos.selectAtoms(system, args.selection)
109 
110 residues = target.splitByResidue()
111 # now remove the backbone -- doing before the split loses the glycines
112 if args.no_backbone:
113  residues = list([loos.selectAtoms(r, "!backbone") for r in residues])
114 
115 frac_contacts = numpy.zeros([len(residues), len(residues), num_trajs],
116  numpy.float64)
117 
118 
119 for traj_id in range(num_trajs):
120  traj = all_trajs[traj_id]
121  for frame in traj:
122  for i in range(len(residues)):
123  for j in range(i+1, len(residues)):
124  if residues[i].contactWith(args.cutoff, residues[j]):
125  frac_contacts[i, j, traj_id] += 1.0
126  frac_contacts[j, i, traj_id] += 1.0
127  frac_contacts[:, :, traj_id] /= len(traj)
128  if (num_trajs > 1) and args.individual:
129  numpy.savetxt(out_names[traj_id], frac_contacts[:, :, traj_id],
130  header=header)
131 
132 average = numpy.add.reduce(frac_contacts, axis=2)
133 average /= len(args.traj)
134 
135 numpy.savetxt(args.out_file, average, header=header)
