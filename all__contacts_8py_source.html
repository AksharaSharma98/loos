108 system = loos.createSystem(args.system_file)
109 all_trajs = []
110 out_names = []
111 num_trajs = len(args.traj_files)
112 for t in args.traj_files:
113  traj = loos.pyloos.Trajectory(t, system)
114  all_trajs.append(traj)
115  if (num_trajs > 1) and args.individual:
116  t_base = basename(t)
117  core, ext = splitext(t_base)
118  out_names.append(core + ".dat")
119 
120 if args.no_hydrogens:
121  no_hydrogens = loos.selectAtoms(system, "!hydrogen")
122  target = loos.selectAtoms(no_hydrogens, args.selection)
123 else:
124  target = loos.selectAtoms(system, args.selection)
125 
126 residues = target.splitByResidue()
127 # now remove the backbone -- doing before the split loses the glycines
128 # Woohoo, look at me, I used a lambda!
129 if args.no_backbone:
130  residues = list(map(lambda r: loos.selectAtoms(r, "!backbone"), residues))
131 
132 frac_contacts = numpy.zeros([len(residues), len(residues), num_trajs],
133  numpy.float)
134 
135 
136 for traj_id in range(num_trajs):
137  traj = all_trajs[traj_id]
138  for frame in traj:
139  for i in range(len(residues)):
140  for j in range(i+1, len(residues)):
141  if residues[i].contactWith(args.cutoff, residues[j]):
142  frac_contacts[i, j, traj_id] += 1.0
143  frac_contacts[j, i, traj_id] += 1.0
144  frac_contacts[:, :, traj_id] /= len(traj)
145  if (num_trajs > 1) and args.individual:
146  numpy.savetxt(out_names[traj_id], frac_contacts[:, :, traj_id],
147  header=header)
148 
149 average = numpy.add.reduce(frac_contacts, axis=2)
150 average /= len(args.traj_files)
151 
152 numpy.savetxt(args.out_file, average, header=header)
