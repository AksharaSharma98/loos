42  GCoord boxFromRemarks(const Remarks& r) {
43  int n = r.size();
44  int i;
45 
46  GCoord c(99999.99, 99999.99, 99999.99);
47 
48  for (i=0; i<n; i++) {
49  std::string s = r[i];
50  if (s.substr(0, 6) == " XTAL ") {
51  std::stringstream is(s.substr(5));
52  if (!(is >> c.x()))
53  throw(ParseError("Unable to parse " + s));
54  if (!(is >> c.y()))
55  throw(ParseError("Unable to parse " + s));
56  if (!(is >> c.z()))
57  throw(ParseError("Unable to parse " + s));
58 
59  break;
60  }
61  }
62 
63  return(c);
64  }
