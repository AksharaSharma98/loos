256  boost::tuple<AtomicGroup::iterator, AtomicGroup::iterator> AtomicGroup::calcSubsetIterators(const int offset, const int len) {
257  unsigned int a, b;
258 
259  if (offset < 0) {
260  b = atoms.size() + offset + 1;
261  a = (len == 0) ? 0 : b - len;
262  } else {
263  a = offset;
264  b = (len == 0) ? atoms.size() : a + len;
265  }
266 
267  if (b-a >= atoms.size())
268  throw(std::range_error("Indices out of bounds for subsetting"));
269 
270  boost::tuple<iterator, iterator> res(atoms.begin() + a, atoms.begin() + b);
271 
272  return(res);
273  }

319  std::vector<AtomicGroup> AtomicGroup::splitByUniqueSegid(void) const {
320  const_iterator i;
321  std::list<std::string> segids;
322

324  if (find(segids.begin(), segids.end(), (*i)->segid()) == segids.end())
325  segids.push_front((*i)->segid());
326 
327  // Reverse the order so that the chunks are in the same order they appear
328  // in the file...
329  uint n = segids.size();
330  std::vector<AtomicGroup> results(n);
331  uint j = n-1;
332  for (std::list<std::string>::const_iterator i = segids.begin(); i != segids.end(); ++i) {
333  SegidSelector segid(*i);
334  results[j--] = select(segid);
335  }
336 
337  std::vector<AtomicGroup>::iterator g;
338  for (g=results.begin(); g!=results.end(); g++) {
339  g->box = box;
340  }
341 
342  return(results);
343  }

346  const_iterator i;

348 
349  // Loop over atoms, adding them to groups based on their name, creating the
350  // map entry for each new name as we find it
351  std::map<std::string, AtomicGroup>::iterator g;
352  for (i = atoms.begin(); i != atoms.end(); ++i) {
353  g = groups.find((*i)->name());

355  AtomicGroup ag;

361  g->second.append(*i);
362  }
363 
364  }

389  std::vector<AtomicGroup> AtomicGroup::sortingSplitByMolecule(void) {
390  std::vector<AtomicGroup> molecules;
391 
392  // If no connectivity, just return the entire group...
393  if (!hasBonds()) {
394  sort();
395  molecules.push_back(*this);
396  } else {
397  HashInt seen; // Track what atoms we've already processed
398  AtomicGroup current; // The molecule we're currently building...
399  AtomicGroup working(*this); // Copy, so we can sort without mucking up original order
400  working.sort();
401 
402  int n = size();
403  for (int i=0; i<n; i++) {
404  HashInt::iterator it = seen.find(working.atoms[i]->id());
405  if (it != seen.end())
406  continue;
407 
408  walkBonds(current, seen, working, working.atoms[i]);
409  if (current.size() != 0) { // Just in case...
410  current.sort();
411  molecules.push_back(current);
412  current = AtomicGroup();
413  }
414 
415  }
416  }
417 
418 
419  // copy the box over
420  std::vector<AtomicGroup>::iterator m;
421  for (m=molecules.begin(); m!=molecules.end(); m++) {
422  m->box = box;
423  }
424  return(molecules);
425  }

469  void AtomicGroup::walkBonds(AtomicGroup& current, HashInt& seen, AtomicGroup& working, pAtom& moi) {
470  int myid = moi->id();
471  HashInt::iterator it = seen.find(myid);
472 
473  // If we've touched this atom before, stop recursing and return.
474  if (it != seen.end())
475  return;
476 
477  // This is a novel atom, so append it into the group we're currently building.
478  seen.insert(myid);
479  current.addAtom(moi);
480 
481  // Just in case it's a solo-atom... This probably should indicate
482  // some kind of error...?
483  if (!(moi->hasBonds()))
484  return;
485 
486  // Now find atoms that are bound to the current atom and recurse
487  // through them...
488 
489  std::vector<int> bonds = moi->getBonds();
490  std::vector<int>::const_iterator citer;
491  for (citer = bonds.begin(); citer != bonds.end(); citer++) {
492  pAtom toi = working.findById(*citer);
493  if (toi != 0)
494  walkBonds(current, seen, working, toi);
495  }
496  }

528  i->box = box;
529 
530  return(residues);

534  // Find an atom based on atomid
535  // Returns 0 (null shared_ptr) if not found...
536  pAtom AtomicGroup::findById_binarySearch(const int id) const {
537  int bottom = 0, top = size()-1, middle;
538 
539  while (top > bottom) {
540  middle = bottom + (top - bottom) / 2;
541  if (atoms[middle]->id() < id)
542  bottom = middle + 1;
543  else
544  top = middle;
545  }
546 
547  if (atoms[bottom]->id() == id)
548  return(atoms[bottom]);
549 
550  return(pAtom());
551  }

595  // Get all atoms associated with the residue that contains the
596  // passed atom... The returned atoms will not be in order. If
597  // you want that, then explicitly sort the group.
598  AtomicGroup AtomicGroup::getResidue(pAtom res) {
599  iterator i;
600  AtomicGroup result;
601 
602  result.box = box;
603  i = find(atoms.begin(), atoms.end(), res);
604  if (i == atoms.end())
605  return(result);
606 
607  iterator j = i;
608 
609  while (j >= atoms.begin()) {
610  if ((*j)->resid() == res->resid() && (*j)->segid() == res->segid())
611  result.addAtom(*j);
612  else
613  break;
614  --j;
615  }
616 
617  j = i+1;
618  while (j < atoms.end()) {
619  if ((*j)->resid() == res->resid() && (*j)->segid() == res->segid())
620  result.addAtom(*j);
621  else
622  break;
623 
624  ++j;
625  }
626 
627  return(result);
628  }

642  void renumberWithBonds(AtomicGroup& grp, const int start, const int stride) {
643  IMap bond_map;
644 
645  int id = start;
646  for (AtomicGroup::iterator i = grp.begin(); i != grp.end(); ++i, id += stride) {
647  bond_map[(*i)->id()] = id;
648  (*i)->id(id);
649  }
650 
651  // Now transform bond list
652  for (AtomicGroup::iterator i = grp.begin(); i != grp.end(); ++i, id += stride) {
653  if ((*i)->hasBonds()) {
654  std::vector<int> bonds = (*i)->getBonds();
655  for (std::vector<int>::iterator j = bonds.begin(); j != bonds.end(); ++j) {
656  IMap::iterator k = bond_map.find(*j);
657  if (k != bond_map.end())
658  *j = k->second;
659  }
660  (*i)->setBonds(bonds);
661  }
662  }
663  }

812  bool AtomicGroup::operator==(const AtomicGroup& rhs) const {
813  if (size() != rhs.size())
814  return(false);
815 
816  if (this == &rhs)
817  return(true);
818 
819  const std::vector<pAtom> *lp;
820  const std::vector<pAtom> *rp;
821  std::vector<pAtom> lhs_atoms, rhs_atoms;
822  CmpById comp;
823  if (!sorted()) {
824  lhs_atoms = atoms;
825  std::sort(lhs_atoms.begin(), lhs_atoms.end(), comp);
826  lp = &lhs_atoms;
827  } else

851  for (auto i = 0; i < atoms.size() - offset - 1; i++){
852  // compute bond vector between atom and its next neighbor.
853  bv1 = atoms[i]->coords() - atoms[i+1]->coords();
854  // compute bond vector between offset atom and its next neighbor.
